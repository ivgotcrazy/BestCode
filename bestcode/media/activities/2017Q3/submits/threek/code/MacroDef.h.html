#ifndef _MACRODEF_H__
#define _MACRODEF_H__

#include <algorithm>

#define ONITEMCALLBACK(msg, map)\
		{\
			std::find_if(map.begin(), map.end(), [&](std::pair<tstring, FMUICallback> pair)->bool{\
				if (pair.first == msg.pSender->GetName().GetData() &&\
					pair.second != nullptr)\
				{\
					pair.second(msg.ptMouse);\
					return true;\
				}\
				return false;\
			});\
		}

#ifndef SAFE_DELETE
#define SAFE_DELETE( x )    \
    if ( x )                \
	        {                       \
	delete x;           \
	x = NULL;           \
	    }
#endif

#define UPDATE_WNDLAYOUT()\
		{\
			m_pm.Invalidate();\
			m_pm.NeedUpdate();\
			m_pm.GetRoot()->NeedUpdate();\
			::SendMessage(GetHWND(), WM_PAINT, 0, 0);\
		}

#endif
