#include "stdafx.h"
#include "AttributeHelper.h"

namespace fm_uimodule 
{
	void SetControlAttr(TiXmlElement* xmlElement, const TagCtrlAttr& attr)
	{
		if (!attr.name.empty())
			xmlElement->SetAttribute("name", UnicodeToAnsi(attr.name).c_str());

		if (!attr.text.empty())
			xmlElement->SetAttribute("text", UnicodeToAnsi(attr.text).c_str());

		if (!attr.textcolor.empty())
			xmlElement->SetAttribute("textcolor", UnicodeToAnsi(attr.textcolor).c_str());

		if (!attr.padding.empty())
			xmlElement->SetAttribute("padding", UnicodeToAnsi(attr.padding).c_str());

		if (!attr.bkcolor.empty())
			xmlElement->SetAttribute("bkcolor", UnicodeToAnsi(attr.bkcolor).c_str());

		if (!attr.bkimage.empty())
			xmlElement->SetAttribute("bkimage", UnicodeToAnsi(attr.bkimage).c_str());

		//¥Û–° Œª÷√
		if (attr.width > 0)
			xmlElement->SetAttribute("width", attr.width);

		if (attr.height > 0)
			xmlElement->SetAttribute("height", attr.height);

		switch (attr.type)
		{
		case FMCT_TABHEADER:
		{
			TagTabAttr header = *(TagTabAttr*)(&attr);
			SetTabHeaderAttr(xmlElement, header);
		}
			break;
		case FMCT_BUTTON:
		{
			TagButtonAttr button = *(TagButtonAttr*)(&attr);
			SetButtonAttr(xmlElement, button);
		}
		default:
			break;
		}
	}

	void SetTabHeaderAttr(TiXmlElement* xmlElement, const TagTabAttr& attr)
	{
		if (!attr.groupname.empty())
			xmlElement->SetAttribute("group", UnicodeToAnsi(attr.groupname).c_str());

		if (!attr.selectedimg.empty())
			xmlElement->SetAttribute("selectedimage", UnicodeToAnsi(attr.selectedimg).c_str());

		xmlElement->SetAttribute("selected", attr.selected ? "true" : "false");
	}

	void SetButtonAttr(TiXmlElement* xmlElement, const TagButtonAttr& attr)
	{
		if (!attr.normalimg.empty())
			xmlElement->SetAttribute("normalimage", UnicodeToAnsi(attr.normalimg).c_str());

		if (!attr.hotimg.empty())
			xmlElement->SetAttribute("hotimage", UnicodeToAnsi(attr.hotimg).c_str());

		if (!attr.pushedimg.empty())
			xmlElement->SetAttribute("pushedimage", UnicodeToAnsi(attr.pushedimg).c_str());

		if (!attr.disabledimg.empty())
			xmlElement->SetAttribute("disabledimage", UnicodeToAnsi(attr.disabledimg).c_str());

		if (!attr.align.empty())
			xmlElement->SetAttribute("align", UnicodeToAnsi(attr.align).c_str());
	}

	void SetFrameCtrlAttr(TiXmlElement* xmlElement, const TagFrameCtrlAttr& attr, bool bChildAttr)
	{
		SetFrameBaseAttr(xmlElement, attr, bChildAttr);
	}

	void SetFrameHeaderAttr(TiXmlElement* xmlElement, const TagFMHeaderCtrl& attr, bool bChildAttr)
	{
		SetFrameBaseAttr(xmlElement, attr, bChildAttr);

		if (!bChildAttr)
		{
			if (attr.width > 0)
				xmlElement->SetAttribute("width", attr.width);

			if (attr.height > 0)
				xmlElement->SetAttribute("height", attr.height);

			xmlElement->SetAttribute("visible", attr.visible ? "true" : "false");
		}
		else
		{
			if (attr.width > 0)
				xmlElement->SetAttribute("width", attr.width);

			if (attr.height > 0)
				xmlElement->SetAttribute("height", attr.height);

			xmlElement->SetAttribute("visible", attr.visible ? "true" : "false");
		}
	}

	void SetTabViewAttr(TiXmlElement* xmlElement, const TagFMTabViewAttr& attr)
	{
		SetFrameCtrlAttr(xmlElement, attr);

		if (attr.pagewidth > 0)
			xmlElement->SetAttribute("pagewidth", attr.pagewidth);

		if (!attr.header.bkcolor.empty())
			xmlElement->SetAttribute("headerbkcolor", UnicodeToAnsi(attr.header.bkcolor).c_str());
		else if (!attr.header.bkimg.empty())
			xmlElement->SetAttribute("headerbkimg", UnicodeToAnsi(attr.header.bkimg).c_str());
		else if (attr.header.height > 0)
			xmlElement->SetAttribute("headerheight", attr.header.height);
	}

	void SetFrameBaseAttr(TiXmlElement* xmlElement, const TagFrameBaseAttr& attr, bool bChildAttr)
	{
		if (!bChildAttr)
		{
			if (!attr.bkimg.empty())
				xmlElement->SetAttribute("bkimage", UnicodeToAnsi(attr.bkimg).c_str());
			else if (!attr.bkcolor.empty())
				xmlElement->SetAttribute("bkcolor", UnicodeToAnsi(attr.bkcolor).c_str());
		}
		else
		{
			if (!attr.bkimg.empty())
				xmlElement->SetAttribute("bkimage", UnicodeToAnsi(attr.bkimg).c_str());
			else if (!attr.bkcolor.empty())
				xmlElement->SetAttribute("bkcolor", UnicodeToAnsi(attr.bkcolor).c_str());
		}
	}

}  // namespace fm_uimodule
