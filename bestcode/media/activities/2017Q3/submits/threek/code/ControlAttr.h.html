#ifndef _CONTROLATTR_H__
#define _CONTROLATTR_H__

namespace fm_uimodule {

    enum ENControlType 
	{
        FMCT_BASE = 0,
        FMCT_TABHEADER,
		FMCT_CONTAINER,
        FMCT_BUTTON,
		FMCT_TEXT
    };

    //基础控件属性
	struct TagCtrlAttr
	{
		ENControlType type;
		tstring name;
		tstring text;
		tstring textcolor;
		tstring padding;  //外边距

        tstring bkcolor;
        tstring bkimage;
     
        int width;
        int height;

		TagCtrlAttr(ENControlType ctype = FMCT_BASE) : 
			type(ctype), width(0), height(0)
        {
		}
    };

	struct TagButtonAttr : public TagCtrlAttr
	{
		tstring normalimg;
		tstring hotimg;
		tstring pushedimg;
		tstring disabledimg;
		tstring align;

		TagButtonAttr(void) :
			TagCtrlAttr(FMCT_BUTTON)
		{
		}
	};

	struct TagTabAttr : public TagButtonAttr
	{
        bool         selected;
        tstring      groupname;
        tstring      selectedimg;
        TagCtrlAttr  button;

        TagTabAttr(void) : selected(false)
        {
			this->type = FMCT_TABHEADER;
        }

        bool HasSubButton(void) const
        {
           return !button.name.empty();
        }
   };

	struct TagFunPanelCellAttr : public TagCtrlAttr
	{};

	struct TagStatusBarCellAttr : public TagCtrlAttr
	{
	};

	//框架控件属性
	struct TagFrameBaseAttr
	{
		tstring bkcolor;
		tstring bkimg;
	};

	struct TagFrameCtrlAttr : public TagFrameBaseAttr
	{
	};

	struct TagFMHeaderCtrl : public TagFrameBaseAttr
	{
		int  width;//tabview会用到
		int  height;
		bool visible;

		TagFMHeaderCtrl() :width(0), height(32), visible(true)
		{}
	};

	struct TagFMPopupViewAttr : public TagFrameCtrlAttr
	{
		TagButtonAttr   popup;
		TagButtonAttr   popback;
		TagFMHeaderCtrl header;
	};

	struct TagFMSplitContainerAttr : public TagFrameCtrlAttr
	{
		TagFMHeaderCtrl header;
	};

	struct TagFMStatusBarAttr : public TagFrameCtrlAttr
	{};

	struct TagFMModulePanelAttr : public TagFrameCtrlAttr
	{
		tstring panelbkimg;
		TagFMHeaderCtrl header;
	};

	struct TagFMTabViewAttr : public TagFrameCtrlAttr
	{
		int pagewidth;
		TagFMHeaderCtrl header;

		TagFMTabViewAttr() :pagewidth(32)
		{}
	};

	struct TagFMTabContainerAttr : public TagFMTabViewAttr
	{
		
	};

	
}
#endif
