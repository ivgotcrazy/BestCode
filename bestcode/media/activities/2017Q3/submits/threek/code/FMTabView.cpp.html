#include "stdafx.h"
#include "FMTabView.h"
#include "Control\FMTabLayoutUI.h"

namespace fm_uimodule
{
	CTabCreateCallback sTabCreateCallback;

	CFMTabView::CFMTabView()
	:WndModule(nullptr),
	 m_pTabLayout(nullptr),
	 m_wndFinalCallback(nullptr)
	{
		m_xmlfile = _T("<Window sizebox=\"4,4,6,6\">")
						_T("<FMTabLayout>")
							_T("<TabView>")
							_T("</TabView>")
						_T("</FMTabLayout>")
					_T("</Window>");

		SetNotify(this);
		SetBuilderCallback(&sTabCreateCallback);
	}

	CFMTabView::~CFMTabView()
	{
		m_vecHwnd.clear();
	}

	void CFMTabView::SetFinalCallback(FMWndCallback wndCallback)
	{
		if (wndCallback != nullptr)
			m_wndFinalCallback = wndCallback;
	}

	void CFMTabView::AddItem(const TagTabAttr& table, HWND hPageWnd)
	{
		if (::IsWindow(hPageWnd))
		{
			tstring strXmlFile = m_xmlfile.GetData();
			CFMItemXmlHelper::GetInstance()->AddTabItem(table, _T("FMTabLayout"), &strXmlFile);
			m_xmlfile = strXmlFile.c_str();

			m_vecHwnd.push_back(hPageWnd);
		}
	}

	void CFMTabView::RemoveItem(const tstring& strItemName)
	{
		if (m_pTabLayout == nullptr)
			return;

		unsigned nIndex = m_pTabLayout->RemoveItem(strItemName.c_str());
		if (nIndex >= 0 && nIndex < m_vecHwnd.size())
		{
			if (m_wndFinalCallback)
				m_wndFinalCallback(m_vecHwnd[nIndex]);

			m_vecHwnd.erase(m_vecHwnd.begin() + nIndex);
		}
	}

	void CFMTabView::RemoveItem(HWND hSubWnd)
	{
		if (m_pTabLayout == nullptr)
			return;

		int nIndex = 0;
		for (HWND hWnd : m_vecHwnd)
		{
			if (hWnd == hSubWnd)
				break;

			nIndex++;
		}

		m_pTabLayout->RemoveAt(nIndex);
	}

	void CFMTabView::SetTabViewAttr(const TagFMTabViewAttr& tvAttr)
	{
		tstring strXmlFile = m_xmlfile.GetData();
		CFMItemXmlHelper::GetInstance()->ModifyTabView(tvAttr, &strXmlFile);
		m_xmlfile = strXmlFile.c_str();
	}

	void CFMTabView::Notify(DuiLib::TNotifyUI& msg)
	{
		if (msg.sType == DUI_MSGTYPE_WINDOWINIT)
		{
			OnPrepare();
		}
		else if (msg.sType == DUI_MSGTYPE_SELECTCHANGED)
		{
			OnSelectChange();
		}
	}

	void CFMTabView::OnPrepare()
	{
		m_pTabLayout = static_cast<CFMTabLayoutUI*>(m_pm.GetRoot());
		if (m_pTabLayout == nullptr)
			return;

		UpdateSubWndPos();
		m_pTabLayout->Select(0);
	}

	void CFMTabView::OnSelectChange()
	{
		if (m_pTabLayout == nullptr)
			return;

		unsigned nIndex = m_pTabLayout->GetSelectItemIndex();
		if (nIndex < 0 || nIndex >= m_vecHwnd.size())
			return;

		for (unsigned i = 0; i < m_vecHwnd.size(); i++)
		{
			::ShowWindow(m_vecHwnd[i], i == nIndex ? SW_SHOW : SW_HIDE);
		}
	}

	//message
	LRESULT CFMTabView::OnSize(UINT uMsg, WPARAM wParam, LPARAM lParam, BOOL* bHandled)
	{
		UPDATE_WNDLAYOUT();
		UpdateSubWndPos();
		return __super::OnSize(uMsg, wParam, lParam, bHandled);
	}

	void CFMTabView::OnFinalMessage(HWND hWnd)
	{
		if (m_wndFinalCallback != nullptr)
		{
			for (HWND hWnd : m_vecHwnd)
			{
				if (::IsWindow(hWnd))
					m_wndFinalCallback(hWnd);
			}
		}
	}

	void CFMTabView::UpdateSubWndPos()
	{
		RECT rcView = m_pTabLayout->GetViewItemPos();
		int iIndex = 0;
		for (HWND hWnd : m_vecHwnd)
		{
			bool bVisible = m_pTabLayout->IsViewItemVisible(iIndex++);
			if (::IsWindow(hWnd))
			{
				SetChildWnd(hWnd, GetHWND());
				::ShowWindow(hWnd, bVisible ? SW_SHOW : SW_HIDE);

				::MoveWindow(hWnd, rcView.left, rcView.top,
					rcView.right - rcView.left,
					rcView.bottom - rcView.top, FALSE);
			}
		}
	}
}
