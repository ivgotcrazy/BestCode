		m_pOptionCtrl = new DuiLib::COptionUI();

	}

	{}

	LPVOID CFMTabHeaderUI::GetInterface(LPCTSTR pstrName)
	{
		if (_tcsicmp(pstrName, DUI_CTR_FMTABHEADER) == 0)
			return static_cast<CFMTabHeaderUI*>(this);
		else
			return __super::GetInterface(pstrName);
	}

	DuiLib::COptionUI* CFMTabHeaderUI::GetHeaderOpt()
	{
		return m_pOptionCtrl;
	}

	DuiLib::CButtonUI* CFMTabHeaderUI::GetHeaderBtn()
	{
		for (int i = 0; i < GetCount(); i++)
		{
			CControlUI* pItem = GetItemAt(i);
			if (pItem && pItem->GetInterface(DUI_CTR_BUTTON) != nullptr)
				return static_cast<DuiLib::CButtonUI*>(pItem);
		}

		return NULL;
	}

	bool CFMTabHeaderUI::Add(CControlUI* pControl)
	{
		if (pControl == NULL)
			return false;

		return __super::AddAt(pControl, 0);
	}

	/*************************** CFMTabLayout¿‡ ***********************************/
	CFMTabLayoutUI::CFMTabLayoutUI() :
		m_nFixedWidth(132),
		m_pTabViewLayout(NULL)
	{
		m_pTabHeaderLayout = new DuiLib::CHorizontalLayoutUI;
		if (m_pTabHeaderLayout != NULL)
		{
			m_pTabHeaderLayout->SetFixedHeight(32);
			__super::Add(m_pTabHeaderLayout);
		}
	}

	CFMTabLayoutUI::~CFMTabLayoutUI()
	{
	}

	bool CFMTabLayoutUI::AddItem(CControlUI* pControl)
	{
		if (pControl == nullptr)
			return false;

		bool bResult = false;
		if (pControl->GetInterface(DUI_CTR_FMTABHEADER) != nullptr)
		{
			bResult = AddHeader(  pControl);
		}
		else if (pControl->GetInterface(DUI_CTR_FMTABVIEW) != nullptr)
		{
			m_pTabViewLayout = static_cast<CFMTabViewUI*>(pControl);
			bResult = __super::Add(pControl);
		}

		return bResult;
	}
	
	bool CFMTabLayoutUI::AddItem2(DuiLib::CControlUI* pControl)
	{
		if (m_pTabViewLayout == nullptr)
			return false;

		m_pTabViewLayout->Add(pControl);
		return true;
	}

	int CFMTabLayoutUI::RemoveItem(DuiLib::CControlUI* pControl)
	{
		int nCtrlIndex = -1;
		
		do {
			if (pControl == nullptr || pControl->GetInterface(DUI_CTR_FMTABHEADER) == nullptr)
				break;

			if (m_pTabHeaderLayout == nullptr || m_pTabViewLayout == nullptr)
				break;

			for (int nIndex = 0; nIndex < m_pTabHeaderLayout->GetCount(); nIndex++)
			{
				if (m_pTabHeaderLayout->GetItemAt(nIndex) == pControl)
				{
					nCtrlIndex = nIndex;
					m_pTabHeaderLayout->RemoveAt(nIndex);
					m_pTabViewLayout->RemoveAt(nIndex);
			
					break;
				}
			}
		} while (false);

		return nCtrlIndex;
	}

	int CFMTabLayoutUI::RemoveItem(LPCTSTR pstrName)
	{
		int nCtrlIndex = -1;

		do {
			if (pstrName == nullptr)
				break;

			if (m_pTabHeaderLayout == nullptr || m_pTabViewLayout == nullptr)
				break;

			for (int nIndex = 0; nIndex < m_pTabHeaderLayout->GetCount(); nIndex++)
			{
				if (m_pTabHeaderLayout->GetItemAt(nIndex)->GetName() == pstrName)
				{
					nCtrlIndex = nIndex;
					m_pTabHeaderLayout->RemoveAt(nIndex);
					m_pTabViewLayout->RemoveAt(nIndex);

					break;
				}
			}
		} while (false);

		return nCtrlIndex;
	}

	void CFMTabLayoutUI::Select(int nIndex)
	{
		if (nIndex >= 0 && nIndex < m_pTabHeaderLayout->GetCount())
		{
			CFMTabHeaderUI* pTabHeader = static_cast<CFMTabHeaderUI*>(m_pTabHeaderLayout->GetItemAt(nIndex));
			if (pTabHeader && pTabHeader->GetHeaderOpt())
			{
				pTabHeader->GetHeaderOpt()->Activate();
			}
		}
	}

	RECT CFMTabLayoutUI::GetViewItemPos(void)
	{
		RECT rcView;
		::SetRectEmpty(&rcView);
		if (!m_pTabViewLayout)
			return rcView;

		for (int nIndex = 0; nIndex < m_pTabViewLayout->GetCount(); nIndex++)
		{
			CControlUI* pView = m_pTabViewLayout->GetItemAt(nIndex);
			if (pView && pView->IsVisible())
			{
				rcView = pView->GetPos();
				break;
			}
		}

		return rcView;
	}

	bool CFMTabLayoutUI::IsViewItemVisible(int nIndex)
	{
		CControlUI* pView = m_pTabViewLayout->GetItemAt(nIndex);
		if (pView)
		{
			return pView->IsVisible();
		}

		return false;
	}

	int CFMTabLayoutUI::GetSelectItemIndex()
	{
		if (m_pTabViewLayout == nullptr)
			return -1;

		return m_pTabViewLayout->GetCurSel();
	}

	void CFMTabLayoutUI::SetAttribute(LPCTSTR pstrName, LPCTSTR pstrValue)
	{
		if (_tcsicmp(pstrName, _T("headerbkimg")) == 0)
			SetHeaderAttribute(_T("bkimage"), pstrValue);
		else if (_tcsicmp(pstrName, _T("headerbkcolor")) == 0)
			SetHeaderAttribute(_T("bkcolor"), pstrValue);
		else if (_tcsicmp(pstrName, _T("headerheight")) == 0)
			SetHeaderAttribute(_T("height"), pstrValue);
		else if (_tcsicmp(pstrName, _T("pagewidth")) == 0)
		{
			m_nFixedWidth = ::_ttoi(pstrValue);
		}
		else
			__super::SetAttribute(pstrName, pstrValue);
	}

	void CFMTabLayoutUI::SetHeaderAttribute(LPCTSTR pstrName, LPCTSTR pstrValue)
	{
		if (m_pTabHeaderLayout != nullptr)
			m_pTabHeaderLayout->SetAttribute(pstrName, pstrValue);
	}

	bool CFMTabLayoutUI::AddHeader(CControlUI* pControl)
	{
		if (m_pTabHeaderLayout == nullptr)
			return false;

		if (pControl == nullptr || pControl->GetInterface(DUI_CTR_FMTABHEADER) == nullptr)
			return false;

		pControl->SetFixedWidth(m_nFixedWidth);

		CFMTabHeaderUI* pTabHeader = static_cast<CFMTabHeaderUI*>(pControl);
		pTabHeader->GetHeaderOpt()->OnNotify += MakeDelegate(this, &CFMTabLayoutUI::OnHeaderClick);

		m_pTabHeaderLayout->Add(pControl);
		return true;
	}

	bool CFMTabLayoutUI::OnHeaderClick(void* param)
	{
		DuiLib::TNotifyUI* pMsg = (DuiLib::TNotifyUI*)param;
		if (pMsg->sType == DUI_MSGTYPE_CLICK)
		{
			CControlUI* pSender = pMsg->pSender;
			if (pSender->GetInterface(DUI_CTR_OPTION) != nullptr)
			{
				CFMTabHeaderUI* pTabHeader = static_cast<CFMTabHeaderUI*>(pSender->GetParent());
				int iIndex = m_pTabHeaderLayout->GetItemIndex(pTabHeader);
				if (m_pTabViewLayout)
				{
					m_pTabViewLayout->SelectItem(iIndex);
				}
			}
		}

		return true;
	}
