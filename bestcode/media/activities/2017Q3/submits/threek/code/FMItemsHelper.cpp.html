	tstring CFMItemXmlHelper::AddNewTabItem(const TagTabAttr& table)
	{				
		TiXmlDocument* xmlDoc = new TiXmlDocument;

		TiXmlElement* pRoot = new TiXmlElement("Window");
		xmlDoc->LinkEndChild(pRoot);

		TiXmlElement* pTabHeader = new TiXmlElement("TabHeader");
		pRoot->LinkEndChild(pTabHeader);

		//设置header属性 
		SetControlAttr(pTabHeader, table);
		if (table.HasSubButton())
		{
			TiXmlElement* pButton = new TiXmlElement("Button");
			pTabHeader->LinkEndChild(pButton);

			//设置button属性
			SetControlAttr(pButton, table.button);
		}

		return ReleaseXmlDoc(pRoot);
	}

	void CFMItemXmlHelper::AddSplitItem(tstring* strXmlFile)

		TiXmlElement* pContainer = new TiXmlElement("VerticalLayout");
		pElem->LinkEndChild(pContainer);

		*strXmlFile = ReleaseXmlDoc(pRoot);
	}

	int CFMItemXmlHelper::AddStatusBarItem(const TagCtrlAttr& stb, tstring* strXmlFile)
	{
		int nIndex = -1;

		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return nIndex;

		TiXmlElement* pElem = pRoot->FirstChildElement("HorizontalLayout");
		if (pElem == nullptr)
			return nIndex;

		const CHAR* pName = pElem->Attribute("name");
		if (pName != nullptr && _stricmp(pName, "ItemsContainer") == 0)
		{
			string strControlType;

			if (stb.type == FMCT_CONTAINER)
			{
				strControlType = "Container";
			}
			else if (stb.type == FMCT_BUTTON)
			{
				strControlType = "Button";
			}
			else
			{
				strControlType = "Text";
			}

			if (!strControlType.empty())
			{
				CheckHolderItem(pElem, stb.padding.empty());

				TiXmlElement* pItem = new TiXmlElement(strControlType.c_str());
				SetControlAttr(pItem, stb);

				pElem->LinkEndChild(pItem);
				nIndex = FindItemIndex(pElem, pItem);

				CheckHolderItem(pElem, stb.padding.empty());
			}
		}

		*strXmlFile = ReleaseXmlDoc(pRoot);
		return nIndex;
	}

	void CFMItemXmlHelper::AddHeaderItem(const TagTabAttr& header, const tstring& strNodeName, tstring* strXmlFile)
	{
		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return;

		TiXmlElement* pElem = pRoot->FirstChildElement(UnicodeToAnsi(strNodeName).c_str());
		if (pElem == nullptr)
			return;

		TiXmlElement* pTabHeader = new TiXmlElement("TabHeader");
		SetControlAttr(pTabHeader, header);

		pElem->LinkEndChild(pTabHeader);

		if (header.HasSubButton())
		{
			TiXmlElement* pButton = new TiXmlElement("Button");
			pTabHeader->LinkEndChild(pButton);

			SetControlAttr(pButton, header.button);
		}

		*strXmlFile = ReleaseXmlDoc(pRoot);
	}

	void CFMItemXmlHelper::AddSubViewItem(const tstring strNodeName, tstring* strXmlFile)
	{
		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return;

		TiXmlElement* pElem = pRoot->FirstChildElement(UnicodeToAnsi(strNodeName).c_str());
		if (pElem == nullptr)
			return;

		TiXmlElement* pTabViewElem = pElem->FirstChildElement("TabView");
		if (pTabViewElem != nullptr)
		{
			TiXmlElement* pContainerElem = new TiXmlElement("Container");
			pTabViewElem->LinkEndChild(pContainerElem);
		}
		
		*strXmlFile = ReleaseXmlDoc(pRoot);
	}

	void CFMItemXmlHelper::AddToolBarItem(const TagButtonAttr& button, tstring* strXmlFile, bool bAddFront)
	{
		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return;

		TiXmlElement* pElem = pRoot->FirstChild("VerticalLayout")->FirstChild("HorizontalLayout")->ToElement();
		if (pElem == nullptr)
			return;

		const CHAR* pName = pElem->Attribute("name");
		if (pName != nullptr && _stricmp(pName, "ToolBar") == 0)
		{
			TiXmlElement* pChildElem = new TiXmlElement("Button");
			if (bAddFront)
				pElem->LinkEndChild(pChildElem);
			else
				pElem->InsertBeforeChild(pElem->FirstChildElement(), *pChildElem);

			SetControlAttr(pChildElem, button);
		}

		*strXmlFile = ReleaseXmlDoc(pRoot);
  	}

	void CFMItemXmlHelper::ModifyPopupViewToolBar(const TagFMPopupViewAttr& pv, tstring* strXmlFile)
	{
		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return;

		TiXmlElement* pFrame = pRoot->FirstChildElement("VerticalLayout");
		if (pFrame != nullptr)
			SetFrameCtrlAttr(pFrame, pv);

		TiXmlElement* pHori = pFrame->FirstChildElement("HorizontalLayout");
		if (pHori != nullptr)
		{
			const CHAR* name = pHori->Attribute("name");
			if (name != nullptr && _stricmp(name, "ToolBar") == 0)
			{
				SetFrameHeaderAttr(pHori, pv.header);
			}

			TiXmlElement* pButton = pHori->FirstChildElement("Button");
			while (pButton != nullptr)
			{
				const CHAR* name = pButton->Attribute("name");
				if (name != nullptr)
				{
					if (_stricmp(name, "Popup") == 0)
						SetControlAttr(pButton, pv.popup);
					else if (_stricmp(name, "Popback") == 0)
						SetControlAttr(pButton, pv.popback);
				}

				pButton = pButton->NextSiblingElement();
			}
		}

		*strXmlFile = ReleaseXmlDoc(pRoot);
	}

	void CFMItemXmlHelper::ModifySplitContainerToolBar(const TagFMSplitContainerAttr& sc, tstring* strXmlFile)
	{
		TiXmlElement* pRoot = GetRootElement(*strXmlFile);
		if (pRoot == nullptr)
			return;

		TiXmlElement* frame = pRoot->FirstChildElement("VerticalLayout");
		if (frame != nullptr)
			SetFrameCtrlAttr(frame, sc);

		TiXmlElement* toolbar = frame->FirstChildElement("HorizontalLayout");
		if (toolbar != nullptr)
		{
			const CHAR* name = toolbar->Attribute("name");
			if (name != nullptr && _stricmp(name, "ToolBar") == 0)
				SetFrameHeaderAttr(toolbar, sc.header, true);
		}

		*strXmlFile = ReleaseXmlDoc(pRoot);
	}

	TiXmlElement* GetRootElement(const tstring& strXmlFile)
	{
		TiXmlDocument* xmlDoc = new TiXmlDocument;

		CHAR szXmlBuf[1024] = { 0 };
		WBASELIB::ConvertUnicodeToAnsi(strXmlFile.c_str(), szXmlBuf, _countof(szXmlBuf));
		xmlDoc->Parse(szXmlBuf);
		if (xmlDoc->Error() != 0)
			return nullptr;

		TiXmlElement* root = xmlDoc->RootElement();
		return root;
	}
	
	tstring ReleaseXmlDoc(TiXmlElement* pRootElement)
	{
		TiXmlDocument* pXmlDoc = pRootElement->GetDocument();
		TIXML_OSTREAM stream;
		stream << *pXmlDoc;

		SAFE_DELETE(pXmlDoc);

		WCHAR szXmlResult[1024] = { 0 };
		WBASELIB::ConvertAnsiToUnicode(stream.c_str(), szXmlResult, _countof(szXmlResult));

		return tstring(szXmlResult);
	}


	void CFMItemXmlHelper::CheckHolderItem(TiXmlElement* pElem, bool bPadEmpty)
	{
		if (bPadEmpty)
		{
			TiXmlElement* pCtrl = new TiXmlElement("Control");
			pElem->LinkEndChild(pCtrl);
		}
	}

	int  CFMItemXmlHelper::FindItemIndex(TiXmlElement* pRoot, TiXmlElement* pElem)
	{
		if (pRoot == nullptr || pElem == nullptr)
			return -1;

		int nIndex = -1;
		TiXmlElement* pChild = pRoot->FirstChildElement();
		while (pChild)
		{
			nIndex++;

			if (pChild == pElem)
				break;

			pChild = pChild->NextSiblingElement();
		}

		return nIndex;
	}
