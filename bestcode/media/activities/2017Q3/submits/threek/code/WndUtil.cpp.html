#include "stdafx.h"

namespace fm_uimodule
{
	void SetChildWnd(HWND hWnd, HWND hParent)
	{
		if (!::IsWindow(hWnd) || !::IsWindow(hParent))
			return;

		int   nStyleOffset = GWL_STYLE;
		DWORD dwRemove = WS_OVERLAPPEDWINDOW;
		DWORD dwAdd = UI_WNDSTYLE_CHILD;

		DWORD dwStyle = ::GetWindowLong(hWnd, nStyleOffset);
		DWORD dwNewStyle = (dwStyle & ~dwRemove) | dwAdd;
		if (dwStyle != dwNewStyle)
			::SetWindowLong(hWnd, nStyleOffset, dwNewStyle);

		::SetParent(hWnd, hParent);
	}

	void SetPopupWnd(HWND hWnd)
	{
		if (!::IsWindow(hWnd))
			return;

		int   nStyleOffset = GWL_STYLE;
		DWORD dwRemove = UI_CLASSSTYLE_CHILD;
		DWORD dwAdd = WS_OVERLAPPEDWINDOW;

		DWORD dwStyle = ::GetWindowLong(hWnd, nStyleOffset);
		DWORD dwNewStyle = (dwStyle & ~dwRemove) | dwAdd;
		if (dwStyle != dwNewStyle)
			::SetWindowLong(hWnd, nStyleOffset, dwNewStyle);

		SetParent(hWnd, NULL);
	}

	string UnicodeToAnsi(const wstring& strValue)
	{
		CHAR szBuf[1024];
		WBASELIB::ConvertUnicodeToAnsi(strValue.c_str(), szBuf, _countof(szBuf));
		return string(szBuf);
	}
}
