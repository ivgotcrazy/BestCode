class BaseLog;
class MediaFactory
{
public: 
    
    MediaFactory();
    ~MediaFactory();

public:
	/*------------------------------------------------------------------------------
	* 描  述：初始化媒体工厂，同时初始化VA渲染器相关设备
	* 参  数：[in] i_log_ptr 外部日志指针 
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    static ERetCode InitFactory(BaseLog* i_log_ptr = nullptr);

	/*------------------------------------------------------------------------------
	* 描  述：反初始化媒体工厂
	* 参  数：无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    static ERetCode UnInitFactory();

	/*------------------------------------------------------------------------------
	* 描  述：创建数据源(根据参数创建软处理源或者硬处理源)
	* 参  数：[in] i_src_props_ptr 数据源属性
	* 参  数：[out] o_source_ptr   创建的数据源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    static ERetCode CreateSource(TSourceProps* i_src_props_ptr, ISource** o_source_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：销数据源
	* 参  数：[in] i_source_ptr 待销毁的数据源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode DestroySource(ISource** i_source_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：创建合成数据源(根据参数创建软处理源或者硬处理源)
	* 参  数：[in] i_src_props_ptr       合成数据源属性
	* 参  数：[out] o_mixer_source_ptr   创建的合成数据源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode CreateMixerSource(TSourceProps* i_src_props_ptr, IMixerSource** o_mixer_source_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：销毁合成数据源
	* 参  数：[in] i_mixer_source_ptr 待销毁的合成数据源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode DestroyMixerSource(IMixerSource** i_mixer_source_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：创建编码器(根据参数创建软编码器或者硬编码器)
	* 参  数：[in] i_enc_props_ptr 编码器属性
	* 参  数：[out] o_encoder_ptr  创建的编码器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    static ERetCode CreateEncoder(TEncInfo* i_enc_props_ptr, IEncoder** o_encoder_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：销毁编码器
	* 参  数：[in] i_mixer_source_ptr 待销毁的编码器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode DestroyEncoder(IEncoder **i_encoder_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：创建解码器
	* 参  数：[in] i_dec_props_ptr 解码器的属性
	* 参  数：[out] o_decoder_ptr  创建的解码器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode CreateDecoder(TSourceProps* i_dec_props_ptr, IDecoder** o_decoder_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：销毁解码器
	* 参  数：[in] i_decoder_ptr 待销毁的解码器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode DestroyDecoder(IDecoder** i_decoder_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：创建渲染器
	* 参  数：[in] i_render_props_ptr 渲染器的属性
	* 参  数：[out] o_decoder_ptr  创建的渲染器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode CreateRender(TMediaRenderProps* i_render_props_ptr, IVRender** o_render_ptr);

	/*------------------------------------------------------------------------------
	* 描  述：销毁渲染器
	* 参  数：[in] i_render_ptr 待销毁的渲染器
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	static ERetCode DestroyRender(IVRender** i_render_ptr);
    
private:

    static uint8_t m_is_create_Log;     // 创建日志标志
};
