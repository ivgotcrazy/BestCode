#pragma once

/**************************** 条件编译选项和头文件 ****************************/

#include <mutex>
#include "RetDefine.h"
#include "iencoder.h"

/**********************************  类声明  ***********************************/

class MediaEncoder : public IEncoder
{
public:
	virtual ~MediaEncoder(){};

	// 创建、并启动
	ERetCode CreateEncoder(TEncInfo* i_enc_props_ptr);

	// 编码数据
	ERetCode EncodeData(uint8_t* i_data_ptr, uint32_t i_data_len);

	// 设置编码数据回调对象
	ERetCode SetEncDataCallBack(IVCallBack* i_callback_ptr);

	//编码数据,o_pData输出或者回调接口回调
	ERetCode EncodeData(uint8_t* i_data_ptr, uint32_t i_data_len, TOutFrame* o_data_ptr = nullptr);

	// 设置输入源， 如果已存在，会覆盖原有的输入源
	ERetCode SetDataSource(ISource* i_source_ptr);

	// 销毁
	ERetCode DestroyEncoder();

private:
	IEncoder* m_encoder_ptr = nullptr;   //编码器
	std::mutex m_encoder_mtx;            //编码器锁
};
