#define PARTITION(ref, field, key, ascending)   \
    while (i <= j) {                            \
    if (ascending) {                        \
    while (ref[i].field < key)          \
    i++;                            \
    while (ref[j].field > key)          \
    j--;                            \
    } else {                                \
    while (ref[i].field > key)          \
    i++;                            \
    while (ref[j].field < key)          \
    j--;                            \
    }                                       \
    if (i <= j) {                           \
    tmp = ref[i];                       \
    ref[i] = ref[j];                    \
    ref[j] = tmp;                       \
    i++;                                \
    j--;                                \
    }                                       \
    }                                           \

enum EReadyState
{
	EM_RS_UNREADY,       // 不能使用，未创建img对象
	EM_RS_PRE_READY,      // 已创建img对象
	EM_RS_BUFFER,        // 数据缓存准备好， 等待传输
	EM_RS_UPDATING,      // 数据传输到显存中
	EM_RS_READY,         // 已完成传输，img准备好    
};
