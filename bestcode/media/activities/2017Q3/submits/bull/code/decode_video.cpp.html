/*------------------------------------------------------------------------------
* 描  述：创建解码器
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode DecodeVideo::CreateDecoder(TSourceProps* i_media_props_ptr)
{
    CHECK_POINTER(i_media_props_ptr, EM_RC_PARAM_ERR);

    std::unique_lock<std::mutex> lock(m_mtx_dec);
    if (!m_is_open_b)
    {
        AVCodecID codec_id = QueryDecoder(i_media_props_ptr->v_codec_id);
        if (codec_id == AV_CODEC_ID_NONE)
        {
            LOG_ERROR("Query The Decoder [%d] Failed", codec_id);
            return EM_RC_DEC_SUPPORT_ERR;
        }

        m_codec_ptr = avcodec_find_decoder(codec_id);
        if (m_codec_ptr == NULL) {
            LOG_ERROR("Could not find the Decoder id = %d", i_media_props_ptr->v_codec_id);
            return EM_RC_DEC_SUPPORT_ERR;
        }

        m_codec_ctx_ptr = avcodec_alloc_context3(m_codec_ptr);
        if (m_codec_ctx_ptr == NULL){
            LOG_ERROR("Could not alloc codec context id = %d", i_media_props_ptr->v_codec_id);
            return EM_RC_DEC_SUPPORT_ERR;
        }

        m_codec_ctx_ptr->extradata = new uint8_t[COMMON_SIZE_1K];
        m_codec_ctx_ptr->extradata_size = 0;

        //wmv9必须等到第一个帧头才能打开解码器
        //这个做法主要是为了应对会议客户端的编码器为MPEG4的时刻
        if (codec_id != AV_CODEC_ID_WMV3)
        {
            if (avcodec_open2(m_codec_ctx_ptr, m_codec_ptr, NULL) < 0) {
                LOG_ERROR("Could Not Open CodecID %d", i_media_props_ptr->v_codec_id);
                SAFE_DELETE(m_codec_ctx_ptr->extradata);
                avcodec_free_context(&m_codec_ctx_ptr);
                return EM_RC_DEC_SUPPORT_ERR;
            }
        }
        else
        {
            m_codec_ctx_ptr->width = i_media_props_ptr->src_size.width;
            m_codec_ctx_ptr->height = i_media_props_ptr->src_size.height;
            m_codec_ctx_ptr->bit_rate = 1024000;
            m_codec_ctx_ptr->pix_fmt = AV_PIX_FMT_YUV420P;
        }

        m_dec_frame.width = i_media_props_ptr->src_size.width;
        m_dec_frame.height = i_media_props_ptr->src_size.height;
        m_dec_frame.data_len = i_media_props_ptr->src_size.width * i_media_props_ptr->src_size.height * 3 / 2;

        m_data_buffer_ptr = new uint8_t[COMMON_SIZE_4M];
        m_frame_ptr = av_frame_alloc();
        memcpy(&m_video_props, i_media_props_ptr, sizeof(TSourceProps));
        m_is_open_b = true;

		LOG_INFO("Create Stream [%d] Decode[%d], Size[%dx%d]", i_media_props_ptr->user_data, codec_id,
			i_media_props_ptr->src_size.width, i_media_props_ptr->src_size.height);
    }

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：解码数据
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode DecodeVideo::DecodeData(uint8_t* i_data_ptr, int i_data_len, uint8_t** o_dec_data_ptr)
{
    std::unique_lock<std::mutex> lock(m_mtx_dec);
    if (!m_is_open_b)
        return EM_RC_INIT_ERR;

    uint8_t *data_ptr = i_data_ptr;
    int data_size = i_data_len;

    if (m_codec_ctx_ptr->codec_id == AV_CODEC_ID_WMV3)
    {
        if (!ReadyWMV9Codec(&data_ptr, &data_size))
            return EM_RC_DEC_FAILED;

        m_pri_changed_b = false;
        m_first_frame_b = false;
    }

    uint8_t* dec_data_ptr = nullptr;
    if (*o_dec_data_ptr == nullptr)
        dec_data_ptr = m_data_buffer_ptr;
    else
        dec_data_ptr = *o_dec_data_ptr;

    AVPacket avpkt = { 0 };
    av_init_packet(&avpkt);
    avpkt.data = const_cast<uint8_t *>(data_ptr);
    avpkt.size = data_size;

    int nRet = avcodec_decode_video2(m_codec_ctx_ptr, m_frame_ptr, &m_got_picture_flag, &avpkt);
    if (nRet < 0)
        return EM_RC_DEC_FAILED;

    if (m_got_picture_flag)
    {
        int i = 0;
        int uv_size = m_video_props.src_size.width * m_video_props.src_size.height / 4;
        for (i = 0; i < m_video_props.src_size.height; i++)
            memcpy(dec_data_ptr + i*m_video_props.src_size.width,
            m_frame_ptr->data[0] + i*m_frame_ptr->linesize[0], m_video_props.src_size.width);

        if (m_codec_ctx_ptr->codec_id == AV_CODEC_ID_WMV3)
        {
            for (i = 0; i < m_video_props.src_size.height / 2; i++)
                memcpy(dec_data_ptr + uv_size * 4 + i*m_video_props.src_size.width / 2,
                m_frame_ptr->data[2] + i*m_frame_ptr->linesize[1], m_video_props.src_size.width / 2);
            for (i = 0; i < m_video_props.src_size.height / 2; i++)
                memcpy(dec_data_ptr + uv_size * 5 + i*m_video_props.src_size.width / 2,
                m_frame_ptr->data[1] + i*m_frame_ptr->linesize[2], m_video_props.src_size.width / 2);
        }
        else
        {
            for (i = 0; i < m_video_props.src_size.height / 2; i++)
                memcpy(dec_data_ptr + uv_size * 5 + i*m_video_props.src_size.width / 2,
                m_frame_ptr->data[1] + i*m_frame_ptr->linesize[1], m_video_props.src_size.width / 2);
            for (i = 0; i < m_video_props.src_size.height / 2; i++)
                memcpy(dec_data_ptr + uv_size * 4 + i*m_video_props.src_size.width / 2,
                m_frame_ptr->data[2] + i*m_frame_ptr->linesize[2], m_video_props.src_size.width / 2);
        }

        if (*o_dec_data_ptr == nullptr && m_callback_ptr)
        {
            m_dec_frame.frame_data_ptr = dec_data_ptr;
            m_callback_ptr->OnDecodeCallBack(&m_dec_frame, m_video_props.user_data);
        }

        av_packet_unref(&avpkt);
        return EM_RC_OK;
    }

    av_packet_unref(&avpkt);
    return EM_RC_DEC_FAILED;
}

/*------------------------------------------------------------------------------
* 描  述：准备WMV9的解码环境
* 参  数：
* 返回值：true 成功/false 失败
-----------------------------------------------------------------------------*/
bool DecodeVideo::ReadyWMV9Codec(uint8_t** i_inbuf_ptr, int* io_size_ptr)
{
	uint8_t*inbuf_ptr = *i_inbuf_ptr;
    uint8_t  frame_type = inbuf_ptr[0];
    inbuf_ptr++;
    *io_size_ptr -= 1;
    uint8_t *extradata_ptr = NULL;
    int extradata_size = 0;
    if (m_codec_ptr == nullptr) {
        return false;
    }
    if (IFRAME_HEADER == frame_type)
    {
        extradata_size = *(uint32_t*)inbuf_ptr;
        inbuf_ptr += 4;
        *io_size_ptr -= 4;

        if (extradata_size > MAX_PRIVATEDATALEN){
            return false;
        }

        m_pri_changed_b = m_codec_ctx_ptr->extradata_size != extradata_size ||
            (extradata_size > 0 && memcmp(m_codec_ctx_ptr->extradata, inbuf_ptr, extradata_size) != 0);

        if (m_pri_changed_b)
        {
            m_codec_ctx_ptr->extradata_size = extradata_size;
            memcpy(m_codec_ctx_ptr->extradata, inbuf_ptr, extradata_size);
        }

        extradata_ptr = inbuf_ptr;
        inbuf_ptr += extradata_size;
        *io_size_ptr -= extradata_size;
    }

    //WMV9需要在第一帧数据时打开
    if (m_first_frame_b)
    {
        if (NULL == extradata_ptr)
            return false;

        if (0 != avcodec_open2(m_codec_ctx_ptr, m_codec_ptr, NULL)){
            LOG_ERROR("Open WMV9 Codec Failed");
            avcodec_free_context(&m_codec_ctx_ptr);
            m_codec_ctx_ptr = NULL;
            return false;
        }
    }
    else if (m_pri_changed_b)
    {
        avcodec_close(m_codec_ctx_ptr);
        if (0 != avcodec_open2(m_codec_ctx_ptr, m_codec_ptr, NULL))
        {
            LOG_ERROR("Open WMV9 Codec Failed");
            avcodec_free_context(&m_codec_ctx_ptr);
            m_codec_ctx_ptr = NULL;
            return false;
        }
    }
    return true;
}

/*------------------------------------------------------------------------------
* 描  述：销毁解码器
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode DecodeVideo::DestroyDecoder()
{
    std::unique_lock<std::mutex> lock(m_mtx_dec);
    m_is_open_b = false;
    if (m_data_buffer_ptr)
        SAFE_DELETE(m_data_buffer_ptr);
    if (m_frame_ptr)
        av_frame_free(&m_frame_ptr);
    if (m_codec_ctx_ptr)
    {
        SAFE_DELETE(m_codec_ctx_ptr->extradata);
        avcodec_close(m_codec_ctx_ptr);
        avcodec_free_context(&m_codec_ctx_ptr);
    }

    LOG_INFO("Destroy Stream [%d] Decoder", m_video_props.user_data);

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：AVCodec和内部解码器ID映射查询
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
AVCodecID DecodeVideo::QueryDecoder(EVideoCodecID i_codec_id)
{
    AVCodecID codec_id = AV_CODEC_ID_NONE;
    if (i_codec_id == EM_MT_H264)
        codec_id = AV_CODEC_ID_H264;
    else if (i_codec_id == EM_MT_VP8)
        codec_id = AV_CODEC_ID_VP8;
    else if (i_codec_id == EM_MT_WMV9)
        codec_id = AV_CODEC_ID_WMV3;
    else if (i_codec_id == EM_MT_VP9)
        codec_id = AV_CODEC_ID_VP9;
    else
        return AV_CODEC_ID_NONE;

    return codec_id;
}
