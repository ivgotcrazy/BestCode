#pragma once

/**************************** 条件编译选项和头文件 ****************************/

#include "iencoder.h"
#include "MediaConfig.h"
#include "BaseThread.h"
#include "MediaTool.h"


/**********************************  类声明  ***********************************/

class EncodeVideo : public IEncoder, public BaseThread
{
public:

	EncodeVideo(){};
	virtual ~EncodeVideo(){};

	// 创建、并启动
	ERetCode CreateEncoder(TEncInfo* i_enc_props_ptr);

	//编码数据（回调接口回调或者o_pData输出）
	ERetCode EncodeData(uint8_t* i_data_ptr, uint32_t i_data_len, TOutFrame* o_data_ptr = nullptr);

	// 设置编码数据回调对象
	ERetCode SetEncDataCallBack(IVCallBack* i_callback_ptr);

	// 设置输入源， 如果已存在，会覆盖原有的输入源
	ERetCode SetDataSource(ISource* i_source_ptr);

	// 关闭
	ERetCode DestroyEncoder();

private:
	ERetCode ThreadFuncProc();

private:

    AVCodec*				    m_codec_ptr = nullptr;         //编码器
    AVCodecContext*	            m_codec_ctx_ptr = nullptr;	   //编码器上下文
    AVFrame*				    m_frame_ptr = nullptr;         //编码帧
    uint8_t*                    m_frame_buffer_ptr = nullptr;  //编码帧缓冲

	IVCallBack*                 m_callBack_ptr = nullptr;      //编码数据回调对象
	ISource*                    m_source_ptr = nullptr;        //数据源

	MediaTool*                  m_media_tool_ptr = nullptr;    //媒体工具

	uint8_t*                    m_buffer_ptr = nullptr;        //缩放缓冲

	TOutFrame                   m_enc_frame;                   //编码帧
	TOutFrame                   m_yuv_frame;                   //YUV缓冲帧

    TEncInfo			        m_enc_props;	               //编码器属性
    int						    m_is_inited = 0;               //是否初始化
    int						    m_enc_count = 0;               //编码计数
};
