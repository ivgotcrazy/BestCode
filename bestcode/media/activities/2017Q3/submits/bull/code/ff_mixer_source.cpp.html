/*------------------------------------------------------------------------------
* 描  述：创建合成源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::CreateSource(TSourceProps* i_src_props_ptr)
{
    CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

    memcpy(&m_src_props, i_src_props_ptr, sizeof(TSourceProps));

    AllocDataBuffer();

    m_media_tool_ptr = new MediaTool(g_rbs_Log_ptr);
    m_soft_logo_ptr = new FFLogo;
    m_soft_logo_ptr->Init();
    m_soft_subtitle_ptr = new FFSubTitle;
    m_soft_subtitle_ptr->Init();
    m_dict_effect_ptr = new DirectEffect;

	LOG_INFO("Create MixerSource[%ld] Format[ %d ], Size[ %dx%d ]",
		m_src_props.user_data, m_src_props.format,
		m_src_props.src_size.width, m_src_props.src_size.height);

    return StartThread();
}

/*------------------------------------------------------------------------------
* 描  述：销毁合成源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::DestroySource()
{
    StopThread();

    ReleaseDataBuffer();

    SAFE_DELETE(m_media_tool_ptr);
    if (m_soft_logo_ptr)
    {
        m_soft_logo_ptr->UnInit();
        SAFE_DELETE(m_soft_logo_ptr);
    }

    if (m_soft_subtitle_ptr)
    {
        m_soft_subtitle_ptr->UnInit();
        SAFE_DELETE(m_soft_subtitle_ptr);
    }

    if (m_dict_effect_ptr)
        SAFE_DELETE(m_dict_effect_ptr);

    LOG_INFO("Destroy MixerSource [%d]", m_src_props.user_data);

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：设置字幕
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::SetSubTitle(TTextInfo *i_text_info_ptr)
{
    std::unique_lock<std::mutex> lock(m_filter_mtx);

    if (!i_text_info_ptr->is_time_changed)
    {
        if (i_text_info_ptr)
			memcpy(&m_text_info, i_text_info_ptr, sizeof(m_text_info));
        else
			memset(&m_text_info, 0, sizeof(m_text_info));

        m_filter_status_flag |= EM_FILTER_MODIFY_SUBTITLE;
    }

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：运行特效
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::RunEffect()
{
	//准备一份特效的底图
	{
		std::unique_lock<std::mutex> lock(m_mixer_mtx);
		ReadyEffectBG();
	}

    {
		 std::unique_lock<std::mutex> lock(m_filter_mtx);
		 m_dict_effect_ptr->SetEffect(m_effect_type);
		 m_dict_effect_ptr->SetDictSize(m_src_props.src_size.width, m_src_props.src_size.height);
		 m_dict_effect_ptr->UpdateLastDictImage(m_effect_bg_buffer);
    }

 return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：获取图像数据
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
VideoSrcImage FFMixerSource::GetReadyImage(TOutFrame* o_data_ptr)
{
	if (o_data_ptr == nullptr)
		return nullptr;

	if ( !m_modify_size_flag )
	{
		o_data_ptr->width = m_src_props.src_size.width;
		o_data_ptr->height = m_src_props.src_size.height;
		o_data_ptr->data_len = m_src_image_size;
		if (m_media_Ring_Buffer.GetData(o_data_ptr->frame_data_ptr, (uint32_t*)&o_data_ptr->data_len) == EM_RC_OK)
			return &m_media_Ring_Buffer;
	}

	return nullptr;
}

/*------------------------------------------------------------------------------
* 描  述：修改合成大小
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ModifyMixerSize(TSize* i_size_ptr)
{
	m_modify_size_flag = 1;
	{
		std::unique_lock<std::mutex> lock(m_mixer_mtx);
		//修改缩放器
		m_src_props.src_size = *i_size_ptr;
		m_src_image_size = RBS_IYUV_SIZE(m_src_props.src_size.width, m_src_props.src_size.height);

		LOG_WARN("Stream [%d] Modify Mixer Size Width:Height [%d:%d]", m_src_props.user_data, i_size_ptr->width, i_size_ptr->height);
	}
	m_modify_size_flag = 0;

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：分辨数据缓冲
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::AllocDataBuffer()
{
	m_buffer_size = RBS_IYUV_SIZE(COMMON_DEFAULT_WIDTH,COMMON_DEFAULT_HEIGHT);
    m_src_image_size = RBS_IYUV_SIZE(m_src_props.src_size.width,m_src_props.src_size.height);

    SAFE_DELETE(m_scale_buffer_ptr);
    m_scale_buffer_ptr = new uint8_t[m_buffer_size];

    SAFE_DELETE(m_read_buffer.frame_data_ptr);
    m_read_buffer.frame_data_ptr = new uint8_t[m_buffer_size];

	SAFE_DELETE(m_mixer_buffer_ptr);
	m_mixer_buffer_ptr = new uint8_t[m_buffer_size];
	
	SAFE_DELETE(m_effect_bg_buffer);
    m_effect_bg_buffer = new uint8_t[COMMON_SIZE_4M];

	m_media_Ring_Buffer.CreateRing(FPS_BUF_COUNT, m_buffer_size);

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：释放分辨数据缓冲
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ReleaseDataBuffer()
{
	if ( m_read_buffer.frame_data_ptr )
		SAFE_DELETE(m_read_buffer.frame_data_ptr);

	if (m_scale_buffer_ptr)
		SAFE_DELETE(m_scale_buffer_ptr);

	if (m_mixer_buffer_ptr)
		SAFE_DELETE(m_mixer_buffer_ptr);

	if (m_effect_bg_buffer)
		SAFE_DELETE(m_effect_bg_buffer);

	m_media_Ring_Buffer.DestroyRing();

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：添加数据源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::AddSource(int i_src_id, TProcItem* i_item_ptr)
{
    CHECK_POINTER(i_item_ptr, EM_RC_PARAM_ERR);
    std::unique_lock<std::mutex> lock(m_mixer_mtx);
    if (m_cur_src_count >= m_max_src_count)
    {
        LOG_ERROR("Reach Max VideoMixer Source Count[%d/%d]", m_cur_src_count, m_max_src_count);
        return EM_RC_SIZE_NOTENOUGH;
    }

    TMixerMember* mixer_member_ptr = new TMixerMember;
    mixer_member_ptr->tmp_ptr = new TMixerMember;
    mixer_member_ptr->rect = i_item_ptr->rect;
    mixer_member_ptr->source_ptr = i_item_ptr->source_ptr;
    mixer_member_ptr->level = i_item_ptr->level;
    mixer_member_ptr->mix_state = EM_SS_ACTIVING;

    m_mixer_map.insert(MapMixerMember::value_type(i_src_id, mixer_member_ptr));
    m_cur_src_count++;
    LOG_INFO("Add Source[%d] To MixerSource [%d], Level[%d], Pos[%d,%d], Size[%dx%d], Count[%d/%d] SourcePtr [%p]",
        i_src_id, m_src_props.user_data, mixer_member_ptr->level, mixer_member_ptr->rect.iX, mixer_member_ptr->rect.iY,
        mixer_member_ptr->rect.width, mixer_member_ptr->rect.height, m_cur_src_count, m_max_src_count, mixer_member_ptr->source_ptr);
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：删除数据源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::DelSource(int i_src_id)
{
    std::unique_lock<std::mutex> lock(m_mixer_mtx);
    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    TMixerMember* mixer_member_ptr = node_map->second;
    SAFE_DELETE(mixer_member_ptr->tmp_ptr);
    SAFE_DELETE(mixer_member_ptr);

    LOG_INFO("MixerSource[%d] Remove Source[%d]", m_src_props.src_id, i_src_id);
    m_mixer_map.erase(node_map);

    m_cur_src_count--;
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：修改源布局信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ModifySrcOutRect(int i_src_id, TRect* i_out_rect_ptr)
{
    CHECK_POINTER(i_out_rect_ptr, EM_RC_PARAM_ERR);

    std::unique_lock<std::mutex> lock(m_mixer_mtx);
    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    TMixerMember* mixer_mem_ptr = node_map->second;
    if (mixer_mem_ptr->mix_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    mixer_mem_ptr->tmp_ptr->rect = *i_out_rect_ptr;
    mixer_mem_ptr->mix_state |= EM_SS_MODIFY_POS;   // modify outrect                
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：修改数据源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ModifySrcSource(int i_src_id, ISource* i_source_ptr)
{
    CHECK_POINTER(i_source_ptr, EM_RC_PARAM_ERR);
    std::unique_lock<std::mutex> lock(m_mixer_mtx);

    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    // 修改source需要加锁，保证线程没有在执行合成
    TMixerMember* mixer_mem_ptr = node_map->second;
    if (mixer_mem_ptr->mix_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    mixer_mem_ptr->source_ptr = i_source_ptr;
    mixer_mem_ptr->mix_state = EM_SS_ACTIVING;
    LOG_INFO("MixerSource[%d] Modify Source[%d] Input Source Object!", m_src_props.src_id, i_src_id);
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：合成源业务线程
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ThreadFuncProc()
{
    uint64_t count = 0;
    uint64_t tick = stGetCurrentMicroSecond();
    uint64_t frame_timestamp = COMMON_TICK_BASE / COMMON_DEFAULT_FPS;
    if (m_src_props.fps)
        frame_timestamp = COMMON_TICK_BASE / m_src_props.fps;

    LOG_INFO("Start FFVideoMixerThread!");
    while (m_exit_code == EM_TS_RUN)
    {
        {
            m_mixer_mtx.lock();
            if (!m_cur_src_count)
            {
                m_mixer_mtx.unlock();
                usleep(COMMON_TICK_SECOND);
                continue;
            }
            m_mixer_mtx.unlock();
        }
  {
      std::unique_lock<std::mutex> lock(m_mixer_mtx);

      CheckFilterState();
      CheckSourceState();
      RunVideoMixerTask();
  }

  count++;
  while (stGetCurrentMicroSecond() - tick < count * frame_timestamp)
      usleep(5000);
    }

    m_exit_code = EM_TS_STOP;
    LOG_INFO("Stop VideoMixerThread!");
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：检查filter状态
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::CheckFilterState()
{
    std::unique_lock<std::mutex> lock(m_filter_mtx);
    if (m_filter_status_flag & EM_FILTER_MODIFY_SUBTITLE && m_soft_subtitle_ptr)
    {
        m_soft_subtitle_ptr->UnInit();
        m_soft_subtitle_ptr->Init();
        m_soft_subtitle_ptr->SetVideoSourceSize(m_src_props.src_size.width, m_src_props.src_size.height);
        int tmp = m_text_info.font_size;
        if (m_src_props.src_size.width < COMMON_TICK_SECOND) // 标清
            m_text_info.font_size = 24;
        else if (m_src_props.src_size.width < COMMON_DEFAULT_WIDTH) //高清
            m_text_info.font_size = 48;
        else//全高清
            m_text_info.font_size = 72;

        m_soft_subtitle_ptr->SetSubTitle(&m_text_info);
        m_text_info.font_size = tmp;
    }
    if (m_filter_status_flag & EM_FILTER_MODIFY_LOGO && m_soft_logo_ptr)
    {
        m_soft_logo_ptr->UnInit();
        m_soft_logo_ptr->Init();
        m_soft_logo_ptr->SetVideoSourceSize(m_src_props.src_size.width, m_src_props.src_size.height);
        m_soft_logo_ptr->SetLogo(&m_logo_info);
    }

    m_filter_status_flag = EM_FILTER_NONE;

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：检查数据源状态
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::CheckSourceState()
{
    TMixerMember* mixer_mem_ptr = nullptr;

    auto node_map = m_mixer_map.begin();
    for (; node_map != m_mixer_map.end(); node_map++)
    {
        mixer_mem_ptr = node_map->second;
        if (mixer_mem_ptr->mix_state == EM_SS_ACTIVING)
            continue;

        int iSrcID = node_map->first;
        if (mixer_mem_ptr->mix_state & EM_SS_MODIFY_LEVEL)
        {// 需要修改输出层级
            mixer_mem_ptr->level = mixer_mem_ptr->tmp_ptr->level;
            LOG_INFO("MixerSource[%d] Modify Source[%d] Level[%d]",
                m_src_props.src_id, iSrcID, mixer_mem_ptr->level);
        }
        if (mixer_mem_ptr->mix_state & EM_SS_MODIFY_POS)
        {// 需要修改输出位置
            mixer_mem_ptr->rect = mixer_mem_ptr->tmp_ptr->rect;
            LOG_INFO("MixerSource[%d] Modify Source[%d] Pos[%dx%d], Size[%dx%d]",
                m_src_props.src_id, iSrcID, mixer_mem_ptr->rect.iX, mixer_mem_ptr->rect.iY,
                mixer_mem_ptr->rect.width, mixer_mem_ptr->rect.height);
        }
        mixer_mem_ptr->mix_state = EM_SS_ACTIVING;
    }
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：准备特效背景
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::ReadyEffectBG()
{
    //准备一帧数据作为特效底图
    TMixerMember* mixer_mem_ptr = nullptr;
    for (auto node_map = m_mixer_map.begin(); node_map != m_mixer_map.end(); node_map++)
    {
        mixer_mem_ptr = node_map->second;
        //保存原来底部最后一帧数据
        LOG_WARN("Mix State [%d], level [%d]", mixer_mem_ptr->mix_state, mixer_mem_ptr->level);
        if (mixer_mem_ptr->mix_state & EM_SS_ACTIVING && mixer_mem_ptr->level == EM_VML_BOTTOM)
        {
            RBSMediaSource* source_ptr = reinterpret_cast<RBSMediaSource*>(mixer_mem_ptr->source_ptr);
            if (source_ptr->GetReadyImage(&m_read_buffer))
            {
                if (m_read_buffer.width != m_src_props.src_size.width || m_read_buffer.height != m_src_props.src_size.height)
                {
                    //对yuv进行缩放
                    m_media_tool_ptr->DirectVideoScale(m_read_buffer.width, m_read_buffer.height, m_read_buffer.frame_data_ptr,
                        mixer_mem_ptr->rect.width, mixer_mem_ptr->rect.height, m_effect_bg_buffer);
                }
                else
                    memcpy(m_effect_bg_buffer, m_read_buffer.frame_data_ptr,
                    RBS_IYUV_SIZE(m_src_props.src_size.width, m_src_props.src_size.height));
            }
        }
    }

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：执行合成任务
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::RunVideoMixerTask()
{
	TMixerMember* mixer_member_ptr = nullptr;
	RBSMediaSource* source_ptr = nullptr;

	auto node_map = m_mixer_map.begin();
	for (; node_map != m_mixer_map.end(); node_map++)
	{
		mixer_member_ptr = node_map->second;
		source_ptr = reinterpret_cast<RBSMediaSource*>(mixer_member_ptr->source_ptr);
		if (mixer_member_ptr->mix_state == EM_SS_ACTIVING && source_ptr != nullptr
			&& mixer_member_ptr->level == EM_VML_BOTTOM)
		{
			// 输入底层源image
			if (source_ptr->GetReadyImage(&m_read_buffer))
			{
				uint8_t *buffer_ptr = nullptr;
				if (mixer_member_ptr->rect.width != m_read_buffer.width || mixer_member_ptr->rect.height != m_read_buffer.height)
				{
					//对yuv进行缩放
					m_media_tool_ptr->DirectVideoScale(m_read_buffer.width, m_read_buffer.height, m_read_buffer.frame_data_ptr,
						mixer_member_ptr->rect.width, mixer_member_ptr->rect.height, m_scale_buffer_ptr);
					buffer_ptr = m_scale_buffer_ptr;
				}
				else
					buffer_ptr = m_read_buffer.frame_data_ptr;

				MixVideoTask({ m_src_props.src_size.width, m_src_props.src_size.height }, m_mixer_buffer_ptr,
					mixer_member_ptr->rect, buffer_ptr);
			}
		}
	}

	node_map = m_mixer_map.begin();
	for (; node_map != m_mixer_map.end(); node_map++)
	{
		mixer_member_ptr = node_map->second;
		source_ptr = reinterpret_cast<RBSMediaSource*>(mixer_member_ptr->source_ptr);
		if (mixer_member_ptr->mix_state == EM_SS_ACTIVING && mixer_member_ptr->source_ptr != nullptr
			&& mixer_member_ptr->level == EM_VML_TOP)
		{
			// 输入顶层源image
			if (source_ptr->GetReadyImage(&m_read_buffer) != nullptr)
			{
				uint8_t *buffer_ptr = nullptr;
				if (mixer_member_ptr->rect.width != m_read_buffer.width || mixer_member_ptr->rect.height != m_read_buffer.height)
				{
					//对yuv进行缩放
					m_media_tool_ptr->DirectVideoScale(m_read_buffer.width, m_read_buffer.height, m_read_buffer.frame_data_ptr,
						mixer_member_ptr->rect.width, mixer_member_ptr->rect.height, m_scale_buffer_ptr);
					buffer_ptr = m_scale_buffer_ptr;
				}
				else
					buffer_ptr = m_read_buffer.frame_data_ptr;

				MixVideoTask({ m_src_props.src_size.width, m_src_props.src_size.height },
					m_mixer_buffer_ptr, mixer_member_ptr->rect, buffer_ptr);
			}
		}
	}

	DoneFilter(m_mixer_buffer_ptr);

	m_media_Ring_Buffer.PutData(m_mixer_buffer_ptr,m_src_image_size);

	return EM_RC_OK;
}//NOLINT(readability/fn_size)

/*------------------------------------------------------------------------------
* 描  述：执行filter
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::DoneFilter(uint8_t* i_buffer_ptr)
{
    CHECK_POINTER(i_buffer_ptr, EM_RC_PARAM_ERR);

    std::unique_lock<std::mutex> lock(m_filter_mtx);
    if (m_logo_info.pos && m_soft_logo_ptr)
    {
        m_soft_logo_ptr->LogoFilter(i_buffer_ptr);
    }
    if (m_text_info.pos && m_soft_subtitle_ptr)
    {
        m_soft_subtitle_ptr->SubTitleFilter(i_buffer_ptr);
    }
    if (m_effect_type != EM_EFFECT_NONE && m_dict_effect_ptr)
    {
        m_dict_effect_ptr->AddEffect(i_buffer_ptr);
    }

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：合成任务（仅仅支持YUV420P）
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMixerSource::MixVideoTask(TSize i_main_size, uint8_t* io_main_image_ptr, TRect i_sub_size,
    uint8_t* i_sub_image_ptr)
{
    CHECK_POINTER(io_main_image_ptr, EM_RC_PARAM_ERR);
    CHECK_POINTER(i_sub_image_ptr, EM_RC_PARAM_ERR);

    /*主画面每行Y占用的字节数*/
    int main_y_byte_per_line = i_main_size.width;
    /*主画面每行U占用的字节数*/
    int main_u_byte_per_line = i_main_size.width / 2;
    /*子画面每行Y占用的字节数*/
    int sub_y_byte_per_line = i_sub_size.width;
    /*子画面每行U占用的字节数*/
    int sub_u_byte_per_line = i_sub_size.width / 2;

    /*主画面U的起始位置*/
    uint8_t *main_u_start_sddr_ptr = io_main_image_ptr + i_main_size.width*i_main_size.height;
    /*主画面V的起始位置*/
    uint8_t *main_v_start_addr_ptr = io_main_image_ptr + i_main_size.width*i_main_size.height + i_main_size.width*i_main_size.height / 4;
    /*分画面U的起始位置*/
    uint8_t *sub_u_start_addr_ptr = i_sub_image_ptr + i_sub_size.width*i_sub_size.height;
    /*分画面V的起始位置*/
    uint8_t *sub_v_start_addr_ptr = i_sub_image_ptr + i_sub_size.width*i_sub_size.height + i_sub_size.width*i_sub_size.height / 4;

    int j = 0;
    for (int i = 0; i < i_sub_size.height; i++)
    {
        /*Y数据拷贝*/
        memcpy(io_main_image_ptr + (i_sub_size.iY + i)*main_y_byte_per_line + i_sub_size.iX,
            i_sub_image_ptr + i*sub_y_byte_per_line, sub_y_byte_per_line);
        /*UV数据拷贝*/
        if (i % 2 == 0)
        {
            memcpy(main_u_start_sddr_ptr + (i_sub_size.iY / 2 + j)*main_u_byte_per_line + i_sub_size.iX / 2,
                sub_u_start_addr_ptr + j*sub_u_byte_per_line, sub_u_byte_per_line);
            memcpy(main_v_start_addr_ptr + (i_sub_size.iY / 2 + j)*main_u_byte_per_line + i_sub_size.iX / 2,
                sub_v_start_addr_ptr + j*sub_u_byte_per_line, sub_u_byte_per_line);
            j++;
        }
    }
    return EM_RC_OK;
}
