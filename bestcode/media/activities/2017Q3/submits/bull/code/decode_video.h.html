/**********************************  类声明  ***********************************/

class DecodeVideo : public IDecoder
{

	DecodeVideo(){};
	~DecodeVideo(){};

public:
	// 初始化解码功能模块
	ERetCode CreateDecoder(TSourceProps* i_media_props_ptr);

	// 设置解码数据回调对象,i_pOutParam暂时未实现
	ERetCode SetDecDataCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_props_ptr);

	//设置解码数据渲染器，nullptr表示关闭
	ERetCode SetRender(/*TRenderInfo* i_pRenderInfo*/);

	//解码之后如果o_pDstData为非空，直接用o_pDstData输出，或者回调输出
	ERetCode DecodeData(uint8_t* i_src_data_ptr, int i_data_len, uint8_t** o_dst_data_ptr = nullptr );

	// 反初始化解码功能模块
	ERetCode DestroyDecoder();

	//软解里面这个接口弃用
	VideoSrcImage GetReadyImage(){ return nullptr; }

private:
	bool ReadyWMV9Codec(uint8_t** i_inbuf_ptr, int* io_size_ptr);// NOLINT(runtime/references)

private:
    AVCodec *m_codec_ptr = nullptr;							//解码器
    AVCodecContext *m_codec_ctx_ptr = nullptr;				//解码器上下文
    AVCodecParserContext *m_codec_parser_ctx_ptr = nullptr; //过滤器上下文

	TDecOutProps m_out_frame;                               //解码输出缓冲帧
	TOutFrame m_dec_frame;                                  //解码帧

	uint8_t* m_data_buffer_ptr = nullptr;                   //数据缓冲帧
    AVFrame	*m_frame_ptr = nullptr;                         //解码YUV帧
    uint8_t *m_out_buffer_ptr = nullptr;                    //ffmpeg缓冲帧

	bool m_pri_changed_b = false;                           //WMV9头是否发生改变
	bool m_first_frame_b = true;                            //WMV9第一帧是否到来

    int m_got_picture_flag = 0;                             //是否获取一帧数据
    bool m_is_open_b = false;                               //是否打开解码器

	std::mutex m_mtx_dec;                                   //解码器锁
};
