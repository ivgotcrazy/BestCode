/*******************************************************************************
* 函  数: CreateVALog
* 描  述: 创建日志
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月01日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
int CreateVALog()
{
    if(g_rbs_Log_ptr != nullptr)
        return 0;

    g_rbs_Log_ptr = new BaseLog();

    char timeBuf[COMMON_SIZE_16] = {0};
    char dir_path[COMMON_SIZE_PATH] = {0};
    time_t tt = time(NULL);
    tm *pTime = localtime(&tt);
	sprintf(timeBuf, "%d%02d%02d%02d%02d%02d", 1900 + pTime->tm_year, 1 + pTime->tm_mon,
        pTime->tm_mday,pTime->tm_hour,pTime->tm_min,pTime->tm_sec);
	sprintf(dir_path, "VAMedia%s.log", timeBuf);

    LOG_INIT(dir_path,1);  
    return 1;
}

/*------------------------------------------------------------------------------
* 描  述：初始化媒体工厂，同时初始化VA渲染器相关设备
* 参  数：[in] i_log_ptr 外部日志指针
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaFactory::InitFactory(BaseLog* i_log_ptr)
{
    if(i_log_ptr == nullptr)
    {
        if( CreateVALog() )
            m_is_create_Log = 1;
    }

    return MediaRender::CreateDisplay();
}

/*------------------------------------------------------------------------------
* 描  述：反初始化媒体工厂
* 参  数：无
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaFactory::UnInitFactory()
{
    if(m_is_create_Log)
    {
        SAFE_DELETE(g_rbs_Log_ptr);
        m_is_create_Log = 0;
    }
    return MediaRender::DestroyDisplay();
}

/*------------------------------------------------------------------------------
* 描  述：创建数据源(根据参数创建软处理源或者硬处理源)
* 参  数：[in] i_src_props_ptr 数据源属性
* 参  数：[out] o_source_ptr   创建的数据源
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaFactory::CreateSource(TSourceProps* i_src_props_ptr, ISource** o_source_ptr)
{
	CHECK_POINTER(i_src_props_ptr, EM_RC_OK);

	*o_source_ptr = new RBSMediaSource();

    return (*o_source_ptr)->CreateSource(i_src_props_ptr);
}

/*------------------------------------------------------------------------------
* 描  述：销数据源
* 参  数：[in] i_source_ptr 待销毁的数据源
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaFactory::DestroySource(ISource** i_source_ptr)
{
	CHECK_POINTER(i_source_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(*i_source_ptr, EM_RC_PARAM_ERR);

	(*i_source_ptr)->DestroySource();
	SAFE_DELETE(*i_source_ptr);

	return EM_RC_OK;
}
