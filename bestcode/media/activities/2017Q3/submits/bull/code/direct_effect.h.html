class DirectEffect;
typedef ERetCode(DirectEffect::*EffectFunc)(uint8_t*); // 特效函数

struct TEffectTable
{
    EffectFunc   effect_func;    
    EEffectType  effect;     
};

/***********************************  类声明  **********************************/

class DirectEffect
{
public:

    DirectEffect();
    ~DirectEffect();

    ERetCode SetDictSize(int i_width, int i_height);
    ERetCode SetEffect(EEffectType i_effect);
    
    ERetCode UpdateLastDictImage(uint8_t* i_dict_data_ptr);    
    ERetCode AddEffect(uint8_t* io_dict_data_ptr);
    
private:

    ERetCode EffectFade(uint8_t* io_dict_data_ptr);    
    ERetCode EffectEraseH(uint8_t* io_dict_data_ptr);
    ERetCode EffectEraseV(uint8_t* io_dict_data_ptr);
    ERetCode EffectShutterH(uint8_t* io_dict_data_ptr);
    ERetCode EffectShutterV(uint8_t* io_dict_data_ptr);    
    ERetCode EffectBoxIn(uint8_t* io_dict_data_ptr);
    ERetCode EffectBoxOut(uint8_t* io_dict_data_ptr);    
    
private:	

    EEffectType  m_dict_effect = EM_EFFECT_NONE;    //导播特效
    EEffectType  m_tmp_effect = EM_EFFECT_NONE;     //临时导播特效
    EEffectState m_effect_state = EM_ES_END;        //特效运行状态
	 
    int           m_width = 0;                      //数据宽
    int           m_height = 0;                     //数据高
    int           m_frame_size = 0;                 //帧大小
    int           m_frame_count = 0;                //帧计数
    
    uint8_t*      m_last_dict_image_ptr = nullptr;  //最后一个帧
    uint8_t*      m_now_dict_image_ptr = nullptr;   //当前特效帧

    static TEffectTable   m_effect_table[];         //特效函数表
};
