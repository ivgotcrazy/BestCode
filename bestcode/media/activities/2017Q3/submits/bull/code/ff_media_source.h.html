#pragma once

/**************************** 条件编译选项和头文件 ****************************/
#include <mutex>
#include <queue>
#include <condition_variable>
#include <thread>

#include "ConfDefine.h"
#include "BaseThread.h"
#include "isource.h"
#include "media_ring_buffer.h"

/**********************************  类声明  ***********************************/

class IDecoder;
class MediaProcRoute;
class FFMediaSource : public BaseThread, public ISource
{
public:
	FFMediaSource(){};
	virtual ~FFMediaSource(){};

	// 创建源对象
	ERetCode CreateSource(TSourceProps* i_src_props_ptr);

	// 输入数据
	ERetCode InputSrcData(uint8_t* i_src_data_ptr, int i_data_len);

	// 销毁源对象
	ERetCode DestroySource();

	ERetCode SetSubTitle(TTextInfo* i_text_info_ptr) { return EM_RC_OK; }
	ERetCode SetLogo(TLogoInfo* i_log_info_ptr) { return EM_RC_OK; }

	ERetCode SetEffect(EEffectType i_effect)  { return EM_RC_OK; }
	ERetCode RunEffect() { return EM_RC_OK; }

	void LockReadyImage(){ return; }
	VideoSrcImage GetReadyImage(TOutFrame* o_data_ptr = nullptr);
	void UnLockReadyImage(){ return; }

	// 获取源属性
	TSourceProps* GetSrcProps();
	
	//如果是编码流采用的解码回调
	ERetCode SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_out_param_ptr);

	//注册其他类型解码器（目前只有一种，软处理支持硬输出，硬处理支持软输出）
	ERetCode RegisterMediaProc(EMediaMode i_media_mode);

	void OnEncodeCallBack(TOutFrame* i_enc_data_ptr, RBSUserData i_userData);
	void OnDecodeCallBack(TOutFrame* i_dec_data_ptr, RBSUserData i_user_data);

public:

	//数据缓冲区处理
	ERetCode EnqueueMediaData(TOutFrame *i_media_pkt_ptr);
	TOutFrame* DequeueMediaData();
	ERetCode ClearMediaData();
	ERetCode CheckMediaDataEmpty();

	//空闲数据缓冲区处理
	ERetCode EnqueueFreeMediaData(TOutFrame *i_media_pkt_ptr);
	TOutFrame* DequeueFreeMediaData();
	ERetCode ClearFreeMediaData();
	ERetCode CheckFreeMediaDataEmpty();

private:
	ERetCode ThreadFuncProc();
	ERetCode AllocDataBuffer(EVideoCodecID i_video_id);
	ERetCode ReleaseDataBuffer();

private:
	int m_is_src_inited = 0;                       //是否初始化
	int m_buffer_size = 0;                         //单个缓冲大小
	int  m_src_image_size = 0;        	           //源图像大小
	
	std::queue<TOutFrame*>m_enc_buffer_que;        //编码数据缓冲
	std::queue<TOutFrame*>m_free_enc_buffer_ptr;   //空闲编码数据缓冲

	std::mutex m_enc_mtx;                          //缓冲区编码数据锁
	std::mutex m_free_enc_mtx;                     //空闲缓冲区锁
	
	TSourceProps m_src_props;                      //数据源参数信息
	
	std::condition_variable m_data_cond;           //编码数据条件变量

	std::mutex m_data_mtx;                         //数据锁
	std::mutex m_modify_mtx;                       //修改锁
	std::mutex m_media_proc_mtx;                    //媒体源输出锁

	IVCallBack* m_dec_callback_ptr = nullptr;      //数据解码回调
	
	TOutFrame m_out_frame;                         //解码输出帧
	uint64_t m_dec_count = 0;                      	//解码计数
	TDecOutProps* m_dec_out_props_ptr = nullptr;    //解码回调输出参数
	EThreadState m_run_status = EM_TS_STOP;         //解码线程运行状态
	std::thread m_thread_handle;                    //解码线程句柄

	IDecoder*  m_decoder_ptr = nullptr;             //解码句柄
	MediaProcRoute* m_media_proc_ptr = nullptr;     //媒体源输出
	MediaRingBufferA m_media_ring_buffer;           //缓冲区
};
