VAMediaSource::VAMediaSource()
{
    pthread_mutex_init(&m_input_mtx, nullptr);
    pthread_cond_init(&m_input_cond, nullptr);
}

VAMediaSource::~VAMediaSource()
{
    pthread_mutex_destroy(&m_input_mtx);
    pthread_cond_destroy(&m_input_cond);
}

/*******************************************************************************
* 函  数: Init
* 描  述: 初始化RawSource
* 参  数: i_pSrcProps: source属性
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMediaSource::CreateSource(TSourceProps* i_src_props_ptr)
{
    CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

    if (m_is_inited)
        return EM_RC_INITED_ERR;

    if (i_src_props_ptr->v_codec_id >= EM_MT_H264)
    {
        m_decoder_ptr = new MediaDecoder;
        m_src_props.media_mode = EM_MEDIA_MODE_HARD;
        memcpy(&m_src_props, i_src_props_ptr, sizeof(TSourceProps));
        //m_pDecoder->SetDecDataCallBack(m_pDecCallBack, m_pDecOutProps);
        ERetCode emRC = m_decoder_ptr->CreateDecoder(&m_src_props);
        if (emRC != EM_RC_OK)
        {
            LOG_ERROR("Media Source [%d] Init Decoder Failed [%d]", i_src_props_ptr->user_data, emRC);
            SAFE_DELETE(m_decoder_ptr);
            return emRC;
        }
    }
    else
    {
        if (m_exit_code != EM_TS_STOP)
        {
            LOG_ERROR("Media Source[%ld] Has Inited!", i_src_props_ptr->user_data);
            return EM_RC_INITED_ERR;
        }

        ERetCode er = CreateRing();
        if (er != EM_RC_OK)
        {
            LOG_ERROR("RawSource[%d] Create Ring Fail[%d]", i_src_props_ptr->src_id, er);
            return er;
        }

        er = CreateImage(i_src_props_ptr);
        if (er != EM_RC_OK)
        {
            LOG_ERROR("RawSource[%d] Create Image Fail[%d]", i_src_props_ptr->src_id, er);
            return er;
        }

        StartThread();
    }

    m_is_inited = 1;

    memcpy(&m_src_props, i_src_props_ptr, sizeof(TSourceProps));
    LOG_INFO("Init Media Source[%d], Format[%s], Size[%dx%d]",
        i_src_props_ptr->src_id, GetStringFromFormat(m_src_props.format),
        m_src_props.src_size.width, m_src_props.src_size.height);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CreateRing
* 描  述: 创建链表
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMediaSource::CreateRing()
{
    TReadySource* source_ptr = new TReadySource;
    m_head_ptr = source_ptr;
    m_cursor_ptr = source_ptr;
    for (int i = 1; i < VA_SOURCE_RING_SIZE; i++)
    {
        source_ptr = new TReadySource;
        m_cursor_ptr->next_ptr = source_ptr;
        source_ptr->pre_ptr = m_cursor_ptr;
        m_cursor_ptr = source_ptr;
    }

    m_head_ptr->pre_ptr = m_cursor_ptr;
    m_cursor_ptr->next_ptr = m_head_ptr;
    m_cursor_ptr = m_head_ptr;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DestroyRing
* 描  述: 销毁链表,必须先调用DestroyImage
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMediaSource::DestroyRing()
{
    for (int i = 0; i < VA_SOURCE_RING_SIZE; i++)
    {// 销毁img、释放资源
        m_cursor_ptr = m_head_ptr;
        if (m_cursor_ptr != nullptr)
        {
            SAFE_DELETE_ARRAY(m_cursor_ptr->src_data_ptr);
            m_head_ptr = m_cursor_ptr->next_ptr;
            SAFE_DELETE(m_cursor_ptr);
        }
    }

    m_head_ptr = nullptr;
    m_cursor_ptr = nullptr;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CreateImage
* 描  述: 创建链表中节点的Image
* 参  数: i_pSrcProps
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMediaSource::CreateImage(TSourceProps* i_src_props_ptr)
{
    if (i_src_props_ptr->format > EM_YUV_FORMAT_NV12)
    {// 422的可以支持，此处未支持
        LOG_ERROR("RawSource[%d] Has't Support Format[%d]", i_src_props_ptr->src_id, i_src_props_ptr->format);
        return EM_RC_FILE_FORMAT_ERR;
    }

    VAImageFormat va_img_format;
    va_img_format.fourcc = GetFourccFromFormat(i_src_props_ptr->format);
    va_img_format.byte_order = VA_LSB_FIRST;
    va_img_format.bits_per_pixel = 12;

    // 创建image，并修改状态
    m_cursor_ptr = m_head_ptr;
    m_src_image_size = i_src_props_ptr->src_size.width*i_src_props_ptr->src_size.height * 3 / 2;
    for (int i = 0; i < VA_SOURCE_RING_SIZE; i++)
    {
        m_cursor_ptr->va_image_ptr = new VAImage();
        vaCreateImage(g_va_display, &va_img_format, i_src_props_ptr->src_size.width,
            i_src_props_ptr->src_size.height, m_cursor_ptr->va_image_ptr);
        vaMapBuffer(g_va_display, m_cursor_ptr->va_image_ptr->buf, (void**)&m_cursor_ptr->image_buf_ptr);

        m_cursor_ptr->state = EM_RS_PRE_READY;
        m_cursor_ptr->src_data_ptr = new uint8_t[m_src_image_size];
        m_cursor_ptr = m_cursor_ptr->next_ptr;
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DestroyImage
* 描  述: 销毁Image
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMediaSource::DestroyImage()
{
    m_cursor_ptr = m_head_ptr;
    for (int i = 0; i < VA_SOURCE_RING_SIZE; i++)
    {
        if (m_cursor_ptr->va_image_ptr != nullptr)
        {// 这里发生了一次destroyimage，但是输入的是bufferid，导致的内存泄漏!!!
            vaUnmapBuffer(g_va_display, m_cursor_ptr->va_image_ptr->buf);
            vaDestroyImage(g_va_display, m_cursor_ptr->va_image_ptr->image_id);
            SAFE_DELETE(m_cursor_ptr->va_image_ptr);
        }
        m_cursor_ptr = m_cursor_ptr->next_ptr;
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UnInit
* 描  述: 反初始化源数据对象
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMediaSource::DestroySource()
{
    if (!m_is_inited)
        return EM_RC_INIT_ERR;

    if (m_src_props.v_codec_id < EM_MT_H264)
    {
        if (m_exit_code != EM_TS_STOP)
        {
            pthread_mutex_lock(&m_input_mtx);
            m_exit_code = EM_TS_STOPING;
            pthread_cond_signal(&m_input_cond);
            pthread_mutex_unlock(&m_input_mtx);

            while (m_exit_code != EM_TS_STOP)
                usleep(1000);
        }
        pthread_mutex_lock(&m_input_mtx);
        DestroyImage();
        DestroyRing();
        pthread_mutex_unlock(&m_input_mtx);
    }
    else
    {
        m_decoder_ptr->DestroyDecoder();
        SAFE_DELETE(m_decoder_ptr);
    }

 {
     std::unique_lock<std::mutex> lock(m_media_proc_mtx);
     if (m_media_proc_ptr)
     {
         m_media_proc_ptr->DestroyMediaProc();
         SAFE_DELETE(m_media_proc_ptr);
     }
 }


 LOG_INFO("UnInit VAMediaSource[%d]", m_src_props.src_id);
 return EM_RC_OK;
}

/*******************************************************************************
* 函  数: InputSrcData
* 描  述: 输入数据
* 参  数: i_pSrcData
* 参  数: i_iDataLen
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMediaSource::InputSrcData(uint8_t* i_pSrcData, int i_iDataLen)
{
    if (!m_is_inited)
        return EM_RC_INIT_ERR;

    CHECK_POINTER(i_pSrcData, EM_RC_PARAM_ERR);
    if (m_src_props.v_codec_id < EM_MT_H264)
    {
        if (m_head_ptr == nullptr || m_cursor_ptr == nullptr || m_exit_code != EM_TS_RUN)
            return EM_RC_UNINIT_ERR;
        if (i_iDataLen > m_src_image_size)// 检查输入数据大小
            return EM_RC_SIZE_NOTENOUGH;

        pthread_mutex_lock(&m_input_mtx);

        // 如果节点正在upload或者已经ready好， 缓存到下一个节点    
        TReadySource* pSource = m_cursor_ptr;
        if (pSource->state != EM_RS_PRE_READY)
            pSource = pSource->next_ptr;

        pSource->state = EM_RS_BUFFER;
        memcpy(pSource->src_data_ptr, i_pSrcData, i_iDataLen);

        pthread_cond_signal(&m_input_cond);
        pthread_mutex_unlock(&m_input_mtx);
    }
    else
        m_decoder_ptr->DecodeData(i_pSrcData, i_iDataLen);

    {
        std::unique_lock<std::mutex> lock(m_media_proc_mtx);
        if (m_media_proc_ptr)
            m_media_proc_ptr->InputSrcData(i_pSrcData, i_iDataLen);
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: GetReadyImage
* 描  述: 获取准备好的Image
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
VideoSrcImage VAMediaSource::GetReadyImage(TOutFrame* o_data_ptr)
{
    CAutoLock lock(m_input_mtx);
    if (o_data_ptr == nullptr)
    {
        if (m_src_props.v_codec_id < EM_MT_H264)
        {
            if (m_head_ptr == nullptr || m_cursor_ptr == nullptr || m_exit_code != EM_TS_RUN)
                return nullptr;

            // 如果游标节点前节点未准备好，说明没有准备好的Image
            if (m_cursor_ptr->pre_ptr->state != EM_RS_READY)
                return nullptr;

            return m_cursor_ptr->pre_ptr->va_image_ptr;
        }
        else
            return m_decoder_ptr->GetReadyImage();
    }
    else
    {
        //YUV数据回调
        if (m_media_proc_ptr)
            return m_media_proc_ptr->GetReadyImage(o_data_ptr);
    }
    return nullptr;
}

ERetCode VAMediaSource::RegisterMediaProc(EMediaMode i_eMediaMode)
{
    if (!m_is_inited)
        return EM_RC_INIT_ERR;

    std::unique_lock<std::mutex> lock(m_media_proc_mtx);
    ERetCode er = EM_RC_OK;
    if (i_eMediaMode != m_src_props.media_mode)
    {
        if (m_media_proc_ptr != nullptr)
        {
            m_media_proc_ptr->DestroyMediaProc();
            SAFE_DELETE(m_media_proc_ptr);
        }

        m_media_proc_ptr = new MediaProcRoute;
        er = m_media_proc_ptr->CreateMediaProc(i_eMediaMode, &m_src_props);
        if (er != EM_RC_OK)
            SAFE_DELETE(m_media_proc_ptr);
    }
    else
        return EM_RC_STATE_ERR;

    return er;
}

ERetCode VAMediaSource::SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_props_ptr)
{
    CHECK_POINTER(i_callback_ptr, EM_RC_PARAM_ERR);

    m_dec_callback_ptr = i_callback_ptr;
    m_dec_props_ptr = i_dec_props_ptr;

	if (m_decoder_ptr)
		m_decoder_ptr->SetDecDataCallBack(i_callback_ptr, i_dec_props_ptr);

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ThreadFuncProc
* 描  述: 线程函数， 上传数据到surface并取出csc后的Image
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMediaSource::ThreadFuncProc()
{
    timespec ts_cond;
    TReadySource* source_ptr = m_cursor_ptr;

    LOG_INFO("Media Source[%d] Start ProcessThread!", m_src_props.src_id);
    while (m_exit_code == EM_TS_RUN)
    {
        pthread_mutex_lock(&m_input_mtx);
        source_ptr = m_cursor_ptr;
        if (source_ptr->state != EM_RS_BUFFER)
        {// 无数据时等待
            ts_cond.tv_nsec = 0;
            ts_cond.tv_sec = (long)time(nullptr) + COMMON_TICK_THREADTMO;

            pthread_cond_timedwait(&m_input_cond, &m_input_mtx, &ts_cond);
            pthread_mutex_unlock(&m_input_mtx);
            continue;
        }
        // 已经buffer， 准备传输
        source_ptr->state = EM_RS_UPDATING;
        pthread_mutex_unlock(&m_input_mtx);

        if (m_src_props.format == EM_YUV_FORMAT_I420)
            UploadIYUVToImage(source_ptr->va_image_ptr, source_ptr->image_buf_ptr, source_ptr->src_data_ptr,
            m_src_props.src_size.width, m_src_props.src_size.height);
        else if (m_src_props.format == EM_YUV_FORMAT_YV12)
            UploadYV12ToImage(source_ptr->va_image_ptr, source_ptr->image_buf_ptr, source_ptr->src_data_ptr,
            m_src_props.src_size.width, m_src_props.src_size.height);
        else if (m_src_props.format == EM_YUV_FORMAT_YUYV)
            UploadYUYVToImage(source_ptr->va_image_ptr, source_ptr->image_buf_ptr, source_ptr->src_data_ptr,
            m_src_props.src_size.width, m_src_props.src_size.height);
        else
            UploadDataToImage(source_ptr->va_image_ptr, source_ptr->image_buf_ptr, source_ptr->src_data_ptr,
                m_src_props.format, m_src_props.src_size.width, m_src_props.src_size.height);

        // 传输完成后， 改变状态，并重置前一个节点状态。
        pthread_mutex_lock(&m_input_mtx);
        source_ptr->state = EM_RS_READY;
        source_ptr->pre_ptr->state = EM_RS_PRE_READY;

        // 游标节点向后移动
        m_cursor_ptr = m_cursor_ptr->next_ptr;
        pthread_mutex_unlock(&m_input_mtx);
    }

    m_exit_code = EM_TS_STOP;
    LOG_INFO("Media Source[%d] Stop ProcessThread!", m_src_props.src_id);
    return EM_RC_OK;
}
