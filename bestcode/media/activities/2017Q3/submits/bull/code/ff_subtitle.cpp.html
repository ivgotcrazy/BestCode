/*******************************************************************************
* 函  数: Init
* 描  述: 初始化输入输出filter
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月19日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFSubTitle::Init()
{   
    AVFilter* buffer_src_ptr  = avfilter_get_by_name("buffer");
    AVFilter* buffer_sink_ptr = avfilter_get_by_name("buffersink");
    enum AVPixelFormat pixel_format[] = { AV_PIX_FMT_YUV420P, AV_PIX_FMT_NONE };
    AVBufferSinkParams* buffer_sink_params_ptr = NULL;

    m_input_ptr  = avfilter_inout_alloc();
    m_output_ptr = avfilter_inout_alloc();
    m_filter_graph_ptr = avfilter_graph_alloc();

    /* buffer video source: the decoded frames from the decoder will be inserted here. */
    char filter_params[COMMON_SIZE_1K] = {0};
    sprintf(filter_params, 
        "video_size=%dx%d:pix_fmt=%d:time_base=%d/%d:pixel_aspect=%d/%d",
        m_src_width, m_src_height, AV_PIX_FMT_YUV420P,
        m_rational.num, m_rational.den, 1, 1);
    if (avfilter_graph_create_filter(&m_buffer_src_ctx_ptr, buffer_src_ptr, "in",filter_params, NULL, m_filter_graph_ptr) < 0)
    {
        LOG_ERROR("Cannot create buffer source!");
        UnInit();
        return EM_RC_FILE_CREATE_ERR;
    }

    /* buffer video sink: to terminate the filter chain. */
    buffer_sink_params_ptr = av_buffersink_params_alloc();
    buffer_sink_params_ptr->pixel_fmts = pixel_format;
    if (avfilter_graph_create_filter(&m_buffer_sink_ctx_ptr, buffer_sink_ptr, "out",NULL, 
										buffer_sink_params_ptr, m_filter_graph_ptr) < 0)
    {
        LOG_ERROR("Cannot create buffer sink!");
        av_free(buffer_sink_params_ptr);
        UnInit();
        return EM_RC_FILE_CREATE_ERR;
    }
    av_free(buffer_sink_params_ptr);
    /* Endpoints for the filter graph. */
    m_output_ptr->name       = av_strdup("in");
    m_output_ptr->filter_ctx = m_buffer_src_ctx_ptr;
    m_output_ptr->pad_idx    = 0;
    m_output_ptr->next       = NULL;

    m_input_ptr->name       = av_strdup("out");
    m_input_ptr->filter_ctx = m_buffer_sink_ctx_ptr;
    m_input_ptr->pad_idx    = 0;
    m_input_ptr->next       = NULL;

    m_frame_buffer_ptr = (uint8_t*)av_malloc(av_image_get_buffer_size(AV_PIX_FMT_YUV420P, m_src_width, m_src_height, 1));
    m_frame_ptr = av_frame_alloc();
    av_image_fill_arrays(m_frame_ptr->data, m_frame_ptr->linesize, m_frame_buffer_ptr, AV_PIX_FMT_YUV420P, 
																		m_src_width, m_src_height, 1);

    m_frame_ptr->width = m_src_width;
    m_frame_ptr->height = m_src_height;
    m_frame_ptr->format=AV_PIX_FMT_YUV420P;

    m_frame_index = 0;    
    memset(m_filter_decs,0,COMMON_SIZE_1K);
    return EM_RC_OK;
}

ERetCode FFSubTitle::SetVideoSourceSize(int i_width, int i_height)
{
    m_src_width = i_width;
    m_src_height = i_height;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: SetSubTitle
* 描  述: 添加字幕水印
* 参  数: i_pTxtInfo: 字幕属性
* 参  数: i_iHasLogo: 是否与台标重合
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月18日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFSubTitle::SetSubTitle(TTextInfo* i_text_info_ptr)
{
    if( i_text_info_ptr == nullptr || i_text_info_ptr->pos > RBS_WM_POS_RB )
        return EM_RC_PARAM_ERR;

    if( !m_is_init_Filter_flag )
        Init();

	RGBTOBGR(i_text_info_ptr);
    char font_file[COMMON_SIZE_256] = {0};
    const char* font_file_ptr = getenv(ENV_PATH_RBSSYS);
    if( font_file_ptr == nullptr )
        sprintf(font_file, "/opt/AIO/Service/kaiti.ttf");
    else
        sprintf(font_file,"%s/kaiti.ttf", font_file_ptr);

    int filter_size = 0;
    int font_pos = i_text_info_ptr->is_convert_logo ? 128 : 0;    
    int subtitle_size = strlen(i_text_info_ptr->acData);
    memset(m_filter_decs, 0, COMMON_SIZE_1K);

    if(i_text_info_ptr->pos == RBS_WM_POS_LT)//左上
    {
		// filter中的color必须满足 BGR: aabbcc,不够的值必须补0，否则失败
        if( i_text_info_ptr->iIsShowTime )
            sprintf(m_filter_decs,"drawtext=fontfile=%s:text=%%{localtime}:fontsize=%d:x=(%d+4): \
			y=4:fontcolor_expr=%s",font_file, i_text_info_ptr->font_size, font_pos, i_text_info_ptr->acFontColor);
        if( subtitle_size > 0 )
        {
            if( i_text_info_ptr->iIsShowTime )
            {
                filter_size = strlen(m_filter_decs);
                sprintf(m_filter_decs+filter_size,",drawtext=fontfile=%s:text='%s':fontsize=%d:x=(%d+4):y=(12+lh): \
				fontcolor_expr=%s",font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size, 
														font_pos,i_text_info_ptr->acFontColor);
            }
            else
                sprintf(m_filter_decs+filter_size,"drawtext=fontfile=%s:text='%s':fontsize=%d:x=(%d+4):y= \
				4:fontcolor_expr=%s",
                font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
        }
    }
    else if(i_text_info_ptr->pos == RBS_WM_POS_RT)//右上
    {
        if( i_text_info_ptr->iIsShowTime )
            sprintf(m_filter_decs,"drawtext=fontfile=%s:text=%%{localtime}:fontsize=%d:x= \
			(w-tw-(%d+4)):y=4:fontcolor_expr=%s", 
            font_file, i_text_info_ptr->font_size,font_pos, i_text_info_ptr->acFontColor);
        if( subtitle_size > 0 )
        {
            if( i_text_info_ptr->iIsShowTime )
            {
                filter_size = strlen(m_filter_decs);
                sprintf(m_filter_decs+filter_size,",drawtext=fontfile=%s:text='%s':fontsize=%d:x=(w-tw-(%d+4)): \
					y=(12+lh):fontcolor_expr=%s",
                    font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
            }
            else
                sprintf(m_filter_decs+filter_size,"drawtext=fontfile=%s:text='%s':fontsize=%d:x= \
				(w-tw-(%d+4)):y=4:fontcolor_expr=%s",
                font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
        }
    }  
    else if(i_text_info_ptr->pos == RBS_WM_POS_LB)//左下
    {
        if( subtitle_size > 0 )
        {
            if( i_text_info_ptr->iIsShowTime )
                sprintf(m_filter_decs,"drawtext=fontfile=%s:text='%s':fontsize=%d:x=(%d+4):y=(h-th*2-12): \
										fontcolor_expr=%s,drawtext=fontfile=%s:text=%%{localtime}:fontsize=%d: \
										x=(%d+4):y=(h-th-4):fontcolor_expr=%s", font_file, i_text_info_ptr->acData, 
										i_text_info_ptr->font_size,font_pos, i_text_info_ptr->acFontColor,
                                       font_file, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
            else
                sprintf(m_filter_decs,"drawtext=fontfile=%s:text='%s':fontsize=%d:x=(%d+4):y=(h-th-4): \
				fontcolor_expr=%s", 
                font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size,font_pos, i_text_info_ptr->acFontColor);
        }
        else if( i_text_info_ptr->iIsShowTime )
            sprintf(m_filter_decs,"drawtext=fontfile=%s:text=%%{localtime}:fontsize=%d:x=(%d+4): \
			y=(h-th-4):fontcolor_expr=%s",
            font_file, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
    }
    else if(i_text_info_ptr->pos == RBS_WM_POS_RB)//右下
    {
        if( subtitle_size > 0 )
        {
            if( i_text_info_ptr->iIsShowTime )
                sprintf(m_filter_decs,"drawtext=fontfile=%s:text='%s':fontsize=%d:x=(w-tw-(%d+4)): \
										y=(h-th*2-12):fontcolor_expr=%s,drawtext=fontfile=%s:text=%% \
										{localtime}:fontsize=%d:x=(w-tw-(%d+4)):y=(h-th-4):fontcolor_expr=%s",
                                       font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size,font_pos, 
									   i_text_info_ptr->acFontColor,font_file, i_text_info_ptr->font_size, 
									   font_pos,i_text_info_ptr->acFontColor);
            else
                sprintf(m_filter_decs,"drawtext=fontfile=%s:text='%s':fontsize=%d:x=(w-tw-(%d+4)): \
				y=(h-th-4):fontcolor_expr=%s", 
                font_file, i_text_info_ptr->acData, i_text_info_ptr->font_size,font_pos, i_text_info_ptr->acFontColor);
        }
        else if( i_text_info_ptr->iIsShowTime )
            sprintf(m_filter_decs,"drawtext=fontfile=%s:text=%%{localtime}:fontsize=%d:x=(w-tw-(%d+4)): \
			y=(h-th-4):fontcolor_expr=%s",
            font_file, i_text_info_ptr->font_size, font_pos,i_text_info_ptr->acFontColor);
    }

    if(avfilter_graph_parse_ptr(m_filter_graph_ptr, m_filter_decs,&m_input_ptr, &m_output_ptr, NULL) < 0)
    {
        UnInit();
        LOG_ERROR("Parse graph failed!");
        return EM_RC_UNKNOW;
    }
    if (avfilter_graph_config(m_filter_graph_ptr, NULL) < 0)
    {
        UnInit();
        LOG_ERROR("Config graph failed!");
        return EM_RC_UNKNOW;
    }

    m_is_init_Filter_flag = 1;
    return EM_RC_OK;
}//NOLINT(readability/fn_size)

ERetCode FFSubTitle::RGBTOBGR(TTextInfo* i_text_info_ptr)
{
	std::string strRGB = i_text_info_ptr->acFontColor;
	if (strRGB.size() != 6)
		return EM_RC_OP_NOT_SUPPORT;

	std::string strBGR;
	strBGR.append(strRGB.substr(strRGB.length() - 2, 2));
	strBGR.append(strRGB.substr(strRGB.length() - 4, 2));
	strBGR.append(strRGB.substr(strRGB.length() - 6, 2));

	memcpy(i_text_info_ptr->acFontColor, strBGR.c_str(), strBGR.size());
	return EM_RC_OK;
}

/*******************************************************************************
* 函  数: SubTitleFilter
* 描  述: 字幕水印数据处理
* 参  数: io_pData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月18日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFSubTitle::SubTitleFilter(uint8_t* io_data_ptr)
{   
    if( !m_is_init_Filter_flag )
        return EM_RC_UNINIT_ERR;

    m_frame_ptr->data[0] = io_data_ptr;
    m_frame_ptr->data[1] = io_data_ptr + m_src_width * m_src_height;
    m_frame_ptr->data[2] = m_frame_ptr->data[1] + m_src_width * m_src_height / 4;

    m_frame_ptr->pts = m_frame_index * av_q2d(m_rational);
    if (av_buffersrc_add_frame(m_buffer_src_ctx_ptr, m_frame_ptr) < 0) 
    {
        LOG_ERROR("Error while feeding the filtergraph!");
        return EM_RC_DATA_ERR;
    }

    /* pull filtered pictures from the filtergraph */
    int iRet = av_buffersink_get_frame(m_buffer_sink_ctx_ptr,m_frame_ptr);
    if (iRet < 0)
        return EM_RC_UNKNOW;

    m_frame_index++;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UnInit
* 描  述: 
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月19日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFSubTitle::UnInit()
{
    if(m_filter_graph_ptr)
    {
        avfilter_graph_free(&m_filter_graph_ptr);
        m_filter_graph_ptr = NULL;
    }
    if(m_output_ptr)
    {
        avfilter_inout_free(&m_output_ptr);
        m_output_ptr = NULL;
    }
    if(m_input_ptr)
    {
        avfilter_inout_free(&m_input_ptr);
        m_input_ptr = NULL;
    }
    if(m_frame_buffer_ptr)
    {
        av_free(m_frame_buffer_ptr);
        m_frame_buffer_ptr = NULL;
    }
    if(m_frame_ptr)
    {
        av_frame_free(&m_frame_ptr);
        m_frame_ptr = NULL;
    }
    m_frame_index = 0;
    m_is_init_Filter_flag = 0;
    return EM_RC_OK;
}
