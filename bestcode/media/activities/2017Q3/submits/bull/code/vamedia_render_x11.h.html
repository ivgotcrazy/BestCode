struct TRenderMember
{
    int            render_state = EM_SS_NONE;  //渲染状态
    EVMixLevel     level = EM_VML_NONE;        //层级

    TRect          rect;                       //渲染区域
    TRenderMember* tmp_ptr = nullptr;          //渲染过度变量
    ISource*     source_ptr = nullptr;         //数据源指针
};

typedef std::map<int, TRenderMember*> MapRenderMember;

/**********************************   变量   **********************************/

/********************************** 函数声明 **********************************/

/**********************************  类声明  ***********************************/

class VARenderX11 : public BaseThread, public IVRender
{
public:

    VARenderX11();
    virtual ~VARenderX11();

public:
    
	//渲染器相关接口
	ERetCode CreateRender(TMediaRenderProps* i_render_props_ptr);

    ERetCode ModifyRenderSize(TSize* i_size_ptr);

	ERetCode RenderData(uint8_t* i_data_ptr);

	ERetCode DestroyRender();

	//数据源相关接口
    ERetCode AddSource(int i_src_id, TProcItem* i_item_ptr);

    ERetCode DelSource(int i_src_id); 
        
    ERetCode ModifySrcOutRect(int i_src_id, TRect* i_rect_ptr);

    ERetCode ModifySrcLevel(int i_src_id, EVMixLevel i_level);
    
    ERetCode ModifySrcSource(int i_src_id, ISource* i_source_ptr);

	//显示器接口
    ERetCode GetSurportResolutionList(TOutputResolutions **o_resolution_ptr);

    void FreeSurportResolutionList(TOutputResolutions *i_resolution_ptr);

    ERetCode SetCurResolution(char *i_output_name_ptr, int i_width, int i_height);

private:

    ERetCode CreateRenderWindow(TRect* i_rect_ptr);
        
    ERetCode ThreadFuncProc();

    ERetCode CheckRootWindowSize();

    ERetCode CheckSourceState();

    ERetCode RunRenderTask();

    int GetOutputConnectionCount();

    void StartHotPlugMonitor();

    static void *HotPlugMonitorThreadAdapter(void *i_thread_arg_ptr);

    void HotPlugMonitorThread();
    
    XRRModeInfo* FindModeByXID(XRRScreenResources *i_screen_res_ptr, RRMode i_xid_mode);
    
    XRRCrtcInfo* GetCrtcByXID(XRRScreenResources *i_screen_res_ptr, RRCrtc i_xid_crtc);

    void FreeCrtcInfo(XRRCrtcInfo *i_crtc_ptr);
    
    ERetCode SetScreenSize(XRROutputInfo *i_output_info_ptr, int i_width, int i_height);
    
    void OpenConnectedMonitor();

private:
        
    int             m_use_surf_id = 0;                      // 使用的ID索引
    Window          m_wnd_root;                             // root窗口
    Window          m_wnd_render = None;                    // 渲染窗口

    VAConfigID      m_va_config_id = 0;                     //ID
    VASurfaceID     m_va_surface_id[COMMON_SIZE_4] = { 0 }; //surfaceID数组
    TRect           m_render_rect;						    //渲染区域
    TSize           m_render_size;						    //渲染大小
    TSize           m_root_wnd_size;                        //窗口大小
    TSize           m_update_size;                          //更新大小
    TSize           m_cur_screen_size;                      //当前屏幕大小
    TRect           m_tmp_rect;                             //临时区域
    TSize           m_tmp_size;                             //临时大小
    
    MapRenderMember m_render_map;					        //渲染成员
    pthread_mutex_t m_render_mtx;                           //渲染锁
    EThreadState    m_monitor_thread_state = EM_TS_STOP;    //监视线程状态

	std::mutex      m_render_surface_mtx;                   //surface渲染锁

    int m_is_inited = 0;                                    //是否初始化
};
