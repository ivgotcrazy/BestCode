    pthread_mutex_destroy(&m_render_mtx);

/*******************************************************************************
* 函  数: Init
* 描  述: 初始化创建窗口
* 参  数: i_pRect
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月16日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VARenderX11::CreateRender(TMediaRenderProps* i_render_props_ptr)
{
    CHECK_POINTER(i_render_props_ptr, EM_RC_PARAM_ERR);
    CHECK_POINTER(g_va_display, EM_RC_UNINIT_ERR);

    m_tmp_size = m_render_size = i_render_props_ptr->size;
    if (i_render_props_ptr->rect.width != 0 || i_render_props_ptr->rect.height != 0)
		memcpy(&m_tmp_rect, &i_render_props_ptr->rect, sizeof(i_render_props_ptr->rect));

    m_yuv_format = i_render_props_ptr->format;

    int screen_id = DefaultScreen(g_display_ptr);
    m_wnd_root = RootWindow(g_display_ptr, screen_id);

    //启动检测线程
    StartHotPlugMonitor();

    if (0 == GetOutputConnectionCount())
    {
        LOG_WARN("has no screen connected, do not start the render thread now");
        return EM_RC_OK;
    }
    ERetCode er = CreateRenderWindow(&i_render_props_ptr->rect);
    if (er != EM_RC_OK)
    {
        LOG_ERROR("Create Render Window Fail[%d]", er);
        return er;
    }

    VASurfaceAttrib va_surf_attr;
    va_surf_attr.type = VASurfaceAttribPixelFormat;
    va_surf_attr.flags = VA_SURFACE_ATTRIB_SETTABLE;
    va_surf_attr.value.type = VAGenericValueTypeInteger;
    va_surf_attr.value.value.i = VA_FOURCC_NV12;
    vaCreateSurfaces(g_va_display, VA_RT_FORMAT_YUV420, m_render_size.width, m_render_size.height,
        &m_va_surface_id[0], COMMON_SIZE_4, &va_surf_attr,1);
    
    m_is_inited = 1;
    LOG_INFO("Init VARenderX11 Success, Size[%dx%d]", i_render_props_ptr->size.width, i_render_props_ptr->size.height);
    return StartThread();
}

/*******************************************************************************
* 函  数: UnInit
* 描  述:
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月16日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VARenderX11::DestroyRender()
{
    //停止监控线程
    if (m_monitor_thread_state == EM_TS_RUN)
    {
        m_monitor_thread_state = EM_TS_STOPING;
        while (m_monitor_thread_state != EM_TS_STOP)
            usleep(1000);
    }
    //停止渲染线程
    if (m_exit_code == EM_TS_RUN)
    {
        m_exit_code = EM_TS_STOPING;
        while (m_exit_code != EM_TS_STOP)
            usleep(1000);
    }

    if (m_va_surface_id[0])
        vaDestroySurfaces(g_va_display, &m_va_surface_id[0], COMMON_SIZE_4);

    if (g_va_display)
    {
        if (m_wnd_render != None)
        {
            XUnmapWindow(g_display_ptr, m_wnd_render);
            XDestroyWindow(g_display_ptr, m_wnd_render);
            m_wnd_render = None;
        }
    }

    auto node_map = m_render_map.begin();
    for (; node_map != m_render_map.end(); node_map++)
    {
        TRenderMember* member_ptr = node_map->second;
        SAFE_DELETE(member_ptr->tmp_ptr);
        SAFE_DELETE(member_ptr);
    }
    m_render_map.clear();
    m_is_inited = 0;
    LOG_INFO("UnInit Render!");
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CreateRenderWindow
* 描  述: 创建渲染窗口
* 参  数: i_pRect: 渲染位置
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::CreateRenderWindow(TRect* i_rect_ptr)
{
    int screen_id = DefaultScreen(g_display_ptr);
    uint32_t uiBlackPixel = BlackPixel(g_display_ptr, screen_id);

    XWindowAttributes xw_attr;
    XGetWindowAttributes(g_display_ptr, m_wnd_root, &xw_attr);
    LOG_INFO("RootWindow Size[%dx%d]", xw_attr.width, xw_attr.height);
    if (i_rect_ptr == nullptr || i_rect_ptr->width == 0 || i_rect_ptr->height == 0)
    {//
        m_render_rect.width = xw_attr.width;
        m_render_rect.height = xw_attr.height;
    }
    else// 如果设置了输出位置，使用输出位置
        m_render_rect = *i_rect_ptr;

    m_root_wnd_size.width = xw_attr.width;
    m_root_wnd_size.height = xw_attr.height;
    m_wnd_render = XCreateSimpleWindow(g_display_ptr, m_wnd_root, m_render_rect.iX, m_render_rect.iY,
        m_render_rect.width, m_render_rect.height, 1, uiBlackPixel, uiBlackPixel);
    LOG_INFO("Create Window, Pos[%dx%d], Size[%dx%d]",
        m_render_rect.iX, m_render_rect.iY, m_render_rect.width, m_render_rect.height);

    XMapWindow(g_display_ptr, m_wnd_render);
    XSync(g_display_ptr, 0);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: AddSource
* 描  述: 添加渲染源
* 参  数: i_iSrcID: 源ID
* 参  数: i_pItem: 源属性
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月13日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::AddSource(int i_src_id, TProcItem* i_item_ptr)
{
    CHECK_POINTER(i_item_ptr, EM_RC_PARAM_ERR);
    CHECK_POINTER(g_va_display, EM_RC_UNINIT_ERR);

    CAutoLock lock(m_render_mtx);
    TRenderMember* member_ptr = new TRenderMember;
    member_ptr->tmp_ptr = new TRenderMember;
    member_ptr->rect = i_item_ptr->rect;
    member_ptr->source_ptr = i_item_ptr->source_ptr;
    member_ptr->level = i_item_ptr->level;
    member_ptr->render_state = EM_SS_ACTIVING;

    m_render_map.insert(MapRenderMember::value_type(i_src_id, member_ptr));
    LOG_INFO("Add Source[%d], Level[%d], Pos[%d,%d], Size[%dx%d]", i_src_id,
        member_ptr->level, member_ptr->rect.iX, member_ptr->rect.iY, member_ptr->rect.width, member_ptr->rect.height);

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DelSource
* 描  述:
* 参  数: i_iSrcID
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月13日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::DelSource(int i_src_id)
{
    auto node_map = m_render_map.find(i_src_id);
    if (node_map == m_render_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    CAutoLock lock(m_render_mtx);
    TRenderMember* member_ptr = node_map->second;
    SAFE_DELETE(member_ptr->tmp_ptr);
    SAFE_DELETE(member_ptr);

    LOG_INFO("Render Remove Source[%d]", i_src_id);
    m_render_map.erase(node_map);
    return EM_RC_OK;
}

ERetCode VARenderX11::RenderData(uint8_t* i_data_ptr)
{
	CHECK_POINTER(i_data_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(g_va_display, EM_RC_UNINIT_ERR);
	if (m_wnd_render == None)
		return EM_RC_UNINIT_ERR;

	std::unique_lock<std::mutex> lock(m_render_surface_mtx);

	VAImage va_image;
	uint8_t* img_buf_ptr = nullptr;

	VAStatus va_status = vaDeriveImage(g_va_display, m_va_surface_id[m_use_surf_id], &va_image);
	CHECK_VASTATUS(va_status, "vaDeriveImage");
	va_status = vaMapBuffer(g_va_display, va_image.buf, (void **)&img_buf_ptr);
	CHECK_VASTATUS(va_status, "MapRenderImage");

	CheckRootWindowSize();
	// 
	//     LOG_INFO("this[%x] Image Format: %s, Surface[%d] ID[%d]",this, GetStringFromFourcc(vaImage.format.fourcc), m_iUseSurfID,m_vaSurfaceID[m_iUseSurfID]);
	//     UploadIYUVToImage(&vaImage, pImgBuf, i_pData,m_sRenderSize.iW, m_sRenderSize.iH);
	UploadDataToImage(&va_image, img_buf_ptr, i_data_ptr, m_yuv_format, m_render_size.width, m_render_size.height);

	vaPutSurface(g_va_display, m_va_surface_id[m_use_surf_id], m_wnd_render,
		0, 0, m_render_size.width, m_render_size.height,
		m_render_rect.iX, m_render_rect.iY,
		m_render_rect.width, m_render_rect.height,
		nullptr, 0, VA_FRAME_PICTURE);

	vaUnmapBuffer(g_va_display, va_image.buf);
	vaDestroyImage(g_va_display, va_image.image_id);

	m_use_surf_id++;
	if (m_use_surf_id >= COMMON_SIZE_4)
		m_use_surf_id = 0;
	return EM_RC_OK;
}

{
    auto node_map = m_render_map.find(i_iSrcID);
    if (node_map == m_render_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    // 修改source需要加锁，保证线程没有在渲染
    CAutoLock lock(m_render_mtx);
    TRenderMember* member_ptr = node_map->second;
    if (member_ptr->render_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    member_ptr->source_ptr = i_pSource;
    member_ptr->render_state = EM_SS_ACTIVING;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ThreadFuncProc
* 描  述: 定时线程
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月02日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::ThreadFuncProc()
{
    uint64_t count = 0;
    uint64_t tick = stGetCurrentMicroSecond();
    uint64_t frame_timestamp = COMMON_TICK_BASE / COMMON_DEFAULT_FPS;

    LOG_INFO("Start RenderThread!");
    while (m_exit_code == EM_TS_RUN)
    {
		{
			std::unique_lock<std::mutex> lock(m_render_surface_mtx);

			if (m_update_size == m_render_size)
				m_update_size.width = m_update_size.height = 0;
			if (m_update_size.width && m_update_size.height)
			{
				m_render_size = m_update_size;
				m_update_size.width = m_update_size.height = 0;
				vaDestroySurfaces(g_va_display, &m_va_surface_id[0], COMMON_SIZE_4);

				VASurfaceAttrib va_surf_attr;
				va_surf_attr.type = VASurfaceAttribPixelFormat;
				va_surf_attr.flags = VA_SURFACE_ATTRIB_SETTABLE;
				va_surf_attr.value.type = VAGenericValueTypeInteger;
				va_surf_attr.value.value.i = VA_FOURCC_NV12;
				vaCreateSurfaces(g_va_display, VA_RT_FORMAT_YUV420, m_render_size.width, m_render_size.height,
					&m_va_surface_id[0], COMMON_SIZE_4, &va_surf_attr, 1);
				LOG_INFO("Change RenderSize[%dx%d]", m_render_size.width, m_render_size.height);
			}
		}

        pthread_mutex_lock(&m_render_mtx);
        if (m_render_map.size() == 0)
        {
            pthread_mutex_unlock(&m_render_mtx);
            usleep(COMMON_TICK_SECOND);
            continue;
        }

        CheckRootWindowSize();
        CheckSourceState();
        RunRenderTask();

        pthread_mutex_unlock(&m_render_mtx);

        count++;
        while (stGetCurrentMicroSecond() - tick < count * frame_timestamp)
            usleep(COMMON_TICK_SECOND);
    }

    m_exit_code = EM_TS_STOP;
    LOG_INFO("Stop RenderThread!");
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CheckRootWindowSize
* 描  述: 检查屏幕分辨率是否变化
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月14日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::CheckRootWindowSize()
{
    XWindowAttributes xw_attr;
    XGetWindowAttributes(g_display_ptr, m_wnd_root, &xw_attr);
    if (xw_attr.width != m_root_wnd_size.width || xw_attr.height != m_root_wnd_size.height)
    {// 屏幕分辨率变化        
        if (!m_render_rect.iX && !m_render_rect.iY &&
            m_root_wnd_size.width == m_render_rect.width &&
            m_root_wnd_size.height == m_render_rect.height)
        {// 全屏输出
            if (m_wnd_render != None)
            {
                LOG_INFO("resize render window from %dx%d to %dx%d", m_render_rect.width,m_render_rect.height, 
					xw_attr.width, xw_attr.height);
                m_render_rect.width = xw_attr.width;
                m_render_rect.height = xw_attr.height;
                XResizeWindow(g_display_ptr, m_wnd_render, xw_attr.width, xw_attr.height);
                XFlush(g_display_ptr);
            }
        }

        m_root_wnd_size.width = xw_attr.width;
        m_root_wnd_size.height = xw_attr.height;
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CheckSourceState
* 描  述: 检查输入源参数是否修改
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VARenderX11::CheckSourceState()
{
    int src_id = -1;
    TRenderMember* member_ptr = nullptr;
    auto node_map = m_render_map.begin();
    for (; node_map != m_render_map.end(); node_map++)
    {
        member_ptr = node_map->second;
        if (member_ptr->render_state == EM_SS_ACTIVING)
            continue;

        src_id = node_map->first;
        if (member_ptr->render_state & EM_SS_MODIFY_LEVEL)
        {// 需要修改输出层级
            member_ptr->level = member_ptr->tmp_ptr->level;
            LOG_INFO("Render Modify Source[%d] Level[%d]", src_id, member_ptr->level);
        }
        if (member_ptr->render_state & EM_SS_MODIFY_POS)
        {// 需要修改输出位置
            member_ptr->rect = member_ptr->tmp_ptr->rect;
            LOG_INFO("Render Modify Source[%d] Pos[%dx%d], Size[%dx%d]", src_id,
                member_ptr->rect.iX, member_ptr->rect.iY, member_ptr->rect.width, member_ptr->rect.height);
        }
        member_ptr->render_state = EM_SS_ACTIVING;
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RunRenderTask
* 描  述: 渲染
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VARenderX11::RunRenderTask()
{
    TRenderMember* member_ptr = nullptr;
    auto node_map = m_render_map.begin();
    for (; node_map != m_render_map.end(); node_map++)
    {
        member_ptr = node_map->second;
        if (member_ptr->render_state == EM_SS_ACTIVING &&
            member_ptr->source_ptr != nullptr && member_ptr->level == EM_VML_BOTTOM)
        {// 输入底层源image
            member_ptr->source_ptr->LockReadyImage();
            VAImage* image_ptr = static_cast<VAImage *>(member_ptr->source_ptr->GetReadyImage());
            if (image_ptr != nullptr)
                vaPutImage(g_va_display, m_va_surface_id[m_use_surf_id], image_ptr->image_id, 0, 0, image_ptr->width,
                image_ptr->height, member_ptr->rect.iX, member_ptr->rect.iY, member_ptr->rect.width, member_ptr->rect.height);
            member_ptr->source_ptr->UnLockReadyImage();
        }
    }

    node_map = m_render_map.begin();
    for (; node_map != m_render_map.end(); node_map++)
    {
        member_ptr = node_map->second;
        if (member_ptr->render_state == EM_SS_ACTIVING &&
            member_ptr->source_ptr != nullptr && member_ptr->level == EM_VML_TOP)
        {// 输入顶层源image
            member_ptr->source_ptr->LockReadyImage();
            VAImage* image_ptr = static_cast<VAImage *>(member_ptr->source_ptr->GetReadyImage());
            if (image_ptr != nullptr)
                vaPutImage(g_va_display, m_va_surface_id[m_use_surf_id], image_ptr->image_id, 0, 0, image_ptr->width,
                image_ptr->height, member_ptr->rect.iX, member_ptr->rect.iY, member_ptr->rect.width, member_ptr->rect.height);
            member_ptr->source_ptr->UnLockReadyImage();
        }
    }

    vaPutSurface(g_va_display, m_va_surface_id[m_use_surf_id], m_wnd_render, 0, 0,
        m_render_size.width, m_render_size.height, m_render_rect.iX, m_render_rect.iY,
        m_render_rect.width, m_render_rect.height, nullptr, 0, VA_FRAME_PICTURE);

    m_use_surf_id++;
    if (m_use_surf_id >= COMMON_SIZE_4)
        m_use_surf_id = 0;
    return EM_RC_OK;
}

XRRModeInfo* VARenderX11::FindModeByXID(XRRScreenResources *i_screen_res_ptr, RRMode i_xid_mode)
{
    XRRModeInfo *tmp_ptr = nullptr;
    for (int i = 0; i < i_screen_res_ptr->nmode; i++)
    {
        if (i_screen_res_ptr->modes[i].id == i_xid_mode)
        {
            tmp_ptr = &i_screen_res_ptr->modes[i];
        }
    }
    return tmp_ptr;
}

XRRCrtcInfo* VARenderX11::GetCrtcByXID(XRRScreenResources *i_screen_res_ptr, RRCrtc i_xid_crtc)
{
    XRRCrtcInfo *tmp_ptr = nullptr;
    for (int i = 0; i < i_screen_res_ptr->ncrtc; i++)
    {

        if (i_screen_res_ptr->crtcs[i] == i_xid_crtc)
            tmp_ptr = XRRGetCrtcInfo(g_display_ptr, i_screen_res_ptr, i_xid_crtc);
    }
    return tmp_ptr;
}

void VARenderX11::FreeCrtcInfo(XRRCrtcInfo *i_crtc_ptr)
{
    XRRFreeCrtcInfo(i_crtc_ptr);
    i_crtc_ptr = nullptr;
}

/*******************************************************************************
* 函  数: GetSurportResolutionList
* 描  述: 获取显示器支持的分辨率列表

********************************************************************************/
ERetCode VARenderX11::GetSurportResolutionList(TOutputResolutions **o_resolution_ptr)
{
    ERetCode er = EM_RC_OK;
    int output_count = 0;
    TResolutionInfo *temp_list_ptr = nullptr;

    CHECK_POINTER(o_resolution_ptr, EM_RC_PARAM_ERR);

    output_count = GetOutputConnectionCount();
    //没有显示器接入,直接返回
    if (output_count == 0)
    {
        LOG_ERROR("has no output connected");
        return EM_RC_NO_SCREEN_CONNECTION;
    }

    temp_list_ptr = new TResolutionInfo[output_count];
    XRRScreenResources *ccreen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);

    if (ccreen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return EM_RC_OP_NOT_SUPPORT;
    }

    //一个screen可以对应多个output(output可以理解为显示器)
    //一个output对应多个mode (分辨率), output中只保存mode id
    //多个output 的mode都保存在screen modes里面
    output_count = 0;
    for (int i = 0; i < ccreen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*pOutputInfo = XRRGetOutputInfo(g_display_ptr, ccreen_res_ptr, ccreen_res_ptr->outputs[i]);
        if (pOutputInfo->connection == 0)
        {
            //拷贝output名字
            snprintf(temp_list_ptr[output_count].name, COMMON_SIZE_256, "%s", pOutputInfo->name);

            //获取output当前的分辨率
            XRRCrtcInfo *crtc_info_ptr = GetCrtcByXID(ccreen_res_ptr, pOutputInfo->crtc);
            temp_list_ptr[output_count].cur_resolution.width = crtc_info_ptr->width;
            temp_list_ptr[output_count].cur_resolution.height = crtc_info_ptr->height;
            FreeCrtcInfo(crtc_info_ptr);
            //获取output支持的分辨率
            int *repeat_mode_ptr = new int[pOutputInfo->nmode];
            TSize *size_list_ptr = new TSize[pOutputInfo->nmode];
            int resolution_count = 0;
            memset(repeat_mode_ptr, 0, pOutputInfo->nmode*sizeof(repeat_mode_ptr[0]));

            for (int j = 0; j < pOutputInfo->nmode; j++)
            {
                XRRModeInfo *jMode = nullptr;
                XRRModeInfo *kMode = nullptr;

                //跳过重复的mode
                if (repeat_mode_ptr[j] == 1)
                    continue;
                jMode = FindModeByXID(ccreen_res_ptr, pOutputInfo->modes[j]);

                //查找之后的mode是否有和jMode重复的
                for (int k = j; k < pOutputInfo->nmode; k++)
                {
                    kMode = FindModeByXID(ccreen_res_ptr, pOutputInfo->modes[k]);
                    //不重复则继续查找
                    if (strcmp(jMode->name, kMode->name) != 0)
                        continue;
                    else
                        repeat_mode_ptr[k] = 1;
                }
                //过滤掉隔行扫描的(1920x1080i 模式)
                if (!strchr(jMode->name, 'i'))
                {
                    size_list_ptr[resolution_count].width = jMode->width;
                    size_list_ptr[resolution_count].height = jMode->height;
                    resolution_count++;
                }
            }
            temp_list_ptr[output_count].count = resolution_count;
            temp_list_ptr[output_count].resolution_ptr = size_list_ptr;
            output_count++;
            SAFE_DELETE_ARRAY(repeat_mode_ptr);
        }
    }

    XRRFreeScreenResources(ccreen_res_ptr);

    TOutputResolutions *res_ptr = new TOutputResolutions;

    res_ptr->output_count = output_count;
    res_ptr->resolution_info_ptr = temp_list_ptr;
    *o_resolution_ptr = res_ptr;

    return er;
}// NOLINT(readability/fn_size)

ERetCode VARenderX11::SetScreenSize(XRROutputInfo *i_output_info_ptr, int i_width, int i_height)
{
    int output_count = 0;
    unsigned int screen_w = i_width, screen_h = i_height, screen_mw = 0, screen_mh = 0;

    //没有显示器接入,直接返回
    output_count = GetOutputConnectionCount();
    if (output_count == 0)
    {
        LOG_ERROR("has no output connected");
        return EM_RC_NO_SCREEN_CONNECTION;
    }

    XRRScreenResources *screen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);
    if (screen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return EM_RC_OP_NOT_SUPPORT;
    }

    for (int i = 0; i < screen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*output_info_ptr = XRRGetOutputInfo(g_display_ptr, screen_res_ptr, screen_res_ptr->outputs[i]);
        if (output_info_ptr->connection == 0 && strcmp(i_output_info_ptr->name, output_info_ptr->name))
        {

            //获取output当前的分辨率
            XRRCrtcInfo *crtc_info_ptr = GetCrtcByXID(screen_res_ptr, output_info_ptr->crtc);
            screen_w = screen_w > crtc_info_ptr->width ? screen_w : crtc_info_ptr->width;
            screen_h = screen_h > crtc_info_ptr->height ? screen_h : crtc_info_ptr->height;
        }
    }

    XRRSetCrtcConfig(g_display_ptr, screen_res_ptr, i_output_info_ptr->crtc, CurrentTime,
        0, 0, None, RR_Rotate_0, NULL, 0);

    int screen_id = DefaultScreen(g_display_ptr);
    double dpi = (25.4 * (double)(DisplayHeight(g_display_ptr, screen_id))) /
        (double)(DisplayHeightMM(g_display_ptr, screen_id));

    screen_mw = (25.4 * ((double)screen_w)) / dpi;
    screen_mh = (25.4 * ((double)screen_h)) / dpi;

    XRRSetScreenSize(g_display_ptr, m_wnd_root, screen_w, screen_h,
        screen_mw, screen_mh);

    XRRFreeScreenResources(screen_res_ptr);
    return EM_RC_OK;
}


/*******************************************************************************
* 函  数: SetCurResolution
* 描  述: 设置显示器的分辨率
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2017年08月25日
* 作  者: 邓旭光(skye)
********************************************************************************/
ERetCode VARenderX11::SetCurResolution(char *i_output_name_ptr, int i_width, int i_height)
{
    ERetCode er = EM_RC_OK;
    if (i_output_name_ptr == nullptr)
        return EM_RC_PARAM_ERR;

    XRRScreenResources *screen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);
    if (screen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return EM_RC_OP_NOT_SUPPORT;
    }

    int i = 0;
    for (i = 0; i < screen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*output_info_ptr = XRRGetOutputInfo(g_display_ptr, screen_res_ptr, screen_res_ptr->outputs[i]);
        if (output_info_ptr->connection == 0)
        {
            if (!strcmp(i_output_name_ptr, output_info_ptr->name))
            {
                XRRCrtcInfo *crtc_info_ptr = GetCrtcByXID(screen_res_ptr, output_info_ptr->crtc);
                RRMode xid_mode;
                int j = 0;
                for (j = 0; j < output_info_ptr->nmode; j++)
                {
                    XRRModeInfo *mode_info_ptr = nullptr;
                    mode_info_ptr = FindModeByXID(screen_res_ptr, output_info_ptr->modes[j]);
                    if (mode_info_ptr->height == (unsigned int)i_height && mode_info_ptr->width == (unsigned int)i_width)
                    {
                        xid_mode = mode_info_ptr->id;
                        break;
                    }
                }

                if (j == output_info_ptr->nmode)
                {
                    LOG_ERROR("unsurport resolution:%dx%d", i_width, i_height);
                    er = EM_RC_UNSUPORT_RESOLUTION;
                    break;
                }

                SetScreenSize(output_info_ptr, i_width, i_height);

                Status ret = XRRSetCrtcConfig(g_display_ptr, screen_res_ptr, output_info_ptr->crtc, CurrentTime,
                    crtc_info_ptr->x, crtc_info_ptr->y, xid_mode, RR_Rotate_0, &screen_res_ptr->outputs[i], 1);

                if (ret != RRSetConfigSuccess)
                {
                    er = EM_RC_OP_NOT_SUPPORT;
                }
                break;
            }
        }
    }

    if (i == screen_res_ptr->noutput)
    {
        LOG_ERROR("no output connection");
        er = EM_RC_NO_SCREEN_CONNECTION;
    }

    XRRFreeScreenResources(screen_res_ptr);
    return er;
}

/*******************************************************************************
* 函  数: GetOutputConnectionCount
* 描  述: 获取接入的显示器数量,如果有显示器接入则打开monitor
* 返回值: 接入的显示器数量
* 时  间: 2017年08月25日
* 作  者: 邓旭光(skye)
********************************************************************************/
int VARenderX11::GetOutputConnectionCount()
{
    int output_count = 0;
    XRRScreenResources *screen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);

    if (screen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return EM_RC_OP_NOT_SUPPORT;
    }

    for (int i = 0; i < screen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*output_info_ptr = XRRGetOutputInfo(g_display_ptr, screen_res_ptr, screen_res_ptr->outputs[i]);
        if (output_info_ptr->connection == 0)
        {
            LOG_INFO("%s connected", output_info_ptr->name);
            output_count++;
        }
    }
    XRRFreeScreenResources(screen_res_ptr);
    return output_count;
}

void VARenderX11::OpenConnectedMonitor()
{
    XRRScreenResources *screen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);
    if (screen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return;
    }

    for (int i = 0; i < screen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*output_info_ptr = XRRGetOutputInfo(g_display_ptr, screen_res_ptr, screen_res_ptr->outputs[i]);
        if (output_info_ptr->connection == 0)
        {
            char cmd_str[256] = { 0 };
            sprintf(cmd_str, "xrandr --output %s --auto", output_info_ptr->name);
            system(cmd_str);
        }
    }
    XRRFreeScreenResources(screen_res_ptr);
    return;
}

void VARenderX11::CloseUnConnectedMonitor()
{
    XRRScreenResources *screen_res_ptr = XRRGetScreenResources(g_display_ptr, m_wnd_root);
    if (screen_res_ptr == NULL)
    {
        LOG_ERROR("could not get screen resource\n");
        return;
    }

    for (int i = 0; i < screen_res_ptr->noutput; i++)
    {
        XRROutputInfo	*output_info_ptr = XRRGetOutputInfo(g_display_ptr, screen_res_ptr, screen_res_ptr->outputs[i]);
        if (output_info_ptr->connection != 0)
        {
            char cmd_str[256] = { 0 };
            sprintf(cmd_str, "xrandr --output %s --off", output_info_ptr->name);
            system(cmd_str);
        }
    }
    XRRFreeScreenResources(screen_res_ptr);
    return;
}

void VARenderX11::StartHotPlugMonitor()
{
    if (m_monitor_thread_state == EM_TS_RUN)
    {
        LOG_WARN("the monitor thread has been started");
        return;
    }

    m_monitor_thread_state = EM_TS_RUN;
    StartThreadEx(HotPlugMonitorThreadAdapter, this);
}

/*******************************************************************************
* 函  数: HotPlugMonitorThreadAdapter
* 描  述: 显示设备热插拔监测
* 返回值: 接入的显示器数量
* 时  间: 2017年08月25日
* 作  者: 邓旭光(skye)
********************************************************************************/
void *VARenderX11::HotPlugMonitorThreadAdapter(void *i_thread_arg_ptr)
{
    if (i_thread_arg_ptr == nullptr)
    {
        LOG_ERROR("invalid parameter");
        return NULL;
    }

    VARenderX11 *obj_ptr = static_cast<VARenderX11 *>(i_thread_arg_ptr);
    obj_ptr->HotPlugMonitorThread();
    return NULL;
}

/******************************************************************************
* 函  数: HotPlugMonitorThread
* 描  述: 显示设备热插拔监测线程
* 返回值: N/A
* 时  间: 2016年08月22日
* 作  者: 邓旭光(skye)
*******************************************************************************/
void VARenderX11::HotPlugMonitorThread()
{
    udev *udev_ptr = NULL;
    int fd = -1;
    fd_set fd_reads;
    struct udev_monitor *monitor_ptr; //NOLINT(legal/name)
    struct udev_device *device_ptr = NULL; //NOLINT(legal/name)
    int connect_count = 0;
    int ret = 0;
    timeval time_out;

    udev_ptr = udev_new();
    if (!udev_ptr)
    {
        LOG_ERROR("Can't create udev\n");
        return;
    }

    LOG_INFO("HotPlugMonitorThread begin...");

    monitor_ptr = udev_monitor_new_from_netlink(udev_ptr, "udev");
    udev_monitor_filter_add_match_subsystem_devtype(monitor_ptr, "drm", NULL);
    udev_monitor_enable_receiving(monitor_ptr);
    fd = udev_monitor_get_fd(monitor_ptr);
    while (m_monitor_thread_state == EM_TS_RUN)
    {
        time_out.tv_sec = 1;
        time_out.tv_usec = 0;
        FD_ZERO(&fd_reads);
        FD_SET(fd, &fd_reads);
        ret = select(fd + 1, &fd_reads, NULL, NULL, &time_out);
        switch (ret)
        {
        case 0:
        case -1:
            continue;
            break;
        default:
            device_ptr = udev_monitor_receive_device(monitor_ptr);
            const char *action_ptr = udev_device_get_action(device_ptr);
            if (!strcmp(action_ptr, "change"))
            {
                connect_count = GetOutputConnectionCount();
                LOG_INFO("screen changed, ConnectionCount[%d]", connect_count);
                //如果没有显示器接入,则停止渲染
                if (connect_count == 0)
                {
                    StopThread();
                }
                else
                {
                    //使能monitor
                    OpenConnectedMonitor();
                    //关闭没有连接的monitor
                    CloseUnConnectedMonitor();

                    //判断是否已经初始化,如果没有，则进行初始化操作
                    if (!m_is_inited)
                    {
                        TMediaRenderProps render_props;
                        render_props.media_mode = EM_MEDIA_MODE_HARD;
                        render_props.size = m_tmp_size;
                        if (m_tmp_rect.height != 0 && m_tmp_rect.width != 0)
                        {
                            render_props.rect = m_tmp_rect;
                            CreateRender(&render_props);
                        }
                        else
                            CreateRender(&render_props);
                    }
                    //有显示器接入，开始渲染
                    else if (m_exit_code == EM_TS_STOP)
                        StartThread();
                }
            }
            break;
        }
    }
    udev_unref(udev_ptr);
    m_monitor_thread_state = EM_TS_STOP;
    LOG_INFO("HotPlugMonitorThread end...");
}// NOLINT(readability/fn_size)
