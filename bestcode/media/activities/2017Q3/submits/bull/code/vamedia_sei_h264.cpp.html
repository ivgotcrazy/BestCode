#include <stdio.h>
#include <stdlib.h> // malloc
#include <string.h> // memset

#include "vamedia_bitstream.h"
#include "vamedia_stream_h264.h"
#include "vamedia_sei_h264.h"

/*------------------------------------------------------------------------------
* 描  述：SEI信息分配
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
SEI* NewSEI()
{
    SEI* s = (SEI*)malloc(sizeof(SEI));
    memset(s, 0, sizeof(SEI));
    s->payload_ptr = NULL;
    return s;
}

/*------------------------------------------------------------------------------
* 描  述：释放SEI信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void FreeSEI(SEI* s_ptr)
{
    if ( s_ptr->payload_ptr != NULL ) free(s_ptr->payload_ptr);
    free(s_ptr);
}

/*------------------------------------------------------------------------------
* 描  述：读取SEI位信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void ReadSEIEndBits(H264Stream* h_ptr, BS* b_ptr )
{
    // if the message doesn't end at a byte border
    if ( !BSByteAligned( b_ptr ) )
    {
        if ( !BSReadU1( b_ptr ) ) fprintf(stderr, "WARNING: bit_equal_to_one is 0!!!!\n");
        while ( ! BSByteAligned( b_ptr ) )
        {
            if ( BSReadU1( b_ptr ) ) fprintf(stderr, "WARNING: bit_equal_to_zero is 1!!!!\n");
        }
    }

    ReadRBSPTrailingBits(h_ptr, b_ptr);
}

// D.1 SEI payload syntax
/*------------------------------------------------------------------------------
* 描  述：读取SEI负载信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void ReadSEIPayload(H264Stream* h_ptr, BS* b_ptr, int payload_type, int payload_size)
{
    SEI* s_ptr = h_ptr->sei;
    s_ptr->payload_ptr = (uint8_t*)malloc(payload_size);

    int i;
    for ( i = 0; i < payload_size; i++ )
        s_ptr->payload_ptr[i] = BSReadU(b_ptr, 8);
        
    ReadSEIEndBits(h_ptr, b_ptr);
}

// D.1 SEI payload syntax
/*------------------------------------------------------------------------------
* 描  述：写SEI位信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void WriteSEIPayload(H264Stream* h_ptr, BS* b_ptr, int payload_type, int payload_size)
{
    SEI* s = h_ptr->sei;

    int i;
    for ( i = 0; i < s->payload_size; i++ )
        BSWriteU(b_ptr, 8, s->payload_ptr[i]);
}
