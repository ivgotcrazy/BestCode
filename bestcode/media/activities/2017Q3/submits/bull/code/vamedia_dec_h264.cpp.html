VADecH264::VADecH264()
{    
    m_h264_info_ptr = NewH264Stream();  
    m_src_props_ptr = new TSourceProps;

    pthread_mutex_init(&m_decode_mtx, nullptr);
    pthread_mutex_init(&m_out_frame_mtx, nullptr);
    pthread_cond_init(&m_out_frame_cond, nullptr);
}

VADecH264::~VADecH264()
{    
    FreeH264Stream(m_h264_info_ptr);
    SAFE_DELETE(m_src_props_ptr);

    pthread_mutex_destroy(&m_decode_mtx);
    pthread_mutex_destroy(&m_out_frame_mtx);
    pthread_cond_destroy(&m_out_frame_cond);
}

/*******************************************************************************
* 函  数: InitVAPicture
* 描  述: 初始化VAPicture
* 参  数: i_pVAPic
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::InitVAPicture(VAPictureH264* i_va_pic_ptr)
{
    i_va_pic_ptr->picture_id          = VA_INVALID_ID;
    i_va_pic_ptr->flags               = VA_PICTURE_H264_INVALID;
    i_va_pic_ptr->TopFieldOrderCnt    = 0;
    i_va_pic_ptr->BottomFieldOrderCnt = 0;
    i_va_pic_ptr->frame_idx           = 0;

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: VAInitDecode
* 描  述: 初始化h264解码器
* 参  数: i_pDecInfo: 解码参数集
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月10日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VADecH264::CreateDecoder(TSourceProps* i_src_props_ptr)
{
    CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

    VAImageFormat va_img_format;
    va_img_format.fourcc = VA_FOURCC_NV12;    
    va_img_format.byte_order = VA_LSB_FIRST;
    va_img_format.bits_per_pixel = 12;

	m_src_size.width = AlignSize(i_src_props_ptr->src_size.width);
	m_src_size.height = AlignSize(i_src_props_ptr->src_size.height);
    m_dec_img_ptr = new VAImage;
	vaCreateImage(g_va_display, &va_img_format, i_src_props_ptr->src_size.width, i_src_props_ptr->src_size.height, m_dec_img_ptr);
        
    memcpy(m_src_props_ptr, i_src_props_ptr, sizeof(*m_src_props_ptr));
    LOG_INFO("Init EncSource[%d], SrcSize[%dx%d]", m_src_props_ptr->src_id, m_src_size.width, m_src_size.height);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: SetCallBack
* 描  述: 设置解码数据回调
* 参  数: i_pCallBack
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月07日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::SetDecDataCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_param_ptr)
{    
    CAutoLock lock(m_decode_mtx);   

    m_out_surf_id_list.clear();
    m_dec_count = 0;    
    
    if( i_callback_ptr != nullptr )
    {
        if( m_out_props_ptr != nullptr )
            SAFE_DELETE(m_out_props_ptr);

        m_out_props_ptr = new TDecOutProps();
        if( i_dec_param_ptr != nullptr )
        {// 按参数取解码数据
            m_set_out_props = 1;
            memcpy(m_out_props_ptr, i_dec_param_ptr, sizeof(TDecOutProps));//NOLINT

            if( m_out_props_ptr->frame_interval <= 0 )
                m_out_props_ptr->frame_interval = 1;
        }
        StartThread();
    }
    else
    {
        if( m_exit_code == EM_TS_RUN )
        {
            pthread_mutex_lock(&m_out_frame_mtx);
            m_exit_code = EM_TS_STOPING;     
            pthread_cond_signal(&m_out_frame_cond);  
            pthread_mutex_unlock(&m_out_frame_mtx);

            while(m_exit_code != EM_TS_STOP)
                usleep(1000);
        }

        SAFE_DELETE(m_out_props_ptr);
    }

    m_callback_ptr = i_callback_ptr;
    if( m_out_props_ptr != nullptr )
        LOG_INFO("Source[%d] Need DecData CallBack, Scale[%.3f], OutType[%d], Space[%d]",
                  m_src_props_ptr->src_id, m_out_props_ptr->scale,
                  m_out_props_ptr->out_type,m_out_props_ptr->frame_interval);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DestroyDecoder
* 描  述: 关闭h264解码器
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月08日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VADecH264::DestroyDecoder()
{
    if (m_exit_code == EM_TS_RUN)
    {
        pthread_mutex_lock(&m_out_frame_mtx);
        m_exit_code = EM_TS_STOPING; 
        pthread_cond_signal(&m_out_frame_cond);
        pthread_mutex_unlock(&m_out_frame_mtx);

        while (m_exit_code != EM_TS_STOP)
            usleep(1000);
    }

    SetDecDataCallBack(nullptr, nullptr);

    CAutoLock lock(m_decode_mtx);
    if( m_is_inited_vaapi )
    {// 初始化后才释放
        vaDestroySurfaces(g_va_display,&m_va_surface_id[0],VA_SURFACE_SIZE);
        vaDestroyBuffer(g_va_display,m_va_picture_id);
        vaDestroyBuffer(g_va_display,m_va_iq_matrix_id);
        vaDestroyContext(g_va_display,m_va_context_id);
        vaDestroyConfig(g_va_display,m_va_config_id);
    }

    if (m_dec_img_ptr != nullptr)
    {
        vaDestroyImage(g_va_display, m_dec_img_ptr->image_id);
        SAFE_DELETE(m_dec_img_ptr);
    }
    
    SAFE_DELETE_ARRAY(m_va_slice_param_ptr);    

    m_use_surf_id = 0;    
    m_is_inited_vaapi = 0;
    m_set_out_props = 0;
    m_dec_count = 0;    
    LOG_INFO("UnInit EncSource[%d]", m_src_props_ptr->src_id);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: InitVAAPI
* 描  述: 通过解码参数来设置vaapi
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月08日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VADecH264::InitVAAPI()
{
    VAStatus   va_status;
    VAEntrypoint* entry_point_ptr = NULL;
    int entry_pt_num = 0,val_entry_pt = 0;
    int max_display_attr = vaMaxNumDisplayAttributes(g_va_display);
    
    if( EM_RC_OK != GetVAProfile())
        return EM_RC_OP_NOT_SUPPORT;    

    entry_point_ptr = new VAEntrypoint[max_display_attr];
    va_status = vaQueryConfigEntrypoints(g_va_display, m_va_profile, entry_point_ptr, &entry_pt_num);
    CHECK_VASTATUS(va_status, "vaQueryConfigEntrypoints");

    for (val_entry_pt = 0; val_entry_pt < entry_pt_num; val_entry_pt++)
    {
        if (entry_point_ptr[val_entry_pt] == VAEntrypointVLD)
            break;
    }
    if (val_entry_pt == entry_pt_num)// not find VLD entry point 
        return EM_RC_DEC_NOTFIND_ENTRYPOINT;
    
    m_va_attrib.type = VAConfigAttribRTFormat;
    vaGetConfigAttributes(g_va_display, m_va_profile, VAEntrypointVLD,&m_va_attrib, 1);
    if ((m_va_attrib.value & VA_RT_FORMAT_YUV420) == 0)// not find desired YUV420 RT format 
        return EM_RC_DEC_SUPPORT_YUV420;

    va_status = vaCreateConfig(g_va_display, m_va_profile, VAEntrypointVLD,&m_va_attrib, 1,&m_va_config_id);
    CHECK_VASTATUS(va_status, "vaQueryConfigEntrypoints");

    va_status = vaCreateSurfaces(g_va_display,VA_RT_FORMAT_YUV420, 
        m_src_size.width, m_src_size.height, &m_va_surface_id[0], VA_SURFACE_SIZE, NULL, 0);
    CHECK_VASTATUS(va_status, "vaCreateSurfaces");

    /* Create a context for this decode pipe */
    va_status = vaCreateContext(g_va_display, m_va_config_id, m_src_size.width, m_src_size.height,
        VA_PROGRESSIVE,&m_va_surface_id[0],VA_SURFACE_SIZE,&m_va_context_id);
    CHECK_VASTATUS(va_status, "vaCreateContext");

    va_status = vaCreateBuffer(g_va_display, m_va_context_id,VAPictureParameterBufferType,
        sizeof(VAPictureParameterBufferH264), 1, NULL, &m_va_picture_id);
    CHECK_VASTATUS(va_status, "vaCreateBuffer_VAPictureParameterBufferType");

    va_status = vaCreateBuffer(g_va_display, m_va_context_id,VAIQMatrixBufferType,
        sizeof(VAIQMatrixBufferH264), 1, NULL, &m_va_iq_matrix_id);
    CHECK_VASTATUS(va_status, "vaCreateBuffer_VAIQMatrixBufferType");
        
    m_va_slice_param_ptr = new VASliceParameterBufferH264[m_h264_info_ptr->num_slices];
    SAFE_DELETE_ARRAY(entry_point_ptr);
    
    LOG_INFO("Source[%d] Init VAAPI Success, Slices[%d]", m_src_props_ptr->src_id, m_h264_info_ptr->num_slices);
    m_is_inited_vaapi = 1; 
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ParseH264
* 描  述: 解析H264信息
* 参  数: i_pH264Data
* 参  数: i_iDataLen
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::ParseH264(uint8_t* i_h264_data_ptr,int i_data_len)
{       
    int nal_size = i_data_len;
    int nal_start = 0;
    int nal_end = 0;
    int nal_offset = 0;
    int find_ret = 0,read_ret = 0;    
    uint8_t* parse_buf_ptr = i_h264_data_ptr;
    m_h264_info_ptr->num_slices = 0;

    do
    {       
        find_ret = FindNalUnit(parse_buf_ptr, nal_size, &nal_start, &nal_end);            
        parse_buf_ptr += nal_start;  
        nal_offset += nal_start;
        read_ret = ReadNalUnit(m_h264_info_ptr, parse_buf_ptr, nal_end - nal_start);
        if(read_ret < 0)
            return EM_RC_FILE_READ_ERR;

        if( m_h264_info_ptr->nal->nal_unit_type == NAL_UNIT_TYPE_CODED_SLICE_IDR || 
            m_h264_info_ptr->nal->nal_unit_type == NAL_UNIT_TYPE_CODED_SLICE_NON_IDR || 
            m_h264_info_ptr->nal->nal_unit_type == NAL_UNIT_TYPE_CODED_SLICE_AUX)
        {
            m_h264_info_ptr->slice_data_offset[m_h264_info_ptr->num_slices] = nal_offset;
            m_h264_info_ptr->num_slices++;
        }

        nal_offset += nal_end-nal_start;
        if(find_ret > 0)
        {
            parse_buf_ptr += (nal_end - nal_start);
            nal_size -= nal_end;    
        }
    }
    while(find_ret > 0);    

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: VADecodeData
* 描  述: 解码函数
* 参  数: i_pSrcData: 源数据指针
* 参  数: i_iDataLen: 源数据大小 
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月09日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VADecH264::DecodeData(uint8_t* i_src_data_ptr,int i_data_len, uint8_t** o_dst_data_ptr)
{
	if (o_dst_data_ptr != nullptr)
		return EM_RC_STATE_ERR;

    ERetCode er = ParseH264(i_src_data_ptr,i_data_len);
    if(er != EM_RC_OK)
        return er;
    
    if( !m_is_inited_vaapi )
    {// 从sps中获取profile信息，初始化vaapi用来解码        
        er = InitVAAPI();
        if(er != EM_RC_OK)
            return er;
    }

    VASurfaceID va_surface_id = m_va_surface_id[m_use_surf_id];
    VAStatus va_status = vaMapBuffer(g_va_display, m_va_picture_id, (void**)&m_va_pic_param_ptr);
    CHECK_VASTATUS(va_status, "vamapBuffer(PictureParam)");  
    va_status = vaMapBuffer(g_va_display, m_va_iq_matrix_id, (void**)&m_va_iq_matrix_ptr);
    CHECK_VASTATUS(va_status, "vamapBuffer(IQMatrix)");  
    
    // 更新引用帧信息(pic和slice)
    UpdatePicRefList();        
    // 更新pic参数
    UpdateH264PicParam();
    // 更新slice参数
    UpdateSliceParam();    
    
    // 更新当前帧的picture_id
    m_va_pic_param_ptr->CurrPic.picture_id = va_surface_id;

    int i = 0;       
    VABufferID va_slice_param_id = 0, va_slice_data_id = 0;
    for( i = 0; i < m_h264_info_ptr->num_slices; i++ )
    {// 创建slice param 和data 的映射        
        va_status = vaCreateBuffer(g_va_display, m_va_context_id,VASliceParameterBufferType, 
                                  sizeof(VASliceParameterBufferH264), 
                                  1, (void*)&m_va_slice_param_ptr[i],&va_slice_param_id);
        CHECK_VASTATUS(va_status, "vaCreateSliceParamBuffer");
        va_status = vaCreateBuffer(g_va_display, m_va_context_id,VASliceDataBufferType, 
                                  m_va_slice_param_ptr[i].slice_data_size, 1, 
                                  i_src_data_ptr+m_h264_info_ptr->slice_data_offset[i], &va_slice_data_id);
        CHECK_VASTATUS(va_status, "vaCreateSliceDataBuffer");

        m_va_slices_id[i*2] = va_slice_param_id;
        m_va_slices_id[i*2+1] = va_slice_data_id;
    }
     
    va_status = vaUnmapBuffer(g_va_display, m_va_picture_id);
    CHECK_VASTATUS(va_status, "vaUnmapBuffer(PictureParam)");    
    va_status = vaUnmapBuffer(g_va_display, m_va_iq_matrix_id);
    CHECK_VASTATUS(va_status, "vaUnmapBuffer(IQMatrix)");    
    
    // 执行解码操作
    RunDecodeTask(va_surface_id);
        
    CAutoLock lock( m_decode_mtx );     
    if( m_callback_ptr != nullptr && m_out_props_ptr != nullptr )
    {// 存在回调，进list，通知线程下载数据
        if( m_dec_count%m_out_props_ptr->frame_interval == 0 )
        {
            pthread_mutex_lock(&m_out_frame_mtx);
            m_out_surf_id_list.push_back(va_surface_id);
            pthread_cond_signal(&m_out_frame_cond);            
            pthread_mutex_unlock(&m_out_frame_mtx);
        }   
        m_dec_count++;
    }

    for( i = 0; i < m_h264_info_ptr->num_slices; i++ )
    {
        vaDestroyBuffer(g_va_display,m_va_slices_id[i*2]);
        vaDestroyBuffer(g_va_display,m_va_slices_id[i*2+1]);
    }
    
    m_use_surf_id++;
    if(m_use_surf_id >= VA_SURFACE_SIZE)
        m_use_surf_id = 0;
    return EM_RC_OK;
}//NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: CalcCurrPoc
* 描  述: 计算当前帧的POC
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月22日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::CalcCurrPoc()
{
    SliceHeader* sh_ptr = m_h264_info_ptr->shs[0];
    if (m_h264_info_ptr->sps->pic_order_cnt_type == 2)
    {// 最简单的模式
        if (sh_ptr->slice_type == SH_SLICE_TYPE_I || sh_ptr->slice_type == SH_SLICE_TYPE_SI ||
            sh_ptr->slice_type == SH_SLICE_TYPE_I_ONLY || sh_ptr->slice_type == SH_SLICE_TYPE_SI_ONLY)
        {// I帧清零
            m_frame_num_offset = 0;
            m_pic_order_cnt = 0;// poc = tmp poc    
            m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt = m_pic_order_cnt;
            m_va_pic_param_ptr->CurrPic.BottomFieldOrderCnt = m_pic_order_cnt;
        }
        else
        {// p帧计算
            int max_frame_num = 1 << (m_h264_info_ptr->sps->log2_max_frame_num_minus4 + 4);
            if (sh_ptr->DecRefPicMarking.memory_management_control_operation)
            {// mmco非0，重置
                m_pre_frame_num = 0;
                m_pre_frame_num_offset = 0;
            }
            if (sh_ptr->frame_num < m_pre_frame_num)
                m_frame_num_offset = m_pre_frame_num_offset + max_frame_num;
            else
                m_frame_num_offset = m_pre_frame_num_offset;

            int tmp_frame_num = m_frame_num_offset + sh_ptr->frame_num;
            if (m_h264_info_ptr->nal->nal_ref_idc == 0)
                m_pic_order_cnt = (2 * tmp_frame_num - 1);
            else
                m_pic_order_cnt = (2 * tmp_frame_num);

            if (sh_ptr->field_pic_flag == 0)
                m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt = m_va_pic_param_ptr->CurrPic.BottomFieldOrderCnt = m_pic_order_cnt;
            else if (sh_ptr->bottom_field_flag == 0)
                m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt = m_pic_order_cnt;
            else
                m_va_pic_param_ptr->CurrPic.BottomFieldOrderCnt = m_pic_order_cnt;
        }

        /*if( m_pH264Info->nal->nal_ref_idc == 0 )*/
        m_pre_frame_num = sh_ptr->frame_num;
        m_pre_frame_num_offset = m_frame_num_offset;
    }
    else
    {
        int max_pic_order_cnt_lsb = (1 << (m_h264_info_ptr->sps->log2_max_pic_order_cnt_lsb));
        if (sh_ptr->pic_order_cnt_lsb  <  m_prev_pic_order_cnt_lsb && 
            (m_prev_pic_order_cnt_lsb - sh_ptr->pic_order_cnt_lsb) >= (max_pic_order_cnt_lsb / 2))
            m_pic_order_cnt_msb = m_prev_pic_order_cnt_msb + max_pic_order_cnt_lsb;
        else if (sh_ptr->pic_order_cnt_lsb > m_prev_pic_order_cnt_lsb && 
            (sh_ptr->pic_order_cnt_lsb - m_prev_pic_order_cnt_lsb) > (max_pic_order_cnt_lsb / 2))
            m_pic_order_cnt_msb = m_prev_pic_order_cnt_msb - max_pic_order_cnt_lsb;
        else
            m_pic_order_cnt_msb = m_prev_pic_order_cnt_msb;

        if (sh_ptr->field_pic_flag == 0)
        {//frame pix
            m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt = m_pic_order_cnt_msb + sh_ptr->pic_order_cnt_lsb;
            m_va_pic_param_ptr->CurrPic.BottomFieldOrderCnt = 
                m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt + sh_ptr->delta_pic_order_cnt_bottom;
        }
        else if (sh_ptr->bottom_field_flag == 0)  //top field
            m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt = m_pic_order_cnt_msb + sh_ptr->pic_order_cnt_lsb;
        else          //bottom field
            m_va_pic_param_ptr->CurrPic.BottomFieldOrderCnt = m_pic_order_cnt_msb + sh_ptr->pic_order_cnt_lsb;

        if (m_h264_info_ptr->nal->nal_ref_idc)
        {
            m_prev_pic_order_cnt_lsb = sh_ptr->pic_order_cnt_lsb;
            m_prev_pic_order_cnt_msb = m_pic_order_cnt_msb;
        }
    }
    return EM_RC_OK;
}// NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: UpdatePicRefList
* 描  述: 设置VA引用列表
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::UpdatePicRefList()
{
    SliceHeader* sh_ptr = m_h264_info_ptr->shs[0];
    if(sh_ptr->slice_type == SH_SLICE_TYPE_I || sh_ptr->slice_type == SH_SLICE_TYPE_I_ONLY)
    {// I帧，重置参数和引用帧列表
        m_prev_pic_order_cnt_lsb = 0;
        m_prev_pic_order_cnt_msb = 0;        
        m_have_long_term_ref_pic = 0;

        int i = 0; 
        for( i = 0; i < VA_SURFACE_SIZE; i++ )
            InitVAPicture(&m_va_pic_param_ptr->ReferenceFrames[i]);
            
        for( i = 0; i < m_h264_info_ptr->num_slices; i++ )
        {
            for( int j = 0; j < VA_SURFACE_SIZE; j++ )
            {
                InitVAPicture(&m_va_slice_param_ptr[i].RefPicList0[j]);
                InitVAPicture(&m_va_slice_param_ptr[i].RefPicList1[j]);
            }
        }
        if(sh_ptr->DecRefPicMarking.long_term_reference_flag)
        {// 长引用标志，保存长引用帧
            m_have_long_term_ref_pic = true;
            m_va_long_term_ref_pic = m_va_pic_param_ptr->CurrPic;
            m_va_long_term_ref_pic.BottomFieldOrderCnt = m_va_long_term_ref_pic.TopFieldOrderCnt = 0;
            m_va_long_term_ref_pic.flags = VA_PICTURE_H264_LONG_TERM_REFERENCE; 
            m_va_long_term_ref_pic.frame_idx = sh_ptr->DecRefPicMarking.long_term_frame_idx;
        }
    }
    else
    { 
        if( m_h264_info_ptr->num_ref_l0_frames > 0 )
        {// 存在引用帧(p帧), 参考ffmpeg，只设置pic的Ref0（用上一个解码帧)
            m_va_pic_param_ptr->CurrPic.flags = VA_PICTURE_H264_SHORT_TERM_REFERENCE;            
            UpdatePicRefLiset();
            UpdateSliceRefList();          
        }

        if (sh_ptr->DecRefPicMarking.memory_management_control_operation)
        {
            if (sh_ptr->bottom_field_flag)
            {
                m_prev_pic_order_cnt_lsb = 0;
                m_prev_pic_order_cnt_msb = 0;
            }
            else
            {
                m_prev_pic_order_cnt_lsb = m_va_pic_param_ptr->CurrPic.TopFieldOrderCnt;
                m_prev_pic_order_cnt_msb = 0;
            }
        }
    }
    // 计算poc
    CalcCurrPoc();  
    m_va_pic_param_ptr->CurrPic.frame_idx = sh_ptr->frame_num;
    m_va_pic_param_ptr->CurrPic.flags = 0;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UpdateSliceParam
* 描  述: 更新Slice参数
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::UpdateSliceParam()
{
    CHECK_POINTER(m_h264_info_ptr, EM_RC_PARAM_ERR);
    for( int i = 0; i < m_h264_info_ptr->num_slices; i++ )
    {
        SliceHeader* sh_ptr = m_h264_info_ptr->shs[i];
        m_va_slice_param_ptr[i].slice_data_bit_offset          = sh_ptr->slice_data_bit_offset;
        m_va_slice_param_ptr[i].slice_data_size                = sh_ptr->slice_data_size;
        m_va_slice_param_ptr[i].slice_data_offset              = 0;
        m_va_slice_param_ptr[i].slice_data_flag                = VA_SLICE_DATA_FLAG_ALL;
        m_va_slice_param_ptr[i].first_mb_in_slice              = sh_ptr->first_mb_in_slice;
        m_va_slice_param_ptr[i].slice_type                     = sh_ptr->slice_type >= 5 ? 
                                                            sh_ptr->slice_type - 5 : sh_ptr->slice_type;
        m_va_slice_param_ptr[i].direct_spatial_mv_pred_flag    = sh_ptr->direct_spatial_mv_pred_flag;
        m_va_slice_param_ptr[i].num_ref_idx_l0_active_minus1   = m_h264_info_ptr->num_ref_l0_frames ? 
                                                            m_h264_info_ptr->num_ref_l0_frames-1 : 0;
        m_va_slice_param_ptr[i].num_ref_idx_l1_active_minus1   = m_h264_info_ptr->num_ref_l1_frames ? 
                                                            m_h264_info_ptr->num_ref_l1_frames-1 : 0;
        m_va_slice_param_ptr[i].cabac_init_idc                 = sh_ptr->cabac_init_idc;
        m_va_slice_param_ptr[i].slice_qp_delta                 = sh_ptr->slice_qp_delta;
        m_va_slice_param_ptr[i].disable_deblocking_filter_idc  = sh_ptr->disable_deblocking_filter_idc;
        m_va_slice_param_ptr[i].slice_alpha_c0_offset_div2     = sh_ptr->slice_alpha_c0_offset_div2;
        m_va_slice_param_ptr[i].slice_beta_offset_div2         = sh_ptr->slice_beta_offset_div2;
        m_va_slice_param_ptr[i].luma_log2_weight_denom         = sh_ptr->PredWeightTable.luma_log2_weight_denom;
        m_va_slice_param_ptr[i].chroma_log2_weight_denom       = sh_ptr->PredWeightTable.chroma_log2_weight_denom;
        m_va_slice_param_ptr[i].luma_weight_l0_flag            = sh_ptr->PredWeightTable.luma_weight_l0_flag[0];
        m_va_slice_param_ptr[i].chroma_weight_l0_flag          = sh_ptr->PredWeightTable.chroma_weight_l0_flag[0];
        m_va_slice_param_ptr[i].luma_weight_l1_flag            = sh_ptr->PredWeightTable.luma_weight_l1_flag[0];
        m_va_slice_param_ptr[i].chroma_weight_l1_flag          = sh_ptr->PredWeightTable.chroma_weight_l1_flag[0];

        memset(m_va_slice_param_ptr[i].luma_weight_l0, 0, 32*sizeof(short));//NOLINT()
        memset(m_va_slice_param_ptr[i].luma_offset_l0, 0, 32*sizeof(short));//NOLINT()
        memset(m_va_slice_param_ptr[i].chroma_weight_l0, 0, 2*32*sizeof(short));//NOLINT()
        memset(m_va_slice_param_ptr[i].chroma_offset_l0, 0, 2*32*sizeof(short));//NOLINT()
        for (int j = 0; j < m_h264_info_ptr->num_ref_l0_frames; j++)
        {
            if (sh_ptr->PredWeightTable.luma_weight_l0_flag[j])
            {
                m_va_slice_param_ptr[i].luma_weight_l0[j] = sh_ptr->PredWeightTable.luma_weight_l0[j];
                m_va_slice_param_ptr[i].luma_offset_l0[j] = sh_ptr->PredWeightTable.luma_offset_l0[j];
            }
            else
            {
                m_va_slice_param_ptr[i].luma_weight_l0[j] = 1 << sh_ptr->PredWeightTable.luma_log2_weight_denom;
                m_va_slice_param_ptr[i].luma_offset_l0[j] = 0;
            }
            if (sh_ptr->PredWeightTable.chroma_weight_l0_flag[j])
            {
                m_va_slice_param_ptr[i].chroma_weight_l0[j][0] = sh_ptr->PredWeightTable.chroma_weight_l0[j][0];
                m_va_slice_param_ptr[i].chroma_offset_l0[j][0] = sh_ptr->PredWeightTable.chroma_offset_l0[j][0];
                m_va_slice_param_ptr[i].chroma_weight_l0[j][1] = sh_ptr->PredWeightTable.chroma_weight_l0[j][1];
                m_va_slice_param_ptr[i].chroma_offset_l0[j][1] = sh_ptr->PredWeightTable.chroma_offset_l0[j][1];
            }
            else
            {
                m_va_slice_param_ptr[i].chroma_weight_l0[j][0] = 1 << sh_ptr->PredWeightTable.chroma_log2_weight_denom; 
                m_va_slice_param_ptr[i].chroma_offset_l0[j][0] = 0;
                m_va_slice_param_ptr[i].chroma_weight_l0[j][1] = 1 << sh_ptr->PredWeightTable.chroma_log2_weight_denom;
                m_va_slice_param_ptr[i].chroma_offset_l0[j][1] = 0;
            }   
        }
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UpdatePicRefLiset
* 描  述: 更新Pic的引用列表
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月21日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::UpdatePicRefLiset()
{
    // 计算pic的参考帧列表
    m_h264_info_ptr->num_ref_frames++;
    if (m_h264_info_ptr->num_ref_frames > m_h264_info_ptr->sps->num_ref_frames)
        m_h264_info_ptr->num_ref_frames = m_h264_info_ptr->sps->num_ref_frames;

    // 处理短引用
    for( int i = m_h264_info_ptr->num_ref_frames - 1; i > 0; i-- )
        m_va_pic_param_ptr->ReferenceFrames[i] = m_va_pic_param_ptr->ReferenceFrames[i - 1];
    m_va_pic_param_ptr->ReferenceFrames[0] = m_va_pic_param_ptr->CurrPic;
        
    // 存在长引用
    if (m_have_long_term_ref_pic)
        m_va_pic_param_ptr->ReferenceFrames[m_h264_info_ptr->num_ref_frames - 1] = m_va_long_term_ref_pic;

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UpdateSliceRefList
* 描  述: 更新slice的引用帧列表
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::UpdateSliceRefList()
{
    CHECK_POINTER(m_h264_info_ptr, EM_RC_PARAM_ERR);

    // 设置引用帧, 暂不支持b帧，即list1
    if( m_h264_info_ptr->num_ref_l0_frames <= 0 )
        return EM_RC_OP_NOT_SUPPORT;
       
    // 计算slice的引用列表，不考虑b帧(即list1)    
    for(int i = 0; i < m_h264_info_ptr->num_slices; i++)
    {// slice reflist 应用的是slice信息中的num_ref_l0_frames, 可能被sliceheader中的override参数覆盖新值
        SliceHeader* sh_ptr = m_h264_info_ptr->shs[i];
        if (sh_ptr->RefPicListRecorder.ref_pic_list_reordering_flag_l0)
        {// 需要进行参考帧重排序
            int ref_pic_num_l0pred = sh_ptr->frame_num;   // 计算初值为当前帧序号
            uint32_t ref_frame_num = 0;                   // 计算的引用帧序号
            int max_frame_num = (1 << (m_h264_info_ptr->sps->log2_max_frame_num_minus4+4));    // 最大帧序号            
            for (int j = 0; j < m_h264_info_ptr->num_ref_l0_frames; j++)
            { // 循环计算参考帧序号
                if (ref_pic_num_l0pred - sh_ptr->RefPicListRecorder.abs_diff_pic_num[j] < 0)
                    ref_frame_num = ref_pic_num_l0pred - sh_ptr->RefPicListRecorder.abs_diff_pic_num[j] + max_frame_num;
                else
                    ref_frame_num = ref_pic_num_l0pred - sh_ptr->RefPicListRecorder.abs_diff_pic_num[j];
                
                for (int k = 0; k < m_h264_info_ptr->num_ref_frames; k++)
                {// 从pic的reflist中查找对应序号的引用帧，保存到slice的引用列表中
                    if (ref_frame_num == m_va_pic_param_ptr->ReferenceFrames[k].frame_idx)
                    {// 复制引用帧(VAPictureH264
                        m_va_slice_param_ptr[i].RefPicList0[j] = m_va_pic_param_ptr->ReferenceFrames[k];
                        break;
                    }
                }
                
                // 预测帧需要使用上一个参考帧序号
                ref_pic_num_l0pred = ref_frame_num;
            }
        }
        else
        {// 不需要重新排序，
            m_va_slice_param_ptr[i].RefPicList0[0] = m_va_pic_param_ptr->CurrPic;
            for( int j = m_h264_info_ptr->num_ref_l0_frames - 1; j > 0; j-- )
                m_va_slice_param_ptr[i].RefPicList0[j] = m_va_slice_param_ptr[i].RefPicList0[j - 1];
        }        
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RunDecodeTask
* 描  述: 执行解码操作
* 参  数: i_vaSurfID
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::RunDecodeTask(VASurfaceID i_va_surf_id)
{
    VAStatus va_status = VA_STATUS_SUCCESS;
    va_status = vaBeginPicture(g_va_display, m_va_context_id, i_va_surf_id); 
    CHECK_VASTATUS(va_status, "vaBeginPicture");
    
    va_status = vaRenderPicture(g_va_display,m_va_context_id, &m_va_picture_id, 1);
    CHECK_VASTATUS(va_status, "vaRenderPicture");
    
    va_status = vaRenderPicture(g_va_display,m_va_context_id, &m_va_iq_matrix_id, 1);
    CHECK_VASTATUS(va_status, "vaRenderPicture");

    va_status = vaRenderPicture(g_va_display,m_va_context_id, m_va_slices_id, m_h264_info_ptr->num_slices*2);
    CHECK_VASTATUS(va_status, "vaRenderPicture");

    CHECK_VASTATUS(va_status, "vaEndPicture");

    va_status = vaSyncSurface(g_va_display, i_va_surf_id);

/*******************************************************************************
* 函  数: ThreadFuncProc
* 描  述: 获取
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月07日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::ThreadFuncProc()
{
    timespec tm_cond;    
    TOutFrame out_frame;
    TDecOutProps dec_out_props;

    VASurfaceID va_tmp_surf_id = 0;
    VASurfaceID va_out_surf_id = 0;
    VAImage va_surf_img, va_out_img;
    VAImageFormat va_img_format;    
    
    int out_frame_size = 0; 
    int src_data_size = m_src_props_ptr->src_size.width*m_src_props_ptr->src_size.height * 3 / 2;

    va_out_img.image_id = 0;
    va_surf_img.image_id = 0;
    
    va_img_format.fourcc = VA_FOURCC_NV12;
    va_img_format.byte_order = VA_LSB_FIRST;
    va_img_format.bits_per_pixel = 12;
    vaCreateImage(g_va_display, &va_img_format, m_src_props_ptr->src_size.width,m_src_props_ptr->src_size.height, &va_surf_img);

    LOG_INFO("EncSource[%d] Start DecOutFrame Thread!", m_src_props_ptr->src_id);
    while(m_exit_code == EM_TS_RUN)
    {   
        pthread_mutex_lock(&m_out_frame_mtx);
        
        if( m_callback_ptr == nullptr || m_out_surf_id_list.size() <= 0 )
        {// 无数据时等待
            tm_cond.tv_nsec = 0;
            tm_cond.tv_sec = (long)time(nullptr) + COMMON_TICK_THREADTMO;

            pthread_cond_timedwait(&m_out_frame_cond, &m_out_frame_mtx, &tm_cond);
            pthread_mutex_unlock(&m_out_frame_mtx);    
            continue;
        }

        va_out_surf_id = m_out_surf_id_list.front();
        m_out_surf_id_list.pop_front();
        pthread_mutex_unlock(&m_out_frame_mtx);    
         
        if( !m_set_out_props )        
        {// 取所有数据
            if (out_frame.frame_data_ptr == nullptr || out_frame.data_len != src_data_size)
            {
                SAFE_DELETE_ARRAY(out_frame.frame_data_ptr);
                out_frame.data_len = src_data_size;
                out_frame.width = m_src_props_ptr->src_size.width;
                out_frame.height = m_src_props_ptr->src_size.height;
                out_frame.frame_data_ptr = new uint8_t[out_frame.data_len];
            }

            pthread_mutex_lock(&m_decode_mtx);
            vaGetImage(g_va_display, va_out_surf_id, 0, 0,
                m_src_props_ptr->src_size.width, m_src_props_ptr->src_size.height, va_surf_img.image_id);
            pthread_mutex_unlock(&m_decode_mtx);
            DownYUVDataFromImage(&va_surf_img, out_frame.frame_data_ptr);
        }
        else
        {
            out_frame.width = (int)(m_src_props_ptr->src_size.width*m_out_props_ptr->scale);
            out_frame.height = (int)(m_src_props_ptr->src_size.height*m_out_props_ptr->scale);
            if (m_out_props_ptr->out_type) // y
                out_frame_size = out_frame.width*out_frame.height;
            else // yuv
                out_frame_size = out_frame.width*out_frame.height * 3 / 2;

            if (out_frame.data_len != out_frame_size)
            {// 参数变化
                SAFE_DELETE_ARRAY(out_frame.frame_data_ptr);
                out_frame.data_len = out_frame_size;
                out_frame.frame_data_ptr = new uint8_t[out_frame.data_len];

                if (va_out_img.image_id)
                {
                    vaDestroyImage(g_va_display, va_out_img.image_id);
                    va_out_img.image_id = 0;
                }
                if (va_tmp_surf_id)
                {
                    vaDestroySurfaces(g_va_display, &va_tmp_surf_id, 1);
                    va_tmp_surf_id = 0;
                }

                vaCreateImage(g_va_display, &va_img_format, out_frame.width, out_frame.height, &va_out_img);
                vaCreateSurfaces(g_va_display, VA_RT_FORMAT_YUV420, AlignSize(out_frame.width),
                    AlignSize(out_frame.height), &va_tmp_surf_id, 1, NULL, 0);
            }

            // 从surface取出image,注意lock，避免和GetReadyImage接口冲突
            pthread_mutex_lock(&m_decode_mtx);
            vaGetImage(g_va_display, va_out_surf_id, 0, 0,
                m_src_props_ptr->src_size.width, m_src_props_ptr->src_size.height, va_surf_img.image_id);
            pthread_mutex_unlock(&m_decode_mtx);
            // 把image put到目标尺寸的surface上，执行缩放
            vaPutImage(g_va_display, va_tmp_surf_id, va_surf_img.image_id, 0, 0,
                m_src_props_ptr->src_size.width, m_src_props_ptr->src_size.height, 0, 0, out_frame.width, out_frame.height);
            // 获取缩放后的image
            vaGetImage(g_va_display, va_tmp_surf_id, 0, 0, out_frame.width, out_frame.height, va_out_img.image_id);

            // 下载Y或yuv数据
            if (m_out_props_ptr->out_type)
                DownYDataFromImage(&va_out_img, out_frame.frame_data_ptr);
            else
                DownYUVDataFromImage(&va_out_img, out_frame.frame_data_ptr);
        }

        m_callback_ptr->OnDecodeCallBack(&out_frame, (RBSUserData)m_src_props_ptr->src_id);
    }

    if (va_surf_img.image_id)
        vaDestroyImage(g_va_display,va_surf_img.image_id);
    if (va_out_img.image_id)
        vaDestroyImage(g_va_display,va_out_img.image_id);
    if (va_tmp_surf_id)
        vaDestroySurfaces(g_va_display,&va_tmp_surf_id,1);
    
    SAFE_DELETE_ARRAY(out_frame.frame_data_ptr);
    LOG_INFO("EncSource[%d] Exit DecOutFrame Thread!", m_src_props_ptr->src_id);
    m_exit_code = EM_TS_STOP;
    return EM_RC_OK;
}// NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: GetProfile
* 描  述: profile_idc转换为VA的profile定义
* 时  间: 17年03月18日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADecH264::GetVAProfile()
{    
    m_va_profile = VAProfileNone;

    if(m_h264_info_ptr->sps->profile_idc == PROFILE_IDC_BASELINE)
        m_va_profile = VAProfileH264Baseline;
    else if(m_h264_info_ptr->sps->profile_idc == PROFILE_IDC_MAIN)
        m_va_profile = VAProfileH264Main;
    else if(m_h264_info_ptr->sps->profile_idc == PROFILE_IDC_HIGH)
        m_va_profile = VAProfileH264High;
    else
        return EM_RC_OP_NOT_SUPPORT;
   
    return EM_RC_OK;
}
