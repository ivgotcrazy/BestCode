	VAMixerSource();

public: // ISource抽象接口

	ERetCode CreateSource(TSourceProps* i_src_props_ptr);

	ERetCode InputSrcData(uint8_t* i_src_data_ptr, int i_data_len) { return EM_RC_OK; };

	ERetCode DestroySource();

	ERetCode SetSubTitle(TTextInfo* i_text_info_ptr);

	ERetCode RunEffect();

	ERetCode SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_props_ptr){ return EM_RC_OK; }

    int             m_effect_frame_num = 0;                        //特效帧数

    pthread_mutex_t m_readyImage_mtx;                              // 图像准备锁
    pthread_cond_t  m_readyimage_cond;                             // 图像准备条件变量
