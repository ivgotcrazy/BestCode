#include "media_encoder.h"

/********************************** 类实现 ***********************************/

/*------------------------------------------------------------------------------
* 描  述：创建编码器
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaEncoder::CreateEncoder(TEncInfo* i_enc_props_ptr)
{
	CHECK_POINTER(i_enc_props_ptr, EM_RC_PARAM_ERR);

	std::unique_lock<std::mutex> lock(m_encoder_mtx);
	if (m_encoder_ptr != nullptr)
	{
		m_encoder_ptr->DestroyEncoder();
		SAFE_DELETE(m_encoder_ptr);
	}

	if (i_enc_props_ptr->media_mode == EM_MEDIA_MODE_SOFT)
		m_encoder_ptr = new EncodeVideo();
	else
		m_encoder_ptr = new VAEncH264();

	CHECK_POINTER(m_encoder_ptr, EM_RC_MEMROY_ALLOC_ERR);

	if ( m_encoder_ptr->CreateEncoder(i_enc_props_ptr) != EM_RC_OK )
	{
		LOG_ERROR("Stream [%d] Init Encoder Failed");
		SAFE_DELETE(m_encoder_ptr);
		return EM_RC_INIT_ERR;
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：编码数据
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaEncoder::EncodeData(uint8_t* i_data_ptr, uint32_t i_data_len, TOutFrame* o_data_ptr /* = nullptr */)
{
	CHECK_POINTER(i_data_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(m_encoder_ptr, EM_RC_INIT_ERR);

	std::unique_lock<std::mutex> lock(m_encoder_mtx);

	return m_encoder_ptr->EncodeData(i_data_ptr, i_data_len, o_data_ptr);
}

/*------------------------------------------------------------------------------
* 描  述：设置输入源， 如果已存在，会覆盖原有的输入源
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaEncoder::SetDataSource(ISource* i_source_ptr)
{
	CHECK_POINTER(i_source_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(m_encoder_ptr, EM_RC_INIT_ERR);

	std::unique_lock<std::mutex> lock(m_encoder_mtx);

	return m_encoder_ptr->SetDataSource(i_source_ptr);
}

/*------------------------------------------------------------------------------
* 描  述： 设置编码数据回调对象
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaEncoder::SetEncDataCallBack(IVCallBack* i_callback_ptr)
{
	CHECK_POINTER(i_callback_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(m_encoder_ptr, EM_RC_INIT_ERR);

	std::unique_lock<std::mutex> lock(m_encoder_mtx);

	return m_encoder_ptr->SetEncDataCallBack(i_callback_ptr);
}

/*------------------------------------------------------------------------------
* 描  述： 销毁编码器
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaEncoder::DestroyEncoder()
{
	CHECK_POINTER(m_encoder_ptr, EM_RC_INIT_ERR);

	std::unique_lock<std::mutex> lock(m_encoder_mtx);

	 m_encoder_ptr->DestroyEncoder();
	 SAFE_DELETE(m_encoder_ptr);

	 return EM_RC_OK;
}
