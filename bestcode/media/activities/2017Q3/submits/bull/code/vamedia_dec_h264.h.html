class VADecH264 : public BaseThread, public IDecoder
{
public:

	VADecH264();
	virtual ~VADecH264();

	// 初始化解码功能模块
	ERetCode CreateDecoder(TSourceProps* i_src_props_ptr);

	// 设置解码数据回调对象
	ERetCode SetDecDataCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_param_ptr);

	//// 设置解码数据渲染器，nullptr表示关闭
	ERetCode SetRender(/*TRenderInfo* i_pRenderInfo*/){ return EM_RC_OK; };

	// 输入编码数据进行解码(一般为了做智能分析才会回调出去，现在直接用VAImage)
	//解码数据直接在o_pDstData输出，不需要用回调接口，硬解里面暂不使用此接口
	ERetCode DecodeData(uint8_t* i_src_data_ptr, int i_data_len, uint8_t** o_dst_data_ptr = nullptr);

	// 反初始化解码功能模块
	ERetCode DestroyDecoder();

	void LockReadyImage(){};
	VideoSrcImage GetReadyImage();
	void UnLockReadyImage(){};

	TSourceProps* GetSrcProps() { return m_src_props_ptr; };

private:

	ERetCode ParseH264(uint8_t* i_h264_data_ptr, int i_data_len);

	ERetCode InitVAAPI();
	ERetCode InitVAPicture(VAPictureH264* i_va_pic_ptr);

	ERetCode GetVAProfile();

	ERetCode UpdatePicRefList();
	ERetCode UpdateH264PicParam();
	ERetCode UpdateSliceParam();
	ERetCode UpdatePicRefLiset();
	ERetCode UpdateSliceRefList();

	ERetCode CalcCurrPoc();
	ERetCode RunDecodeTask(VASurfaceID i_va_surf_id);
	ERetCode ThreadFuncProc();

private:

	VAConfigID      m_va_config_id = 0;                                
	VAContextID     m_va_context_id = 0;                                
	VASurfaceID     m_va_surface_id[VA_SURFACE_SIZE] = { 0 };

	VAImage*        m_dec_img_ptr = nullptr;                            //解码VAImage
	VAProfile       m_va_profile = VAProfileNone;                       //profile等级

	VABufferID      m_va_picture_id = 0;                                //缓冲区ID
	VABufferID      m_va_iq_matrix_id = 0;                              //渲染缓冲区ID
	VABufferID      m_va_slices_id[VA_SURFACE_SIZE] = { 0 };      
	VAPictureH264   m_va_long_term_ref_pic;                             
	VAConfigAttrib  m_va_attrib;                                        //Config属性                           

	VAIQMatrixBufferH264*          m_va_iq_matrix_ptr = nullptr;        //量化矩阵参数
	VAPictureParameterBufferH264*  m_va_pic_param_ptr = nullptr;        //图像参数
	VASliceParameterBufferH264*    m_va_slice_param_ptr = nullptr;      //slice参数

	int             m_prev_pic_order_cnt_msb = 0;                       
	int             m_prev_pic_order_cnt_lsb = 0;                       
	int             m_pic_order_cnt_msb = 0;                            
	int             m_pic_order_cnt_lsb = 0;                            
	int             m_pic_order_cnt = 0;                                
	int             m_pre_frame_num = 0;                                
	int             m_pre_frame_num_offset = 0;                         
	int             m_frame_num_offset = 0;                             //帧数据位移

	int             m_have_long_term_ref_pic = 0;                       //是否有参考图像
	int             m_use_surf_id = 0;                                  //当前使用surf索引
	int             m_is_inited_vaapi = 0;                              //是否初始化vaapi
	int             m_set_out_props = 0;                                //设置解码输出标识
	uint64_t        m_dec_count = 0;                                    //解码计数

	TSize           m_src_size;                                         //输入数据大小
	H264Stream*     m_h264_info_ptr = nullptr;                          //264码流分析器
	TDecOutProps*   m_out_props_ptr = nullptr;                          //解码输出参数
	TSourceProps*   m_src_props_ptr = nullptr;                          //数据源参数
	IVCallBack*     m_callback_ptr = nullptr;                           //解码回调句柄

	std::list<VASurfaceID>  m_out_surf_id_list;                         //surf列表

	pthread_mutex_t m_decode_mtx;                                       //解码锁
	pthread_mutex_t m_out_frame_mtx;                                    //帧数据锁
	pthread_cond_t  m_out_frame_cond;                                   //帧数据条件
};
