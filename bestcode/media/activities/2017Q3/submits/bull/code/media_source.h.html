#pragma once

/**************************** 条件编译选项和头文件 ****************************/

#include "RetDefine.h"
#include "isource.h"

/**********************************  类声明  ***********************************/

class RBSMediaSource : public ISource
{
public:

	//////////////////////////////////非合成源相关抽象接口////////////////////////////////
	~RBSMediaSource(){};

	// 创建源对象
	ERetCode CreateSource(TSourceProps* i_src_props_ptr);

	// 输入数据
	ERetCode InputSrcData(uint8_t* i_src_data_ptr, int i_data_len);

	// 销毁源对象
	ERetCode DestroySource();

	//设置台标和字幕
	// uint8_t*为nullptr表示删除, 反之则添加
	ERetCode SetSubTitle(TTextInfo* i_text_info_ptr);
	ERetCode SetLogo(TLogoInfo* i_logo_info_ptr);

	// 设置特效
	ERetCode SetEffect(EEffectType i_effect);

	//运行特效
	ERetCode RunEffect();

	// 获取转换好的源Image( VAImage )
	void       LockReadyImage(){ return; };
	VideoSrcImage GetReadyImage(TOutFrame* o_data_ptr = nullptr);
	void       UnLockReadyImage(){ return; };

	ERetCode RegisterMediaProc(EMediaMode i_media_mode);

	ERetCode SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_props_ptr);

	// 获取源属性
	TSourceProps* GetSrcProps();

private:
	ISource* m_source_ptr = nullptr;          //数据源
};

class MediaMixerSource : public IMixerSource
{
	//////////////////////////////////合成源相关抽象接口////////////////////////////////
public:

	~MediaMixerSource(){};

	// 创建源对象
	ERetCode CreateSource(TSourceProps* i_src_props_ptr);

	// 输入数据
	ERetCode InputSrcData(uint8_t* i_src_data_ptr, int i_data_len);

	// 销毁源对象
	ERetCode DestroySource();

	//设置台标和字幕
	// uint8_t*为nullptr表示删除, 反之则添加
	ERetCode SetSubTitle(TTextInfo* i_text_info_ptr);
	ERetCode SetLogo(TLogoInfo* i_logo_info_ptr);

	// 设置特效
	ERetCode SetEffect(EEffectType i_effect);

	//运行特效
	ERetCode RunEffect();

	// 获取转换好的源Image( VAImage )
	void       LockReadyImage();
	VideoSrcImage GetReadyImage(TOutFrame* o_data_ptr = nullptr);
	void       UnLockReadyImage();

	//合成源不使用此接口
	ERetCode SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_props_ptr);

	ERetCode RegisterMediaProc(EMediaMode i_media_mode);

	// 获取源属性
	TSourceProps* GetSrcProps();

	// 添加数据源   
	ERetCode AddSource(int i_src_id, TProcItem* i_item_ptr);

	// 移除输入源
	ERetCode DelSource(int i_src_id);

	// 修改成员输出位置
	ERetCode ModifySrcOutRect(int i_src_id, TRect* i_rect_ptr);

	// 修改成员层级
	ERetCode ModifySrcLevel(int i_src_id, EVMixLevel i_level);

	// 修改输入源
	ERetCode ModifySrcSource(int i_src_id, ISource* i_source_ptr);

	//修改合成源大小
	ERetCode ModifyMixerSize(TSize* i_size_ptr);

private:
	IMixerSource* m_mixer_source_ptr = nullptr;        //合成数据源
};
