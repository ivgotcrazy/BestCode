VAMixerSource::VAMixerSource()
{
    m_src_props_ptr = new TSourceProps;
    pthread_mutex_init(&m_mixer_mtx, nullptr);
    pthread_mutex_init(&m_readyImage_mtx, nullptr);
}

VAMixerSource::~VAMixerSource()
{
    SAFE_DELETE(m_src_props_ptr);
    pthread_mutex_destroy(&m_mixer_mtx);
    pthread_mutex_destroy(&m_readyImage_mtx);
}

/*******************************************************************************
* 函  数: Init
* 描  述: 初始化合成器
* 参  数: i_pSrcProps: 合成参数
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月07日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::CreateSource(TSourceProps* i_src_props_ptr)
{
    CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);
    if (m_src_props_ptr->src_id != -1)
    {
        LOG_ERROR("MixSource[%d] Init!, Please UnInit", m_src_props_ptr->src_id);
        return EM_RC_INITED_ERR;
    }

    ERetCode er = CreateSurface(i_src_props_ptr);
    if (er != EM_RC_OK)
    {
        LOG_ERROR("MixSource[%d] Create Surface Fail[%d]", i_src_props_ptr->src_id, er);
        return er;
    }

    vaCreateConfig(g_va_display, VAProfileNone, VAEntrypointVideoProc, nullptr, 0, &m_va_config_id);
    memcpy(m_src_props_ptr, i_src_props_ptr, sizeof(*m_src_props_ptr));

    m_watermark_ptr = new MediaTool(g_rbs_Log_ptr);

    m_exit_code = EM_TS_RUN;
    LOG_INFO("Init MixerSource[%d], Size[%dx%d]", m_src_props_ptr->src_id,
        m_src_props_ptr->src_size.width, m_src_props_ptr->src_size.height);
    return StartThread();
}

/*******************************************************************************
* 函  数: CreateSurface
* 描  述: 创建合成surface和链表(Image)
* 参  数: i_pSrcProps
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月17日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::CreateSurface(TSourceProps* i_src_props_ptr)
{
    // 创建合成、特效渲染的Surface
    VASurfaceAttrib va_surf_attr;
    va_surf_attr.type = VASurfaceAttribPixelFormat;
    va_surf_attr.flags = VA_SURFACE_ATTRIB_SETTABLE;
    va_surf_attr.value.type = VAGenericValueTypeInteger;
    va_surf_attr.value.value.i = VA_FOURCC_NV12;

    vaCreateSurfaces(g_va_display, VA_RT_FORMAT_YUV420, AlignSize(i_src_props_ptr->src_size.width),
        AlignSize(i_src_props_ptr->src_size.height), &m_va_surface_id[0], VA_SOURCE_RING_SIZE, &va_surf_attr, 1);
    vaCreateSurfaces(g_va_display, VA_RT_FORMAT_YUV420, AlignSize(i_src_props_ptr->src_size.width),
        AlignSize(i_src_props_ptr->src_size.height), &m_va_effSurf_id, 1, &va_surf_attr, 1);

    // 创建特效Image
    VAImageFormat img_fmt;
    img_fmt.fourcc = VA_FOURCC_NV12;
    img_fmt.byte_order = VA_LSB_FIRST;
    img_fmt.bits_per_pixel = 12;
    vaCreateImage(g_va_display, &img_fmt, i_src_props_ptr->src_size.width, i_src_props_ptr->src_size.height, &m_va_mix_image);

    // 重置特效标志
    m_effect_frame_num = 0;
    m_effect_state = EM_ES_END;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DestroySurface
* 描  述: 销毁合成surface和image
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月09日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::DestroySurface()
{
    vaDestroySurfaces(g_va_display, &m_va_effSurf_id, 1);
    vaDestroySurfaces(g_va_display, &m_va_surface_id[0], VA_SOURCE_RING_SIZE);

    if (m_va_effect_img.image_id)
        vaDestroyImage(g_va_display, m_va_effect_img.image_id);
    vaDestroyImage(g_va_display, m_va_mix_image.image_id);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UnInit
* 描  述: 释放资源
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月10日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMixerSource::DestroySource()
{
    if (m_exit_code != EM_TS_STOP)
    {
        m_exit_code = EM_TS_STOPING;
        while (m_exit_code != EM_TS_STOP)
            usleep(0);
    }

    CAutoLock lock(m_mixer_mtx);
    auto map_node = m_mixer_map.begin();
    for (; map_node != m_mixer_map.end(); map_node++)
    {// 清空合成源对象参数
        TMixerMember* mixer_member_ptr = map_node->second;
        SAFE_DELETE(mixer_member_ptr->tmp_ptr);
        SAFE_DELETE(mixer_member_ptr);
    }
    m_mixer_map.clear();

    SAFE_DELETE(m_watermark_ptr);

    DestroySurface();
    vaDestroyConfig(g_va_display, m_va_config_id);

    m_use_surf_id = 0;
    m_cur_src_count = 0;
    m_is_has_title = m_is_has_logo = 0;
    LOG_INFO("UnInit MixerSource[%d]", m_src_props_ptr->src_id);
    return EM_RC_OK;
}

ERetCode VAMixerSource::LoadLogo()
{
    SAFE_DELETE_ARRAY(m_wm_data_ptr[1]);
    SAFE_DELETE_ARRAY(m_wm_data_ptr[2]);
    m_is_has_logo = 0;

    if (m_logo_info.pos)
    {// 存在台标，且未初始化
        if (EM_RC_OK != m_watermark_ptr->DecodePNG(m_logo_info.logo_path,
            &m_wm_data_ptr[1], &m_wm_data_ptr[2], &m_wm_rect[1].width, &m_wm_rect[1].height))
            return EM_RC_DEC_FAILED;
        
        // 计算台标位置
        if (m_logo_info.pos == RBS_WM_POS_LT)//左上
            m_wm_rect[1].iX = m_wm_rect[1].iY = COMMON_SIZE_4;
        else if (m_logo_info.pos == RBS_WM_POS_RT)//右上
        {
            m_wm_rect[1].iX = m_subtile_info.width - m_wm_rect[1].width - COMMON_SIZE_4;
            m_wm_rect[1].iY = COMMON_SIZE_4;
        }
        else if (m_logo_info.pos == RBS_WM_POS_LB)//左下
        {
            m_wm_rect[1].iX = COMMON_SIZE_4;
            m_wm_rect[1].iY = m_subtile_info.height - m_wm_rect[1].height - COMMON_SIZE_4;
        }
        else if (m_logo_info.pos == RBS_WM_POS_RB)//右下
        {
            m_wm_rect[1].iX = m_subtile_info.width - m_wm_rect[1].width - COMMON_SIZE_4;
            m_wm_rect[1].iY = m_subtile_info.height - m_wm_rect[1].height - COMMON_SIZE_4;
        }

        LOG_INFO("SetLogo Pos[%dx%d], Size[%dx%d]",
            m_wm_rect[1].iX, m_wm_rect[1].iY, m_wm_rect[1].width, m_wm_rect[1].height);

        m_is_has_logo = 1;
    }
    return EM_RC_OK;
}

ERetCode VAMixerSource::LoadSubtitle()
{
    if (!m_subtile_info.is_time_changed)
    {
        SAFE_DELETE_ARRAY(m_wm_data_ptr[0]);
        if (m_subtile_info.pos)// 存在字幕,计算字幕位置
        {
            // 原来的Size值存储的是英文字符个数(汉字会*2)        
            int char_count = m_subtile_info.font_size;
            m_subtile_info.font_size = m_subtile_info.width / COMMON_SIZE_32;

            // 20是时间字符串的字符个数
            int char_max_len = (char_count + 1)*m_subtile_info.font_size / 2;
            if (m_subtile_info.iIsShowTime)
                char_max_len = MAX_RBS(char_count + 1, 20) * m_subtile_info.font_size / 2;

            // ffmpeg的drawtext filter，宽度必须是64的整数倍
            m_wm_rect[0].width = ((char_max_len + COMMON_SIZE_64 - 1) / COMMON_SIZE_64)*COMMON_SIZE_64;
            int left_width = m_subtile_info.width - (m_logo_info.pos ? m_wm_rect[1].width : 0);
            if (m_wm_rect[0].width > left_width)
            {// 字幕叠加的宽度超过视频宽度，缩小字体
                m_wm_rect[0].width = (left_width / COMMON_SIZE_64)*COMMON_SIZE_64;
                m_subtile_info.font_size = m_wm_rect[0].width * 2 / char_count;
                m_subtile_info.font_size = (m_subtile_info.font_size / 2) * 2;
            }

            // 楷体的宽高比约为1:0.7,且宽高必须为偶数
            m_wm_rect[0].height = m_subtile_info.font_size*4.5;
            m_wm_rect[0].height = ((m_wm_rect[0].height + 1) / 2) * 2;

            if (m_subtile_info.pos == RBS_WM_POS_LT)//左上
            {
                m_wm_rect[0].iX = m_subtile_info.is_convert_logo ? m_wm_rect[1].width : 0;
                m_wm_rect[0].iY = 0;
            }
            else if (m_subtile_info.pos == RBS_WM_POS_RT)//右上
            {
                m_wm_rect[0].iX = m_subtile_info.width-m_wm_rect[0].width-(m_subtile_info.is_convert_logo?m_wm_rect[1].width:0);
                m_wm_rect[0].iY = 0;
            }
            else if (m_subtile_info.pos == RBS_WM_POS_LB)//左下
            {
                m_wm_rect[0].iX = m_subtile_info.is_convert_logo ? m_wm_rect[1].width : 0;
                m_wm_rect[0].iY = m_subtile_info.height - m_wm_rect[0].height;
            }
            else if (m_subtile_info.pos == RBS_WM_POS_RB)//右下
            {
                m_wm_rect[0].iX = m_subtile_info.width-m_wm_rect[0].width-(m_subtile_info.is_convert_logo?m_wm_rect[1].width:0);
                m_wm_rect[0].iY = m_subtile_info.height - m_wm_rect[0].height;
            }

            m_watermark_ptr->SetSubTitle(&m_subtile_info, m_wm_rect[0].width, m_wm_rect[0].height);
			m_wm_data_ptr[0] = new uint8_t[m_wm_rect[0].width*m_wm_rect[0].height * 3 / 2];

            // 背景白色
            memset(m_wm_data_ptr[0], 0xFF, m_wm_rect[0].width*m_wm_rect[0].height);
            memset(m_wm_data_ptr[0] + m_wm_rect[0].width*m_wm_rect[0].height, 0x80, m_wm_rect[0].width*m_wm_rect[0].height / 2);

            if (!m_subtile_info.iIsShowTime)
            {// 第一帧叠加不上字幕
                m_watermark_ptr->SubTitleFilter(m_wm_data_ptr[0]);
                m_watermark_ptr->SubTitleFilter(m_wm_data_ptr[0]);
            }
            LOG_INFO("Set SubTitle CharCount[%d], FontSize[%d], Pos[%dx%d], Size[%dx%d]",
                char_count,m_subtile_info.font_size,m_wm_rect[0].iX,m_wm_rect[0].iY,m_wm_rect[0].width,m_wm_rect[0].height);
            m_subtile_info.font_size = char_count;

            m_is_has_title = 1;
        }
        else
            m_is_has_title = 0;
    }
    else
    {
        if (m_subtile_info.pos && m_subtile_info.iIsShowTime)
        {
            memset(m_wm_data_ptr[0], 0xFF, m_wm_rect[0].width*m_wm_rect[0].height);
            memset(m_wm_data_ptr[0] + m_wm_rect[0].width*m_wm_rect[0].height, 0x80, m_wm_rect[0].width*m_wm_rect[0].height / 2);
            m_watermark_ptr->SubTitleFilter(m_wm_data_ptr[0]);
        }
    }

    return EM_RC_OK;
}//NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: RunEffect
* 描  述: 运行特效
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RunEffect()
{
    CAutoLock lockMix(m_mixer_mtx);
    if (m_effect_state != EM_ES_END)
        return EM_RC_OP_NOT_SUPPORT;

    m_effect_state = EM_ES_READY;
    if (m_cur_effect == EM_EFFECT_RANDOM)
    {
        m_tmp_effect = EM_EFFECT_RANDOM;
        m_cur_effect = (EEffectType)(EM_EFFECT_FADE + 1 + stGetCurrentMicroSecond() % (EM_EFFECT_RANDOM - 2));
    }

    LOG_INFO("Run Effect, Type[%d]", m_cur_effect);
    m_effect_frame_num = 0;
    m_effect_state = EM_ES_RUN;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: AddSource
* 描  述: 增加合成源
* 参  数: i_iSrcID
* 参  数: i_pItem
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月09日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::AddSource(int i_src_id, TProcItem* i_item_ptr)
{
    CHECK_POINTER(i_item_ptr, EM_RC_PARAM_ERR);

    CAutoLock lock(m_mixer_mtx);
    if (m_cur_src_count >= m_max_src_count)
    {
        LOG_ERROR("Reach Max VideoMixer Source Count[%d/%d]", m_cur_src_count, m_max_src_count);
        return EM_RC_SIZE_NOTENOUGH;
    }
	 
    TMixerMember* mixer_member_ptr = new TMixerMember;
    mixer_member_ptr->tmp_ptr = new TMixerMember;
    mixer_member_ptr->rect = i_item_ptr->rect;
    mixer_member_ptr->source_ptr = i_item_ptr->source_ptr;
    mixer_member_ptr->level = i_item_ptr->level;
    mixer_member_ptr->mix_state = EM_SS_ACTIVING;

    m_mixer_map.insert(MapMixerMember::value_type(i_src_id, mixer_member_ptr));

    m_cur_src_count++;
    LOG_INFO("Add Source[%d] To MixSource[%d], Level[%d], Pos[%d,%d], Size[%dx%d], Count[%d/%d]",
        i_src_id, m_src_props_ptr->src_id, mixer_member_ptr->level, mixer_member_ptr->rect.iX, mixer_member_ptr->rect.iY,
        mixer_member_ptr->rect.width, mixer_member_ptr->rect.height, m_cur_src_count, m_max_src_count);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: VARemoveSource
* 描  述: 移除指定的合成源
* 参  数: i_iSrcID
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月24日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMixerSource::DelSource(int i_src_id)
{
    if (m_cur_src_count < 0 || m_cur_src_count >= m_max_src_count)
        return EM_RC_PARAM_ERR;

    CAutoLock lock(m_mixer_mtx);
    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    TMixerMember* mixer_member_ptr = node_map->second;
    SAFE_DELETE(mixer_member_ptr->tmp_ptr);
    SAFE_DELETE(mixer_member_ptr);

    LOG_INFO("MixerSource[%d] Remove Source[%d]", m_src_props_ptr->src_id, i_src_id);
    m_mixer_map.erase(node_map);

    m_cur_src_count--;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ModifySrcOutRect
* 描  述: 修改布局方式
* 参  数: i_iSrcID: 源id
* 参  数: i_pOutRect: 新位置
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月07日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::ModifySrcOutRect(int i_src_id, TRect* i_rect_ptr)
{//暂时只支持切换布局，不支持增减流数、修改流类型。
    CHECK_POINTER(i_rect_ptr, EM_RC_PARAM_ERR);

    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    TMixerMember* mixer_member_ptr = node_map->second;
    if (mixer_member_ptr->mix_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    mixer_member_ptr->tmp_ptr->rect = *i_rect_ptr;
    mixer_member_ptr->mix_state |= EM_SS_MODIFY_POS;   // modify outrect                
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ModifySrcLevel
* 描  述: 修改层级
* 参  数: i_iSrcID
* 参  数: i_pDecode
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月07日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::ModifySrcLevel(int i_src_id, EVMixLevel i_level)
{
    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    TMixerMember* mixer_member_ptr = node_map->second;
    if (mixer_member_ptr->mix_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    mixer_member_ptr->tmp_ptr->level = i_level;
    mixer_member_ptr->mix_state |= EM_SS_MODIFY_LEVEL;
    return EM_RC_OK;
}

/*******************************************************************************

* 参  数: i_iSrcID
* 参  数: i_pSource
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::ModifySrcSource(int i_src_id, ISource* i_source_ptr)
{
    CHECK_POINTER(i_source_ptr, EM_RC_PARAM_ERR);
    auto node_map = m_mixer_map.find(i_src_id);
    if (node_map == m_mixer_map.end())
        return EM_RC_XML_NODE_NOTEXIST;

    // 修改source需要加锁，保证线程没有在执行合成
    CAutoLock lockMix(m_mixer_mtx);
    TMixerMember* mixer_member_ptr = node_map->second;
    if (mixer_member_ptr->mix_state == EM_SS_NONE)
        return EM_RC_PARAM_ERR;

    mixer_member_ptr->source_ptr = i_source_ptr;
    mixer_member_ptr->mix_state = EM_SS_ACTIVING;
    LOG_INFO("MixerSource[%d] Modify Source[%d] Input Source Object!", m_src_props_ptr->src_id, i_src_id);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: ThreadFuncProc
* 描  述: 定时合成线程
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月28日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::ThreadFuncProc()
{
    uint64_t count = 0;
    uint64_t tick = stGetCurrentMicroSecond();
    uint64_t frame_timestamp = COMMON_TICK_BASE / COMMON_DEFAULT_FPS;
    if (m_src_props_ptr->fps)
        frame_timestamp = COMMON_TICK_BASE / m_src_props_ptr->fps;

    LOG_INFO("MixerSource[%d] Start VideoMixerThread!", m_src_props_ptr->src_id);
    while (m_exit_code == EM_TS_RUN)
    {
        if (m_update_size == m_src_props_ptr->src_size)
            m_update_size.width = m_update_size.height = 0;

        if (m_update_size.width && m_update_size.height)
        {// 检查合成尺寸是否修改
            pthread_mutex_lock(&m_readyImage_mtx);
            m_src_props_ptr->src_size = m_update_size;
            m_update_size.width = m_update_size.height = 0;

            DestroySurface();
            CreateSurface(m_src_props_ptr);
            pthread_mutex_unlock(&m_readyImage_mtx);
            LOG_INFO("MixSource[%d] Change MixSize[%dx%d]", m_src_props_ptr->src_id,
                m_src_props_ptr->src_size.width, m_src_props_ptr->src_size.height);
        }

        pthread_mutex_lock(&m_mixer_mtx);
        if (!m_cur_src_count)
        {
            pthread_mutex_unlock(&m_mixer_mtx);
            usleep(COMMON_TICK_SECOND);
            continue;
        }

        // 执行检查map、合成、渲染
        CheckSourceState();
        RunMixerTask();
        RunEffectTask();

        m_use_surf_id++;
        if (m_use_surf_id >= VA_SOURCE_RING_SIZE)
            m_use_surf_id = 0;
        pthread_mutex_unlock(&m_mixer_mtx);

        RunWaterMarkTask();

        count++;//休眠        
        while (stGetCurrentMicroSecond() - tick < count * frame_timestamp)
            usleep(COMMON_TICK_SECOND);
    }

    m_exit_code = EM_TS_STOP;
    LOG_INFO("MixerSource[%d] Stop VideoMixerThread!", m_src_props_ptr->src_id);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: CheckSourceState
* 描  述: 检查输入源参数是否修改
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月08日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::CheckSourceState()
{
    TMixerMember* mixer_member_ptr = nullptr;
    auto node_map = m_mixer_map.begin();
    for (; node_map != m_mixer_map.end(); node_map++)
    {
        mixer_member_ptr = node_map->second;
        if (mixer_member_ptr->mix_state == EM_SS_ACTIVING)
            continue;

        int src_id = node_map->first;
        if (mixer_member_ptr->mix_state & EM_SS_MODIFY_LEVEL)
        {// 需要修改输出层级
            mixer_member_ptr->level = mixer_member_ptr->tmp_ptr->level;
            LOG_INFO("MixerSource[%d] Modify Source[%d] Level[%d]", m_src_props_ptr->src_id, src_id, mixer_member_ptr->level);
        }
        if (mixer_member_ptr->mix_state & EM_SS_MODIFY_POS)
        {// 需要修改输出位置
            mixer_member_ptr->rect = mixer_member_ptr->tmp_ptr->rect;
            LOG_INFO("MixerSource[%d] Modify Source[%d] Pos[%dx%d], Size[%dx%d]",
                m_src_props_ptr->src_id, src_id, mixer_member_ptr->rect.iX, mixer_member_ptr->rect.iY,
                mixer_member_ptr->rect.width, mixer_member_ptr->rect.height);
        }
        mixer_member_ptr->mix_state = EM_SS_ACTIVING;
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RunMixerTask
* 描  述: 执行合成操作
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年07月12日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAMixerSource::RunMixerTask()
{
    VAImage* image_ptr = nullptr;
    TMixerMember* mixer_member_ptr = nullptr;
    VASurfaceID va_surf_id = m_va_surface_id[m_use_surf_id];
    if (m_effect_state == EM_ES_RUN)
        va_surf_id = m_va_effSurf_id;

    auto node_map = m_mixer_map.begin();
    for (; node_map != m_mixer_map.end(); node_map++)
    {
        mixer_member_ptr = node_map->second;
        if (mixer_member_ptr->mix_state == EM_SS_ACTIVING &&
            mixer_member_ptr->source_ptr != nullptr && mixer_member_ptr->level == EM_VML_BOTTOM)
        {// 输入底层源image
            mixer_member_ptr->source_ptr->LockReadyImage();
            image_ptr = static_cast<VAImage *>(mixer_member_ptr->source_ptr->GetReadyImage());
            if (image_ptr != nullptr)
                vaPutImage(g_va_display, va_surf_id, image_ptr->image_id, 0, 0, image_ptr->width, image_ptr->height,
                mixer_member_ptr->rect.iX, mixer_member_ptr->rect.iY, mixer_member_ptr->rect.width, mixer_member_ptr->rect.height);
            mixer_member_ptr->source_ptr->UnLockReadyImage();
        }
    }

    node_map = m_mixer_map.begin();
    for (; node_map != m_mixer_map.end(); node_map++)
    {
        mixer_member_ptr = node_map->second;
        if (mixer_member_ptr->mix_state == EM_SS_ACTIVING &&
            mixer_member_ptr->source_ptr != nullptr && mixer_member_ptr->level == EM_VML_TOP)
        {// 输入顶层源image
            mixer_member_ptr->source_ptr->LockReadyImage();
            image_ptr = static_cast<VAImage *>(mixer_member_ptr->source_ptr->GetReadyImage());
            if (image_ptr != nullptr)
                vaPutImage(g_va_display, va_surf_id, image_ptr->image_id, 0, 0, image_ptr->width, image_ptr->height,
                mixer_member_ptr->rect.iX, mixer_member_ptr->rect.iY, mixer_member_ptr->rect.width, mixer_member_ptr->rect.height);
            mixer_member_ptr->source_ptr->UnLockReadyImage();
        }
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RunEffectTask
* 描  述: 执行特效渲染工作
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月17日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RunEffectTask()
{
    if (m_effect_state == EM_ES_RUN)
    {// 运行特效
        if (!m_effect_frame_num)// 取出特效模板
            vaDeriveImage(g_va_display, m_va_surface_id[m_use_surf_id], &m_va_effect_img);

        RenderEffect();
        if (m_effect_frame_num >= EFFECT_RUN_COUNT)
        {// 结束特效
            vaDestroyImage(g_va_display, m_va_effect_img.image_id);
            m_va_effect_img.image_id = 0;

            m_effect_frame_num = 0;
            m_effect_state = EM_ES_END;
            if (m_tmp_effect != EM_EFFECT_NONE)
            {
                m_cur_effect = m_tmp_effect;
                m_tmp_effect = EM_EFFECT_NONE;
            }
        }
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RenderEffect
* 描  述: 渲染特效
* 参  数: pVAImg
* 参  数: i_pSrcSize
* 参  数: i_pOutRect
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年04月01日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RenderEffect()
{
    VAImage va_image;
    VASurfaceID va_surf_id = m_va_surface_id[m_use_surf_id];
    vaDeriveImage(g_va_display, m_va_effSurf_id, &va_image);

    int width = m_src_props_ptr->src_size.width;
    int height = m_src_props_ptr->src_size.height;
    switch (m_cur_effect)
    {
    case EM_EFFECT_FADE:
        break;
    case EM_EFFECT_ERASEH:
    {
        int offset = EFFECT_EVEN(width*m_effect_frame_num / EFFECT_RUN_COUNT);
        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id, 0, 0, width, height, 0, 0, width, height);
        vaPutImage(g_va_display, va_surf_id, va_image.image_id, 0, 0, offset, height, 0, 0, offset, height);
    }
    break;
    case EM_EFFECT_ERASEV:
    {
        int offset = EFFECT_EVEN(height*m_effect_frame_num / EFFECT_RUN_COUNT);
        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id, 0, 0, width, height, 0, 0, width, height);
        vaPutImage(g_va_display, va_surf_id, va_image.image_id, 0, 0, width, offset, 0, 0, width, offset);
    }
    break;
    case EM_EFFECT_SHUTTERH:
    {
        int shutter = height / EFFECT_SHUTTER_COUNT;
        int offset = EFFECT_EVEN(m_effect_frame_num*shutter / EFFECT_RUN_COUNT);

        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id, 0, 0, width, height, 0, 0, width, height);
        for (int i = 0; i < EFFECT_SHUTTER_COUNT; i++)
        {// 12叶窗
            vaPutImage(g_va_display, va_surf_id, va_image.image_id,
                0, i*shutter, width, offset, 0, i*shutter, width, offset);
        }
    }
    break;
    case EM_EFFECT_SHUTTERV:
    {
        int shutter = width / EFFECT_SHUTTER_COUNT;
        int offset = EFFECT_EVEN(m_effect_frame_num*shutter / EFFECT_RUN_COUNT);
        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id, 0, 0, width, height, 0, 0, width, height);

        for (int i = 0; i < EFFECT_SHUTTER_COUNT; i++)
        {// 12叶窗
            vaPutImage(g_va_display, va_surf_id, va_image.image_id,
                i*shutter, 0, offset, height, i*shutter, 0, offset, height);
        }
    }
    break;
    case EM_EFFECT_BOXIN:
    {
        int box_w = 2 * (EFFECT_EVEN((width - m_effect_frame_num*width / EFFECT_RUN_COUNT) >> 1));
        int box_h = 2 * (EFFECT_EVEN((height - m_effect_frame_num*height / EFFECT_RUN_COUNT) >> 1));
        box_w = ALIGN_SIZE_32(box_w);
        int x_offset = (width - box_w) >> 1;
        int y_offset = (height - box_h) >> 1;

        vaPutImage(g_va_display, va_surf_id, va_image.image_id, 0, 0, width, height, 0, 0, width, height);
        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id,
            x_offset, y_offset, box_w, box_h, x_offset, y_offset, box_w, box_h);
    }
    break;
    case EM_EFFECT_BOXOUT:
    {
        int box_w = EFFECT_EVEN(m_effect_frame_num*width / EFFECT_RUN_COUNT);
        int box_h = EFFECT_EVEN(m_effect_frame_num*height / EFFECT_RUN_COUNT);
        box_w = ALIGN_SIZE_32(box_w);
        int x_offset = (width - box_w) >> 1;
        int y_offset = (height - box_h) >> 1;

        vaPutImage(g_va_display, va_surf_id, m_va_effect_img.image_id, 0, 0, width, height, 0, 0, width, height);
        vaPutImage(g_va_display, va_surf_id, va_image.image_id,
            0, 0, width, height, x_offset, y_offset, box_w, box_h);
    }
    break;
    default:
        break;
    }

    m_effect_frame_num++;
    vaDestroyImage(g_va_display, va_image.image_id);
    return EM_RC_OK;
}// NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: RunWaterMarkTask
* 描  述: 执行水印任务
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RunWaterMarkTask()
{
    pthread_mutex_lock(&m_readyImage_mtx);

    int surf_id = m_use_surf_id ? m_use_surf_id - 1 : VA_SOURCE_RING_SIZE - 1;
    vaGetImage(g_va_display, m_va_surface_id[surf_id], 0, 0, m_src_props_ptr->src_size.width,
        m_src_props_ptr->src_size.height, m_va_mix_image.image_id);

    if (m_is_has_logo + m_is_has_title)
    {
        uint8_t* image_buf = nullptr;
        vaMapBuffer(g_va_display, m_va_mix_image.buf, (void**)&image_buf);
        if (m_is_has_logo)
            RenderLogo(&m_va_mix_image, image_buf);
        if (m_is_has_title)
            RenderTitle(&m_va_mix_image, image_buf);
        vaUnmapBuffer(g_va_display, m_va_mix_image.buf);
    }

    pthread_mutex_unlock(&m_readyImage_mtx);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RenderLogo
* 描  述: 计算字幕透明混合
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月28日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RenderLogo(VAImage* va_image_ptr, uint8_t* image_buf)
{
    int x = m_wm_rect[1].iX, sub_x = x >> 1;
    int y = m_wm_rect[1].iY, sub_y = y >> 1;
    int w = m_wm_rect[1].width >> 1;
    int h = m_wm_rect[1].height >> 1;
    int size = w*h;

    int y_pitch = va_image_ptr->pitches[0];
    int u_pitch = va_image_ptr->pitches[1];
    int v_pitch = va_image_ptr->pitches[2];
    int u_offset = va_image_ptr->offsets[1];
    int v_offset = va_image_ptr->offsets[2];

    int y_start = x + y*y_pitch;
    int uv_start = x + sub_y*u_pitch + u_offset;

    int line = w * 4 * 2;
    int alpha[4] = { 0 };
    for (int i = 0; i < h; i++)
    {
        for (int j = 0; j < w; j++)
        {// 进行alpha混合
            int x_dst = y_start + i * 2 * y_pitch + 2 * j;
            int cur_pos = i*line * 2 + 8 * j;
            int x_src = i*m_wm_rect[1].width * 2 + 2 * j;
            alpha[0] = m_wm_data_ptr[1][cur_pos + 3];
            if (alpha[0])
                image_buf[x_dst] = ((image_buf[x_dst] * (0xff - alpha[0])) >> 8) + ((alpha[0] * m_wm_data_ptr[2][x_src]) >> 8);
            alpha[1] = m_wm_data_ptr[1][cur_pos + 7];
            if (alpha[1])
                image_buf[x_dst + 1] = ((image_buf[x_dst+1]*(0xff-alpha[1]))>>8)+((alpha[1]*m_wm_data_ptr[2][x_src+1])>> 8);
            alpha[2] = m_wm_data_ptr[1][cur_pos + line + 3];
            if (alpha[2])
                image_buf[x_dst + y_pitch] = ((image_buf[x_dst + y_pitch] * (0xff - alpha[2])) >> 8) +
                ((alpha[2] * m_wm_data_ptr[2][x_src + m_wm_rect[1].width]) >> 8);
            alpha[3] = m_wm_data_ptr[1][cur_pos + line + 7];
            if (alpha[3])
                image_buf[x_dst + y_pitch + 1] = ((image_buf[x_dst + y_pitch + 1] * (0xff - alpha[3])) >> 8) +
                ((alpha[3] * m_wm_data_ptr[2][x_src + m_wm_rect[1].width + 1]) >> 8);

            if (alpha[0] + alpha[1] + alpha[2] + alpha[3])
            {// 只要uv对应的4个y中有一个不是0，即着色
                if (va_image_ptr->format.fourcc == VA_FOURCC_NV12)
                {
                    x_dst = uv_start + i*u_pitch + 2 * j;
                    image_buf[x_dst] = m_wm_data_ptr[2][size * 4 + i*w + j];
                    image_buf[x_dst + 1] = m_wm_data_ptr[2][size * 5 + i*w + j];
                }
                else if (va_image_ptr->format.fourcc == VA_FOURCC_YV12)
                {
                    x_dst = u_offset + sub_x + (sub_y + i)*u_pitch + j;
                    image_buf[x_dst] = m_wm_data_ptr[2][size * 5 + i*w + j];
                    x_dst = v_offset + sub_x + (sub_y + i)*v_pitch + j;
                    image_buf[x_dst] = m_wm_data_ptr[2][size * 4 + i*w + j];
                }
            }
        }
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: RenderTitle
* 描  述: 计算字幕混合
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月28日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAMixerSource::RenderTitle(VAImage* va_image_ptr, uint8_t* image_buf)
{
    int x = m_wm_rect[0].iX, sub_x = x >> 1;
    int y = m_wm_rect[0].iY, sub_y = y >> 1;
    int w = m_wm_rect[0].width >> 1;
    int h = m_wm_rect[0].height >> 1;
    int size = w*h;

    int y_pitch = va_image_ptr->pitches[0];
    int u_pitch = va_image_ptr->pitches[1];
    int v_pitch = va_image_ptr->pitches[2];
    int u_offset = va_image_ptr->offsets[1];
    int v_offset = va_image_ptr->offsets[2];

    int y_start = x + y*y_pitch;
    int uv_start = x + sub_y*u_pitch + u_offset;
    int white[4] = { 0 }; // 4y = 1u = 1v
    for (int i = 0; i < h; i++)
    {
        for (int j = 0; j < w; j++)
        {// 进行alpha混合
            int x_src = i*w * 4 + 2 * j;
            int x_dst = y_start + i * 2 * y_pitch + 2 * j;

            white[0] = m_wm_data_ptr[0][x_src];
            if (white[0] != 0xff)
                image_buf[x_dst] = (image_buf[x_dst] * white[0]) >> 8;
            white[1] = m_wm_data_ptr[0][x_src + 1];
            if (white[1] != 0xff)
                image_buf[x_dst + 1] = (image_buf[x_dst + 1] * white[1]) >> 8;
            white[2] = m_wm_data_ptr[0][x_src + w * 2];
            if (white[2] != 0xff)
                image_buf[x_dst + y_pitch] = (image_buf[x_dst + y_pitch] * white[2]) >> 8;
            white[3] = m_wm_data_ptr[0][x_src + w * 2 + 1];
            if (white[3] != 0xff)
                image_buf[x_dst + y_pitch + 1] = (image_buf[x_dst + y_pitch + 1] * white[3]) >> 8;

            if (white[0] != 0xff || white[1] != 0xff || white[2] != 0xff || white[3] != 0xff)
            {// 只要uv对应的4个y中有一个不是白色，即着色
                if (va_image_ptr->format.fourcc == VA_FOURCC_NV12)
                {
                    x_dst = uv_start + i*u_pitch + 2 * j;
                    image_buf[x_dst] = m_wm_data_ptr[0][size * 4 + i*w + j];
                    image_buf[x_dst + 1] = m_wm_data_ptr[0][size * 5 + i*w + j];
                }
                else if (va_image_ptr->format.fourcc == VA_FOURCC_YV12)
                {
                    x_dst = u_offset + sub_x + (sub_y + i)*u_pitch + j;
                    image_buf[x_dst] = m_wm_data_ptr[0][size * 5 + i*w + j];
                    x_dst = v_offset + sub_x + (sub_y + i)*v_pitch + j;
                    image_buf[x_dst] = m_wm_data_ptr[0][size * 4 + i*w + j];
                }
            }
        }
    }
    return EM_RC_OK;
}
