{
    int size = (int)(sizeof(g_fourcc_map) / sizeof(TFourccMap));

/*******************************************************************************
* 函  数: VACreateDisplayDrm
* 描  述: 通过drm来启动va
* 时  间: 2016年06月15日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VACreateDisplayDrm()
{
    const char* ac_drm_device_paths[] =
    {
        "/dev/dri/card0",
        "/dev/dri/renderD128",
        nullptr,
    };

    for (int i = 0; ac_drm_device_paths[i]; i++)
    {
        int fd_drm = open(ac_drm_device_paths[i], O_RDWR);
        if (fd_drm < 0)
        {
            LOG_ERROR("Open %s Fail!", ac_drm_device_paths[i]);
            continue;
        }

        g_va_display = vaGetDisplayDRM(fd_drm);
        if (g_va_display != nullptr)
        {
            close(fd_drm);
            LOG_INFO("Init VADisplay With[%s] Success!", ac_drm_device_paths[i]);
            return EM_RC_OK;
        }
        close(fd_drm);
    }

    return EM_RC_INITED_ERR;
}

/*******************************************************************************
* 函  数: VACreateDisplayX11
* 描  述: 通过x11启动
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月02日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VACreateDisplayX11()
{
    CAutoLock lock(g_display_mtx);

    if (g_display_ptr == nullptr)
    {
        XInitThreads();
        g_display_ptr = XOpenDisplay(nullptr);
        if (g_display_ptr == nullptr)
        {
            LOG_ERROR("Can't Connect XServer, Error[%d], Info: %s", errno, strerror(errno));
            return EM_RC_INIT_ERR;
        }
    }

    if (g_va_display == nullptr)
    {
        g_va_display = vaGetDisplay(g_display_ptr);
        if (g_va_display == nullptr)
        {
            LOG_ERROR("Create VADisplay Failed!");
            return EM_RC_INIT_ERR;
        }

        int iMajorVer, iMinorVer;
        VAStatus vaStatus = vaInitialize(g_va_display, &iMajorVer, &iMinorVer);
        if (vaStatus != VA_STATUS_SUCCESS)
        {
            LOG_ERROR("Init Vaapi Fail[%d]", vaStatus);
            return EM_RC_INIT_ERR;
        }
    }

    g_ref++;
    LOG_INFO("Init VADisplay Success, Ref[%d]", g_ref);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: VADestroyDisplay
* 描  述: 销毁va句柄
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年03月04日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VADestroyDisplay()
{
    CAutoLock lock(g_display_mtx);

    g_ref--;
    if (g_ref == 0)
    {
        if (g_display_ptr != nullptr)
        {
            XCloseDisplay(g_display_ptr);
            g_display_ptr = nullptr;
        }
        if (g_va_display != nullptr)
        {
            vaTerminate(g_va_display);
            g_va_display = nullptr;
        }
    }

    LOG_INFO("UnInit VADisplay, Ref[%d]", g_ref);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DownDataFromSurface
* 描  述: down解码数据
* 参  数: i_vaDisplay
* 参  数: i_vaSurfID
* 参  数: i_sSize
* 参  数: o_pDst
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月09日
* 作  者: 刘峰(Arthur)
********************************************************************************/
int DownYUVDataFromImage(VAImage* i_va_image_ptr, uint8_t* o_data_ptr)
{
    uint8_t *surf_ptr = nullptr;
    uint8_t *y_start_ptr = nullptr, *u_start_ptr = nullptr;
    int y_pitch = 0, u_pitch = 0;
    int row = 0;
    int width = i_va_image_ptr->width;
    int height = i_va_image_ptr->height;

    VAStatus va_status = vaMapBuffer(g_va_display, i_va_image_ptr->buf, (void **)&surf_ptr);
    CHECK_VASTATUS(va_status, "vaMapBuffer(download_surface");

    y_start_ptr = surf_ptr;
    y_pitch = i_va_image_ptr->pitches[0];
    u_start_ptr = surf_ptr + i_va_image_ptr->offsets[1];
    u_pitch = i_va_image_ptr->pitches[1];

    for (row = 0; row < height; row++)
    {
        uint8_t *pYRow = y_start_ptr + row * y_pitch;
        memcpy(o_data_ptr + row*width, pYRow, width);
    }

    int uv_height = height >> 1;
    uint8_t *u_row_ptr = NULL;
    uint8_t *dst_u_ptr = o_data_ptr + width*height;

    for (row = 0; row < uv_height; row++)
    {
        u_row_ptr = u_start_ptr + row * u_pitch;
        memcpy(dst_u_ptr + row * width, u_row_ptr, width);
    }

    vaUnmapBuffer(g_va_display, i_va_image_ptr->buf);
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：上传数据到VAImage
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode UploadDataToImage(VAImage* i_va_image_ptr, uint8_t* i_surf_ptr, uint8_t* i_src_data_ptr, 
    EYUVFormat i_format, int i_src_width, int i_src_height)
{
    uint8_t *y_start_ptr = NULL, *u_start_ptr = NULL;
    int y_pitch = 0, u_pitch = 0, ii = 0, j = 0;
    int src_width = i_src_width;
    int src_height = i_src_height;

    y_start_ptr = i_surf_ptr;
    u_start_ptr = (uint8_t *)i_surf_ptr + i_va_image_ptr->offsets[1];
    y_pitch = i_va_image_ptr->pitches[0];
    u_pitch = i_va_image_ptr->pitches[1];

    if (i_format == EM_YUV_FORMAT_YUYV)
    {//422  
        int src_align_width = src_width << 1;
        int pixel_width = src_width >> 1;
        int pixel_height = src_height >> 1;
        uint8_t* pYUYVBuffer = i_src_data_ptr;

        for (ii = 0; ii < pixel_height; ++ii)
        {
            uint8_t* dest_y = y_start_ptr + (ii << 1)*y_pitch;
            uint8_t* dest_uv = u_start_ptr + ii*u_pitch;
            for (j = 0; j < pixel_width; ++j)
            {
                int u;
                int v;
                int sub_j = j << 2;
                int z_j = j << 1;

                //422的U数据，取1、2行同列的U平均值作为420的U数据。
                u = pYUYVBuffer[sub_j + 1];
                u += pYUYVBuffer[sub_j + 1 + src_align_width];
                dest_uv[z_j] = u >> 1;

                //422的V数据，取1、2行同列的V平均值作为420的V数据。
                v = pYUYVBuffer[sub_j + 3];
                v += pYUYVBuffer[sub_j + 3 + src_align_width];
                dest_uv[z_j + 1] = v >> 1;

                //COPY两行Y数据
                dest_y[z_j] = pYUYVBuffer[sub_j];
                dest_y[z_j + 1] = pYUYVBuffer[sub_j + 2];
                dest_y[y_pitch + z_j] = pYUYVBuffer[sub_j + src_align_width];
                dest_y[y_pitch + z_j + 1] = pYUYVBuffer[sub_j + src_align_width + 2];
            }
            // 一次循环从422中读取4*W的数据，计算得到W*2的Y数据、W/2的U数据、W/2的V数据，
            // 共得到2行420数据:W*3.
            // 源数据指针移动W*4,继续计算，共H/2次。
            pYUYVBuffer += (src_width << 2);
        }
    }
    else
    {//420
        int y_size = src_width*src_height;
        int uv_size = y_size >> 2;
        for (ii = 0; ii < src_height; ii++)
        {//按pitch，每次copy一行数据到NV12数组中
            memcpy(y_start_ptr + ii*y_pitch, i_src_data_ptr + ii*src_width, src_width);
        }
        if (i_format == EM_YUV_FORMAT_NV12)
        {
            for (ii = 0; ii < src_height >> 1; ii++)
            {//按pitch，每次copy一行数据到NV12数组中
                memcpy(u_start_ptr + ii*u_pitch, i_src_data_ptr + y_size + ii*src_width, src_width);
            }
        }
        else if (i_format == EM_YUV_FORMAT_I420 || i_format == EM_YUV_FORMAT_YV12)
        {//IYUV和YV12的格式一样，指示U/V数据的位置互换。
            uint8_t *pUData = NULL, *pVData = NULL;
            if (i_format == EM_YUV_FORMAT_I420)
            {
                pUData = i_src_data_ptr + y_size;
                pVData = pUData + uv_size;
            }
            else
            {
                pVData = i_src_data_ptr + y_size;
                pUData = pVData + uv_size;
            }
            for (ii = 0; ii < src_height >> 1; ii++)
            {//因为NV12是UV数据交替存放在大小为W*H的数组中，所以需要循环赋值。
                uint8_t* pURow = u_start_ptr + ii * u_pitch;
                uint8_t* pUPtr = pUData + ii * (src_width >> 1);
                uint8_t* pVPtr = pVData + ii * (src_width >> 1);
                for (j = 0; j < src_width >> 1; j++)
                {
                    pURow[j << 1] = pUPtr[j];
                    pURow[(j << 1) + 1] = pVPtr[j];
                }
            }
        }
        else
        {
            LOG_ERROR("Not Support Format: %s\n", GetStringFromFormat(i_format));
            return EM_RC_YUV_FORMAT_ERR;
        }
    }
    return EM_RC_OK;
}// NOLINT(readability/fn_size)

/*------------------------------------------------------------------------------
* 描  述：上传IYUV数据到VAImage
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
int UploadIYUVToImage(VAImage* i_va_image_ptr, uint8_t* i_surf_ptr, uint8_t* i_src_data_ptr, int i_src_width, int i_src_height)
{
    uint8_t* y_start_ptr = i_surf_ptr;
    uint8_t* u_start_ptr = (uint8_t *)i_surf_ptr + i_va_image_ptr->offsets[1];
    uint8_t* v_start_ptr = (uint8_t *)i_surf_ptr + i_va_image_ptr->offsets[2];

    int y_size = i_src_width*i_src_height;
    int u_size = y_size >> 2;
    int uv_width = i_src_width >> 1;
    int uv_height = i_src_height >> 1;

    uint8_t* u_src_data_ptr = i_src_data_ptr + y_size;
    uint8_t* v_src_data_ptr = u_src_data_ptr + u_size;

    int i = 0;
    for (i = 0; i < i_src_height; ++i) //y
        memcpy(y_start_ptr + i_va_image_ptr->pitches[0] * i, i_src_data_ptr + i_src_width*i, i_src_width);
    for (i = 0; i < uv_height; ++i)//u
        memcpy(u_start_ptr + i_va_image_ptr->pitches[1] * i, u_src_data_ptr + uv_width*i, uv_width);
    for (i = 0; i < uv_height; ++i)//v
        memcpy(v_start_ptr + i_va_image_ptr->pitches[2] * i, v_src_data_ptr + uv_width*i, uv_width);

    return 0;
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：
-----------------------------------------------------------------------------*/
uint32_t VASWAP32(uint32_t i_val)
{
    uint8_t *val_ptr = (uint8_t *)&i_val;

    return ((val_ptr[0] << 24) | (val_ptr[1] << 16) |
        (val_ptr[2] << 8) | (val_ptr[3] << 0));
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamStart(BitStream *i_bs_ptr)
{
    i_bs_ptr->max_size_in_dword = BITSTREAM_ALLOCATE_STEPPING;
    i_bs_ptr->buffer = (uint32_t*)calloc(i_bs_ptr->max_size_in_dword * sizeof(i_bs_ptr->bit_offset), 1);
    i_bs_ptr->bit_offset = 0;
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamEnd(BitStream *i_bs_ptr)
{
    int pos = (i_bs_ptr->bit_offset >> 5);
    int bit_offset = (i_bs_ptr->bit_offset & 0x1f);
    int bit_left = 32 - bit_offset;

    if (bit_offset) 
        i_bs_ptr->buffer[pos] = VASWAP32((i_bs_ptr->buffer[pos] << bit_left));
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamPutUI(BitStream *i_bs_ptr, uint32_t i_val, int i_size_in_bits)
{
    int pos = (i_bs_ptr->bit_offset >> 5);
    int bit_offset = (i_bs_ptr->bit_offset & 0x1f);
    int bit_left = 32 - bit_offset;

    if (!i_size_in_bits)
        return;

    i_bs_ptr->bit_offset += i_size_in_bits;

    if (bit_left > i_size_in_bits) {
        i_bs_ptr->buffer[pos] = (i_bs_ptr->buffer[pos] << i_size_in_bits | i_val);
    }
    else {
        i_size_in_bits -= bit_left;
        i_bs_ptr->buffer[pos] = (i_bs_ptr->buffer[pos] << bit_left) | (i_val >> i_size_in_bits);
        i_bs_ptr->buffer[pos] = VASWAP32(i_bs_ptr->buffer[pos]);

        if (pos + 1 == i_bs_ptr->max_size_in_dword) {
            i_bs_ptr->max_size_in_dword += BITSTREAM_ALLOCATE_STEPPING;
            i_bs_ptr->buffer = (uint32_t*)realloc(i_bs_ptr->buffer, i_bs_ptr->max_size_in_dword * sizeof(uint32_t));
        }

        i_bs_ptr->buffer[pos + 1] = i_val;
    }
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamPutUE(BitStream *i_bs_ptr, uint32_t i_val)
{
    int size_in_bits = 0;
    int tmp_val = ++i_val;

    while (tmp_val)
    {
        tmp_val >>= 1;
        size_in_bits++;
    }

    BitStreamPutUI(i_bs_ptr, 0, size_in_bits - 1); // leading zero
    BitStreamPutUI(i_bs_ptr, i_val, size_in_bits);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamPutSE(BitStream *i_bs_ptr, int i_val)
{
    uint32_t new_val;

    if (i_val <= 0)
        new_val = -2 * i_val;
    else
        new_val = 2 * i_val - 1;

    BitStreamPutUE(i_bs_ptr, new_val);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void BitStreamByteAligning(BitStream *i_bs_ptr, int i_bit)
{
    int bit_offset = (i_bs_ptr->bit_offset & 0x7);
    int bit_left = 8 - bit_offset;
    int new_val;

    if (!bit_offset)
        return;

    assert(i_bit == 0 || i_bit == 1);

    if (i_bit)
        new_val = (1 << bit_left) - 1;
    else
        new_val = 0;

    BitStreamPutUI(i_bs_ptr, new_val, bit_left);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void  RBSPTrailingBits(BitStream *i_bs_ptr)
{
    BitStreamPutUI(i_bs_ptr, 1, 1);
    BitStreamByteAligning(i_bs_ptr, 0);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void  NalStartCodePrefix(BitStream *i_bs_ptr)
{
    BitStreamPutUI(i_bs_ptr, 0x00000001, 32);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void  NalHeader(BitStream *i_bs_ptr, int i_nal_ref_idc, int i_nal_unit_type)
{
    BitStreamPutUI(i_bs_ptr, 0, 1);                /* forbidden_zero_bit: 0 */
    BitStreamPutUI(i_bs_ptr, i_nal_ref_idc, 2);
    BitStreamPutUI(i_bs_ptr, i_nal_unit_type, 5);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void SortOne(VAPictureH264 ref[], int i_left, int i_right,
    int i_ascending, int frame_idx)
{
    int i = i_left, j = i_right;
    uint32_t key;
    VAPictureH264 pic_tmp;

    if (frame_idx)
    {
        key = ref[(i_left + i_right) / 2].frame_idx;
        PARTITION(ref, frame_idx, key, i_ascending);
    }
    else
    {
        key = ref[(i_left + i_right) / 2].TopFieldOrderCnt;
        PARTITION(ref, TopFieldOrderCnt, (signed int)key, i_ascending);
    }

    /* recursion */
    if (i_left < j)
        SortOne(ref, i_left, j, i_ascending, frame_idx);

    if (i < i_right)
        SortOne(ref, i, i_right, i_ascending, frame_idx);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：无
-----------------------------------------------------------------------------*/
void SortTwo(VAPictureH264 ref[], int i_left, int i_right, uint32_t i_key, uint32_t frame_idx,
    int i_partition_ascending, int i_list0_ascending, int i_list1_ascending)
{
    int i = i_left, j = i_right;
    VAPictureH264 tmp;

	if (frame_idx)
	{
		PARTITION(ref, frame_idx, i_key, i_partition_ascending);
	}
	else
	{
		PARTITION(ref, TopFieldOrderCnt, (signed int)i_key, i_partition_ascending);
	}

    SortOne(ref, i_left, i - 1, i_list0_ascending, frame_idx);
    SortOne(ref, j + 1, i_right, i_list1_ascending, frame_idx);
}
