/*******************************************************************************
* 描  述: 创建编码器
* 参  数: i_enc_props_ptr: 视频编码的参数
* 返回值: 成功：enumRC_OK  失败：XX
* 时  间: 2015年10月31日
* 作  者: Arthur
********************************************************************************/
// 创建、并启动
ERetCode EncodeVideo::CreateEncoder(TEncInfo* i_enc_props_ptr)
{
    CHECK_POINTER(i_enc_props_ptr, EM_RC_PARAM_ERR);

    if (m_is_inited)
    {
        LOG_ERROR("Have inited h264 encode!");
        return EM_RC_ENC_INITED_ERR;
    }

    m_codec_ptr = avcodec_find_encoder(AV_CODEC_ID_H264);
    if (m_codec_ptr == NULL)
    {
        LOG_ERROR("Not find h264 encode!");
        return EM_RC_ENC_NOTFIND_ERR;
    }

    m_codec_ctx_ptr = avcodec_alloc_context3(m_codec_ptr);

    //输入的尺寸必须是32的整数倍，否则会打开编码器失败
    m_codec_ctx_ptr->width = i_enc_props_ptr->encode_size.width;
    m_codec_ctx_ptr->height = i_enc_props_ptr->encode_size.height;
    m_codec_ctx_ptr->time_base.num = 1;
    m_codec_ctx_ptr->time_base.den = i_enc_props_ptr->fps;
    m_codec_ctx_ptr->gop_size = i_enc_props_ptr->fps * 2;
    m_codec_ctx_ptr->bit_rate = i_enc_props_ptr->bitrate;
    m_codec_ctx_ptr->max_b_frames = 0;
    m_codec_ctx_ptr->pix_fmt = AV_PIX_FMT_YUV420P;

    av_opt_set(m_codec_ctx_ptr->priv_data, "preset", "veryfast", 0);	//编码速度较快且能支持固定码率、帧率
    //av_opt_set(m_pCodecCtx->priv_data, "preset",  "ultrafast",    0);
    av_opt_set(m_codec_ctx_ptr->priv_data, "tune", "zerolatency", 0);	//编码零延迟
    av_opt_set(m_codec_ctx_ptr->priv_data, "profile", "main", 0);	//main用于数字视频存储

    m_codec_ctx_ptr->thread_count = 4;
    if (avcodec_open2(m_codec_ctx_ptr, m_codec_ptr, NULL) < 0)
    {
        LOG_ERROR("Open H264Encode[%dx%d] Fail!", i_enc_props_ptr->encode_size.width, i_enc_props_ptr->encode_size.height);
        avcodec_free_context(&m_codec_ctx_ptr);
        return EM_RC_PARAM_ERR;
    }

    m_frame_ptr = av_frame_alloc();
    m_frame_buffer_ptr = (uint8_t*)av_malloc(
        av_image_get_buffer_size(AV_PIX_FMT_YUV420P, m_codec_ctx_ptr->width, m_codec_ctx_ptr->height, 1));
    av_image_fill_arrays(m_frame_ptr->data, m_frame_ptr->linesize, m_frame_buffer_ptr,
        AV_PIX_FMT_YUV420P, m_codec_ctx_ptr->width, m_codec_ctx_ptr->height, 1);

    m_frame_ptr->format = AV_PIX_FMT_YUV420P;
    m_frame_ptr->width = m_codec_ctx_ptr->width;
    m_frame_ptr->height = m_codec_ctx_ptr->height;

    SAFE_DELETE(m_enc_frame.frame_data_ptr);
    m_enc_frame.frame_data_ptr = new uint8_t[COMMON_SIZE_4M];

    SAFE_DELETE(m_yuv_frame.frame_data_ptr);
    m_yuv_frame.frame_data_ptr = new uint8_t[COMMON_SIZE_4M];

    m_buffer_ptr = new uint8_t[COMMON_SIZE_4M];

    m_enc_props = *i_enc_props_ptr;

    m_media_tool_ptr = new MediaTool(g_rbs_Log_ptr);

    m_enc_frame.width = m_enc_props.encode_size.width;
    m_enc_frame.height = m_enc_props.encode_size.height;

    LOG_INFO("Create The Width:Height [%d:%d] Encoder [%d] Success",
        m_enc_props.encode_size.width, m_enc_props.encode_size.height, m_enc_props.userdata);

    m_is_inited = 1;
    return StartThread();
}

/*******************************************************************************
* 描  述: 视频编码
* 参  数: i_pData: 原始数据指针
i_iLen: 数据长度
o_pMediaPkt: 输出的编码数据缓冲区
* 返回值: 成功：enumRC_OK  失败：XX
* 时  间: 2015年10月31日
* 作  者: Arthur
********************************************************************************/
ERetCode EncodeVideo::EncodeData(uint8_t* i_data_ptr, uint32_t i_len, TOutFrame* o_media_pkt_ptr)
{
    CHECK_POINTER(i_data_ptr, EM_RC_PARAM_ERR);

    if (!m_is_inited || m_codec_ctx_ptr == nullptr || m_frame_ptr == nullptr)
        return EM_RC_ENC_UNINIT_ERR;

    AVPacket enc_pkt;
    av_init_packet(&enc_pkt);
    enc_pkt.data = m_enc_frame.frame_data_ptr;
    enc_pkt.size = COMMON_SIZE_4M;
   
    m_frame_ptr->data[0] = i_data_ptr;
    m_frame_ptr->data[2] = i_data_ptr + m_enc_props.encode_size.width*m_enc_props.encode_size.height;
    m_frame_ptr->data[1] = i_data_ptr + m_enc_props.encode_size.width*m_enc_props.encode_size.height * 5 / 4;
    m_frame_ptr->pts = m_enc_count++;

    int got_Picture = 0;
    int iEncRet = avcodec_encode_video2(m_codec_ctx_ptr, &enc_pkt, m_frame_ptr, &got_Picture);
    if (iEncRet < 0)
        return EM_RC_ENC_ERR;
        
    if (o_media_pkt_ptr == nullptr)
    {
        if (got_Picture)
        {
            m_enc_frame.data_len = enc_pkt.size;
            m_enc_frame.key_frame_flag = enc_pkt.flags ? 1 : 0;
            if (m_callBack_ptr)
                m_callBack_ptr->OnEncodeCallBack(&m_enc_frame, m_enc_props.userdata);
        }
        else
            m_enc_frame.data_len = 0;
    }
    else
    {
        if (got_Picture)
        {
            o_media_pkt_ptr->key_frame_flag = enc_pkt.flags ? 1 : 0;
            o_media_pkt_ptr->data_len = enc_pkt.size;
            memcpy(o_media_pkt_ptr->frame_data_ptr, enc_pkt.data, enc_pkt.size);
        }
        else
        {
            o_media_pkt_ptr->data_len = 0;
            av_packet_unref(&enc_pkt);
            return EM_RC_ENC_ERR;
        }
    }

    av_packet_unref(&enc_pkt);
    return EM_RC_OK;
}

/*******************************************************************************
* 描  述: 反初始化编码器
* 参  数: 无
* 返回值: 成功：enumRC_OK  失败：XX
* 时  间: 2015年10月31日
* 作  者: Arthur
********************************************************************************/
ERetCode EncodeVideo::DestroyEncoder()
{
    if (!m_is_inited)
        return EM_RC_ENC_UNINIT_ERR;

    StopThread();

    if (m_codec_ctx_ptr)
    {
        avcodec_close(m_codec_ctx_ptr);
        avcodec_free_context(&m_codec_ctx_ptr);
        m_codec_ctx_ptr = nullptr;
    }

    if (m_frame_buffer_ptr)
    {
        av_free(m_frame_buffer_ptr);
        m_frame_buffer_ptr = nullptr;
    }
    if (m_frame_ptr)
        av_frame_free(&m_frame_ptr);

    if (m_enc_frame.frame_data_ptr)
        SAFE_DELETE(m_enc_frame.frame_data_ptr);

    if (m_yuv_frame.frame_data_ptr)
        SAFE_DELETE(m_yuv_frame.frame_data_ptr);

    if (m_buffer_ptr)
        SAFE_DELETE(m_buffer_ptr);

    if (m_media_tool_ptr)
        SAFE_DELETE(m_media_tool_ptr);

    LOG_INFO("Destroy Encoder [%d]", m_enc_props.userdata);

    m_is_inited = 0;
    m_enc_count = 0;
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：编码器线程
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode EncodeVideo::ThreadFuncProc()
{
    int loop_num = 0;
    uint8_t* buffer_ptr = nullptr;
    uint64_t tick = stGetCurrentMicroSecond();
    uint64_t frame_timestamp = COMMON_TICK_BASE / COMMON_DEFAULT_FPS;

    if (m_enc_props.fps)
        frame_timestamp = COMMON_TICK_BASE / m_enc_props.fps;

    LOG_INFO("Start EncodeThread[%d]", m_enc_props.userdata);
    while (m_exit_code == EM_TS_RUN)
    {
        if (m_source_ptr == nullptr)
        {
            usleep(COMMON_TICK_SECOND);
            continue;
        }

        VideoSrcImage image_ptr = m_source_ptr->GetReadyImage(&m_yuv_frame);
        if (image_ptr == nullptr)
        {
            usleep(10 * 1000);
            continue;
        }

        if (m_yuv_frame.width != m_enc_props.encode_size.width || m_yuv_frame.height != m_enc_props.encode_size.height)
        {
            m_media_tool_ptr->DirectVideoScale(m_yuv_frame.width, m_yuv_frame.height, m_yuv_frame.frame_data_ptr,
                m_enc_props.encode_size.width, m_enc_props.encode_size.height, m_buffer_ptr);
            buffer_ptr = m_buffer_ptr;
        }
        else
            buffer_ptr = m_yuv_frame.frame_data_ptr;

        EncodeData(buffer_ptr, m_enc_props.encode_size.width * m_enc_props.encode_size.height * 3 / 2);

        loop_num++;
        while (stGetCurrentMicroSecond() - tick < loop_num * frame_timestamp)
            usleep(COMMON_TICK_SECOND);
    }

    m_exit_code = EM_TS_STOP;
    LOG_INFO("Stop EncodeThread[%ld]", m_enc_props.userdata);
    return EM_RC_OK;
}
