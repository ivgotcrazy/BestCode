//挂名为软渲染，内部渲染其实还是VA渲染，只是为了保持软硬处理的差异化而已
class FFMixerRender : public BaseThread, public IVRender
{
public:
	FFMixerRender(){};
	~FFMixerRender(){};

public: // VASource抽象接口

	ERetCode CreateRender(TMediaRenderProps* i_render_props_ptr);

	ERetCode DestroyRender();

	ERetCode AddSource(int i_src_id, TProcItem* i_item_ptr);
	ERetCode DelSource(int i_src_id);

	ERetCode RenderData(uint8_t* i_data_ptr);

	ERetCode ModifyRenderSize(TSize* i_size_ptr);
	ERetCode ModifySrcOutRect(int i_src_id, TRect* i_out_rect_ptr);
	ERetCode ModifySrcLevel(int i_src_id, EVMixLevel i_level);
	ERetCode ModifySrcSource(int i_src_id, ISource* i_source_ptr);

	ERetCode SetCurResolution(char *i_output_name_ptr, int i_wight, int i_height);

	//获取显示器支持的分辨率列表
	ERetCode GetSurportResolutionList(TOutputResolutions **o_resolution_ptr);

	//释放分辨率列表
	void FreeSurportResolutionList(TOutputResolutions *i_resolution_ptr);


private:
	//线程函数
	ERetCode ThreadFuncProc();
	//合成任务
	ERetCode RunVideoRenderTask();
	//画面合成
	ERetCode MixVideoTask(TSize i_main_size, uint8_t* io_main_image_ptr, TRect i_sub_size, uint8_t* i_sub_image_ptr);
	//检查源状态
	ERetCode CheckSourceState();
	//分配数据缓冲
	ERetCode AllocDataBuffer();
	//释放数据缓冲区
	ERetCode ReleaseDataBuffer();

private:
	int m_is_inited = 0;                                  	//初始化

	MediaTool* m_media_tool_ptr = nullptr;                  //媒体工具(叠加字幕，特效，台标，编码) 
	int	m_max_src_count = VA_MAX_SOURCE_COUNT;	            // 最大合成输入源个数
	int  m_cur_src_count = 0;                         	    // 合成的输入流数
	std::mutex m_mixer_mtx;                                 // 合成锁
	
	TSourceProps m_src_props;                               //合成源属性
	
	EEffectType  m_effect_type = EM_EFFECT_NONE;            //特效类型
	MapMixerMember  m_mixer_map;                            //合成源数组
	uint8_t * m_mixer_buffer_ptr = nullptr;                 //合成缓冲
	uint8_t* m_bg_buffer_ptr = nullptr;                     //背景缓冲图
	TOutFrame m_read_buffer;                                //读取缓冲

	int m_buffer_size = 0;                                  //缓冲区大小
	int m_src_image_size = 0;                               //源图片大小
	
	TSize m_render_size;                                    //渲染大小
	TSize m_update_size;                                    //更新大小
	TRect m_render_rect;                                    //渲染位置
	
	VARenderX11* m_render_ptr = nullptr;                    //VA硬渲染
};
