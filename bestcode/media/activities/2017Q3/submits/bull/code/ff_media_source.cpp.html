/*------------------------------------------------------------------------------
* 描  述：创建源对象
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::CreateSource(TSourceProps* i_src_props_ptr)
{
	CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

	if (m_is_src_inited)
		return EM_RC_INITED_ERR;

	if (m_src_props.format > EM_YUV_FORMAT_NV12)
	{
		// 422的可以支持，此处未支持
		LOG_ERROR("Has't Support Source Format[%d] !", m_src_props.format);
		return EM_RC_FILE_FORMAT_ERR;
	}

	memcpy(&m_src_props, i_src_props_ptr, sizeof(TSourceProps));

	m_out_frame.width = m_src_props.src_size.width;
	m_out_frame.height = m_src_props.src_size.height;

	if (i_src_props_ptr->v_codec_id >= EM_MT_H264)
	{
		m_decoder_ptr = new MediaDecoder;
		TSourceProps media_props;
		media_props.media_mode = EM_MEDIA_MODE_SOFT;
		memcpy(&media_props, i_src_props_ptr, sizeof(TSourceProps));
		ERetCode emRC = m_decoder_ptr->CreateDecoder(&media_props);
		if ( emRC != EM_RC_OK )
		{
			LOG_ERROR("Media Source [%d] Init Decoder Failed [%d]", i_src_props_ptr->user_data, emRC);
			SAFE_DELETE(m_decoder_ptr);
			return emRC;
		}

		StartThread();
	}

	AllocDataBuffer(m_src_props.v_codec_id);

	m_is_src_inited = 1;

	LOG_INFO("Create FFMediaSource [%ld], Width:Height [%d:%d]", i_src_props_ptr->user_data, 
									i_src_props_ptr->src_size.width, i_src_props_ptr->src_size.height);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：输入数据
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::InputSrcData(uint8_t* i_src_data_ptr, int i_data_len)
{
	CHECK_POINTER(i_src_data_ptr, EM_RC_PARAM_ERR);
	if (!m_is_src_inited)
		return EM_RC_INIT_ERR;
	
	if (m_src_props.v_codec_id == EM_MT_RAW)
		m_media_ring_buffer.PutData(i_src_data_ptr, i_data_len);
	else
	{
		std::unique_lock <std::mutex> lock(m_data_mtx);
		TOutFrame* media_pkt_ptr = DequeueFreeMediaData();
		CHECK_POINTER(media_pkt_ptr, EM_RC_MEMROY_ALLOC_ERR);
		memcpy(media_pkt_ptr->frame_data_ptr, i_src_data_ptr, i_data_len);
		media_pkt_ptr->data_len = i_data_len;
		EnqueueMediaData(media_pkt_ptr);
		m_data_cond.notify_all();
	}

	{
		std::unique_lock<std::mutex> lock(m_media_proc_mtx);
		if ( m_media_proc_ptr )
			m_media_proc_ptr->InputSrcData(i_src_data_ptr, i_data_len);
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：销毁源对象
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::DestroySource()
{
	if (!m_is_src_inited)
		return EM_RC_INIT_ERR;

	if (m_exit_code != EM_TS_STOP)
	{
		m_data_cond.notify_all();
		m_exit_code = EM_TS_STOPING;
		while (m_exit_code != EM_TS_STOP)
			usleep(1000);
	}

	ReleaseDataBuffer();
	if (m_decoder_ptr)
	{
		m_decoder_ptr->DestroyDecoder();
		SAFE_DELETE(m_decoder_ptr);
	}

	{
		std::unique_lock<std::mutex> lock(m_media_proc_mtx);
		if (m_media_proc_ptr)
		{
			m_media_proc_ptr->DestroyMediaProc();
			SAFE_DELETE(m_media_proc_ptr);
		}
	}

	m_is_src_inited = 0;
	m_src_image_size = 0;
	LOG_INFO("Destroy FFMediaSource[%ld]", m_src_props.user_data);
	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：获取媒体数据（o_data_ptr为null,获取的是VAImage，否则是YUV）
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
VideoSrcImage FFMediaSource::GetReadyImage(TOutFrame* o_data_ptr)
{
	if ( !m_is_src_inited )
		return nullptr;

	std::unique_lock<std::mutex> lock(m_modify_mtx);
	if ( o_data_ptr == nullptr )
	{
		//VAImage数据
		if ( m_media_proc_ptr )
			return m_media_proc_ptr->GetReadyImage();
	}
	else
	{
		o_data_ptr->width = m_src_props.src_size.width;
		o_data_ptr->height = m_src_props.src_size.height;
		o_data_ptr->data_len = m_src_image_size;
		if (m_media_ring_buffer.GetData(o_data_ptr->frame_data_ptr, (uint32_t*)&o_data_ptr->data_len) == EM_RC_OK)
			return &m_media_ring_buffer;
	}

	return nullptr;
}

/*------------------------------------------------------------------------------
* 描  述：注册其他类型解码器（目前只有一种，软处理支持硬输出，硬处理支持软输出）
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode  FFMediaSource::RegisterMediaProc(EMediaMode i_media_mode)
{
	if (!m_is_src_inited)
		return EM_RC_INIT_ERR;

	std::unique_lock<std::mutex> lock(m_media_proc_mtx);
	ERetCode er = EM_RC_OK;
	if (i_media_mode != m_src_props.media_mode)
	{
		if ( m_media_proc_ptr != nullptr )
		{
			m_media_proc_ptr->DestroyMediaProc();
			SAFE_DELETE(m_media_proc_ptr);
		}

		m_media_proc_ptr = new MediaProcRoute;
		er = m_media_proc_ptr->CreateMediaProc(i_media_mode, &m_src_props);
		if (er != EM_RC_OK)
			SAFE_DELETE(m_media_proc_ptr);
	}
	else
		return EM_RC_STATE_ERR;

	return er;
}

/*------------------------------------------------------------------------------
* 描  述：设置解码回调和参数（编码数据有用）
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_out_param_ptr)
{
	m_dec_callback_ptr = i_callback_ptr;
	m_dec_out_props_ptr = i_out_param_ptr;

	if (m_decoder_ptr)
		m_decoder_ptr->SetDecDataCallBack(i_callback_ptr, i_out_param_ptr);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：编码数据解码线程
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::ThreadFuncProc()
{
	LOG_INFO("Start Media Source [%d] ProcessThread!", m_src_props.user_data);
	TOutFrame* media_enc_pkt_ptr = nullptr;
	TOutFrame* media_yuv_pkt_ptr = nullptr;
	ERetCode er = EM_RC_OK;
	while (m_exit_code == EM_TS_RUN)
	{
		if ( CheckMediaDataEmpty() == EM_RC_OK )
		{
			std::unique_lock <std::mutex> lock(m_data_mtx);
			m_data_cond.wait_for(lock, std::chrono::milliseconds(20));
			continue;
		}

		if (m_decoder_ptr == nullptr)
		{
			LOG_DEBUG("Media Source [%d] Decoder Not Init", m_src_props.user_data);
			continue;
		}

		media_enc_pkt_ptr = DequeueMediaData();
		er = m_decoder_ptr->DecodeData((uint8_t*)media_enc_pkt_ptr->frame_data_ptr, media_enc_pkt_ptr->data_len,
																		&m_out_frame.frame_data_ptr);
		if (er == EM_RC_OK)
		{
			m_dec_count++;
			if (m_dec_out_props_ptr != nullptr)
			{
				if (m_dec_count % m_dec_out_props_ptr->frame_interval == 0)
					m_dec_callback_ptr->OnDecodeCallBack(&m_out_frame, m_src_props.user_data);
			}

			m_media_ring_buffer.PutData(m_out_frame.frame_data_ptr, m_src_image_size);
		}

		EnqueueFreeMediaData(media_enc_pkt_ptr);
	}

	m_exit_code = EM_TS_STOP;
	LOG_INFO("Stop Media Source [%d] ProcessThread!", m_src_props.user_data);
	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：数据缓存函数
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::AllocDataBuffer(EVideoCodecID i_video_id)
{
	m_buffer_size = COMMON_DEFAULT_WIDTH * COMMON_DEFAULT_HEIGHT * 3 / 2;
	m_src_image_size = m_src_props.src_size.width*m_src_props.src_size.height * 3 / 2;

	SAFE_DELETE(m_out_frame.frame_data_ptr);
	m_out_frame.frame_data_ptr = new uint8_t[m_buffer_size];
	m_out_frame.data_len = m_src_image_size;

	//创建YUV缓冲区
	m_media_ring_buffer.CreateRing(FPS_BUF_COUNT, m_buffer_size);

	if ( i_video_id == EM_MT_H264 )
	{
		//创建编码数据缓冲
		for (int i = 0; i < FF_SOURCE_RING_SIZE; i++)
		{
			TOutFrame *pkt_ptr = new TOutFrame;
			memset(pkt_ptr, 0, sizeof(TOutFrame));
			pkt_ptr->frame_data_ptr = new uint8_t[m_buffer_size];
			m_free_enc_buffer_ptr.push(pkt_ptr);
		}
	}

	LOG_INFO("Source[%ld] Format[ %d ], Size[ %dx%d ] CodecID [%d]",
		m_src_props.user_data, m_src_props.format,
		m_src_props.src_size.width, m_src_props.src_size.height, i_video_id);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：缓存释放函数
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::ReleaseDataBuffer()
{
	ClearMediaData();
	ClearFreeMediaData();

	m_media_ring_buffer.DestroyRing();

	SAFE_DELETE(m_out_frame.frame_data_ptr);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：媒体数据入队
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::EnqueueMediaData(TOutFrame *i_media_pkt_ptr)
{

	std::unique_lock<std::mutex> lock(m_enc_mtx);
	m_enc_buffer_que.push(i_media_pkt_ptr);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：媒体数据出队
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
TOutFrame* FFMediaSource::DequeueMediaData()
{
	TOutFrame* media_pkt_ptr = nullptr;

	std::unique_lock <std::mutex> lock(m_enc_mtx);
	if (m_enc_buffer_que.empty())
		return nullptr;

	media_pkt_ptr = m_enc_buffer_que.front();
	m_enc_buffer_que.pop();

	return media_pkt_ptr;
}

/*------------------------------------------------------------------------------
* 描  述：清理媒体数据
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::ClearMediaData()
{
	std::unique_lock <std::mutex> lock(m_enc_mtx);
	while (!m_enc_buffer_que.empty())
	{
		TOutFrame* media_pkt_ptr = m_enc_buffer_que.front();
		m_enc_buffer_que.pop();
		EnqueueFreeMediaData(media_pkt_ptr);
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：检查媒体数据缓存
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::CheckMediaDataEmpty()
{
	std::unique_lock<std::mutex> lock(m_enc_mtx);
	ERetCode er = m_enc_buffer_que.empty() ? EM_RC_OK : EM_RC_QUEUE_EMPTY_ERR;

	return er;
}

/*------------------------------------------------------------------------------
* 描  述：缓冲入队
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::EnqueueFreeMediaData(TOutFrame *i_media_pkt_ptr)
{
	CHECK_POINTER(i_media_pkt_ptr, EM_RC_PARAM_ERR);

	std::unique_lock<std::mutex> lock(m_free_enc_mtx);
	m_free_enc_buffer_ptr.push(i_media_pkt_ptr);

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：缓冲出队
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
TOutFrame* FFMediaSource::DequeueFreeMediaData()
{
	TOutFrame* media_pkt_ptr = nullptr;
	std::unique_lock <std::mutex> lock(m_free_enc_mtx);
	if (m_free_enc_buffer_ptr.empty())
		return nullptr;

	media_pkt_ptr = m_free_enc_buffer_ptr.front();
	m_free_enc_buffer_ptr.pop();

	return media_pkt_ptr;
}

/*------------------------------------------------------------------------------
* 描  述：释放所有缓冲区
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::ClearFreeMediaData()
{
	std::unique_lock <std::mutex> lock(m_free_enc_mtx);
	while (!m_free_enc_buffer_ptr.empty())
	{
		TOutFrame* media_pkt_ptr = m_free_enc_buffer_ptr.front();
		SAFE_DELETE(media_pkt_ptr->frame_data_ptr);
		SAFE_DELETE(media_pkt_ptr);
		m_free_enc_buffer_ptr.pop();
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：检查缓冲区是否为空
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode FFMediaSource::CheckFreeMediaDataEmpty()
{
	std::unique_lock<std::mutex> lock(m_free_enc_mtx);

	ERetCode er = m_free_enc_buffer_ptr.empty() ? EM_RC_OK : EM_RC_QUEUE_EMPTY_ERR;

	return er;
}
