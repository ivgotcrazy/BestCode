class ISource
{
public:
    
    virtual ~ISource(){};

	/*------------------------------------------------------------------------------
	* 描  述： 创建数据源对象
	* 参  数：[in] i_src_props_ptr 解码器属性
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode CreateSource(TSourceProps* i_src_props_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 输入数据
	* 参  数：[in] i_src_data_ptr 媒体数据
	* 参  数：[in] i_data_len     媒体数据大小
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode InputSrcData(uint8_t* i_src_data_ptr, int i_data_len) = 0;
        
	/*------------------------------------------------------------------------------
	* 描  述： 销毁源对象
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode DestroySource() = 0;
    
	/*------------------------------------------------------------------------------
	* 描  述： 设置字幕（合成源才会使用此接口）
	* 参  数：[in] i_text_info_ptr nullptr表示删除, 反之则添加
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode SetSubTitle(TTextInfo* i_text_info_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 设置台标（合成源才会使用此接口）
	* 参  数：[in] i_logo_info_ptr nullptr表示删除, 反之则添加
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode SetLogo(TLogoInfo* i_logo_info_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 设置特效（合成源才会使用此接口）
	* 参  数：[in] i_effect 特效类型
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode SetEffect(EEffectType i_effect) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 启动特效（合成源才会使用此接口）
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode RunEffect() = 0;

	/*------------------------------------------------------------------------------
	* 描  述： VAImage锁处理，硬处理源使用
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual void LockReadyImage()   = 0;    

	/*------------------------------------------------------------------------------
	* 描  述： 获取图像数据（o_data_ptr为null时，返回值为VAImage，否则o_data_ptr为YUV数据）
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual VideoSrcImage GetReadyImage(TOutFrame* o_data_ptr = nullptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： VAImage解锁处理，硬处理源使用
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual void UnLockReadyImage() = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 注册媒体输出（硬源输出软源数据，软源输出硬源数据）
	* 参  数：[in] i_media_mode 媒体模式（资源媒体模式和导播源媒体模式不同才需要使用此接口）
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode RegisterMediaProc(EMediaMode i_media_mode) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 设置解码回调
	* 参  数：[in] i_callback_ptr  解码回调对象
	* 参  数：[in] i_dec_param_ptr 解码参数
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode SetDecCallBack(IVCallBack* i_callback_ptr, TDecOutProps* i_dec_param_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 获取源属性
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual TSourceProps* GetSrcProps() = 0;
};

struct TProcItem
{
    TRect      rect;                       //输出区域
    ISource* source_ptr = nullptr;         //数据源指针
    EVMixLevel level = EM_VML_NONE;        //层级
    int        tile_flag = 0;              // 是否平铺，1表示是，铺满指定位置，0表示保持比例
};

class ISrcProc
{    
public:
    
    virtual ~ISrcProc(){};

	/*------------------------------------------------------------------------------
	* 描  述：  添加数据源 
	* 参  数：[in] i_src_id   源ID
	* 参  数：[in] i_item_ptr 源的布局信息
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode AddSource(int i_src_id, TProcItem* i_item_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述：删除输入源
	* 参  数：[in] i_src_id   源ID
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode DelSource(int i_src_id) = 0;

	/*------------------------------------------------------------------------------
	* 描  述：  修改成员输出位置
	* 参  数：[in] i_src_id       源ID
	* 参  数：[in] i_out_rect_ptr 源的布局信息
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcOutRect(int i_src_id, TRect* i_out_rect_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 修改成员层级
	* 参  数：[in] i_src_id       源ID
	* 参  数：[in] i_level		  源的层级
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcLevel(int i_src_id, EVMixLevel i_level) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 修改输入源
	* 参  数：[in] i_src_id       源ID
	* 参  数：[in] i_source_ptr	  新的源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcSource(int i_src_id, ISource* i_source_ptr) = 0;
};

// 虚拟合成源对象，由IVASource派生类合成，支持添加、删除、修改Source合成属性
class IMixerSource : public ISource, public ISrcProc
{
public:
	virtual ~IMixerSource(){};

	/*------------------------------------------------------------------------------
	* 描  述： 修改合成的大小
	* 参  数：[in] i_size_ptr       合成源大小
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
    virtual ERetCode ModifyMixerSize(TSize* i_size_ptr) = 0;
};
