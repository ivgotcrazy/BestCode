FFLogo::FFLogo()
    :m_is_init_filter_flag(0),
    m_frame_buffer_ptr(NULL),
    m_frame_ptr(NULL), 
    m_buffer_sink_ctx_ptr(NULL),
    m_buffer_src_ctx_ptr(NULL),
    m_filter_graph_ptr(NULL),
    m_input_ptr(NULL),
    m_output_ptr(NULL)
{
    m_frame_index = 0;
    m_rational.num = 1;
    m_rational.den = 25;
}

FFLogo::~FFLogo()
{
    if(m_is_init_filter_flag)
        UnInit();
}

ERetCode FFLogo::SetVideoSourceSize(int i_width, int i_height)
{
    m_src_width = i_width;
    m_src_height = i_height;
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: Init
* 描  述: 初始化输入输出filter
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月19日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFLogo::Init()
{   
    AVFilter* buffer_src_ptr  = avfilter_get_by_name("buffer");
    AVFilter* buffer_sink_ptr = avfilter_get_by_name("buffersink");
    enum AVPixelFormat pixel_format[] = { AV_PIX_FMT_YUV420P, AV_PIX_FMT_NONE };
    AVBufferSinkParams* buffer_sink_params = NULL;

    m_input_ptr  = avfilter_inout_alloc();
    m_output_ptr = avfilter_inout_alloc();
    m_filter_graph_ptr = avfilter_graph_alloc();

    /* buffer video source: the decoded frames from the decoder will be inserted here. */
    char filter_param[COMMON_SIZE_1K] = {0};
    sprintf(filter_param,
        "video_size=%dx%d:pix_fmt=%d:time_base=%d/%d:pixel_aspect=%d/%d",
        m_src_width, m_src_height, AV_PIX_FMT_YUV420P,
        1, 25,1,1);
    if (avfilter_graph_create_filter(&m_buffer_src_ctx_ptr, buffer_src_ptr, "in",filter_param, NULL, m_filter_graph_ptr) < 0)
    {
        LOG_ERROR("Cannot create buffer source!");
        UnInit();
        return EM_RC_FILE_CREATE_ERR;
    }

    /* buffer video sink: to terminate the filter chain. */
    buffer_sink_params = av_buffersink_params_alloc();
    buffer_sink_params->pixel_fmts = pixel_format;
    if (avfilter_graph_create_filter(&m_buffer_sink_ctx_ptr, buffer_sink_ptr, "out",NULL, 
												buffer_sink_params, m_filter_graph_ptr) < 0) 
    {
        LOG_ERROR("Cannot create buffer sink!");
        av_free(buffer_sink_params);
        UnInit();
        return EM_RC_FILE_CREATE_ERR;
    }
    av_free(buffer_sink_params);
    /* Endpoints for the filter graph. */
    m_output_ptr->name       = av_strdup("in");
    m_output_ptr->filter_ctx = m_buffer_src_ctx_ptr;
    m_output_ptr->pad_idx    = 0;
    m_output_ptr->next       = NULL;

    m_input_ptr->name       = av_strdup("out");
    m_input_ptr->filter_ctx = m_buffer_sink_ctx_ptr;
    m_input_ptr->pad_idx    = 0;
    m_input_ptr->next       = NULL;

    m_frame_buffer_ptr = (uint8_t*)av_malloc(av_image_get_buffer_size(AV_PIX_FMT_YUV420P, m_src_width, m_src_height, 1));
    m_frame_ptr = av_frame_alloc();
    av_image_fill_arrays(m_frame_ptr->data, m_frame_ptr->linesize, m_frame_buffer_ptr, 
																	AV_PIX_FMT_YUV420P, m_src_width, m_src_height, 1);

    m_frame_ptr->width = m_src_width;
    m_frame_ptr->height = m_src_height;
    m_frame_ptr->format=AV_PIX_FMT_YUV420P;

    m_frame_index = 0;
    LOG_INFO("Init Logo Filter Success!");
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: SetLogo
* 描  述: 
* 参  数: i_pTxtInfo
* 参  数: i_pIcoInfo
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月18日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFLogo::SetLogo(TLogoInfo* i_logo_info_ptr)
{
    if(!m_is_init_filter_flag)
        Init();

    int src_fm = AV_PIX_FMT_BGRA;
    int dst_fm = AV_PIX_FMT_YUVA420P;
    if(i_logo_info_ptr->pos == RBS_WM_POS_LT)//左上
        sprintf(m_filter_decs,"movie=%s,scale=iw*%f:-1:src_format=%d:dst_format=%d[over],[in][over]overlay=4:4[out]",
																			i_logo_info_ptr->logo_path,1.0,src_fm,dst_fm);
    else if(i_logo_info_ptr->pos == RBS_WM_POS_RT)//右上
        sprintf(m_filter_decs,"movie=%s,scale=iw*%f:-1:src_format=%d:dst_format=%d[over],[in][over]overlay= \
											(main_w-overlay_w-4):4[out]",i_logo_info_ptr->logo_path, 1.0,src_fm,dst_fm);
    else if(i_logo_info_ptr->pos == RBS_WM_POS_LB)//左下
        sprintf(m_filter_decs,"movie=%s,scale=iw*%f:-1:src_format=%d:dst_format=%d[over],[in][over]overlay=4:\
												(main_h-overlay_h-4)[out]",i_logo_info_ptr->logo_path, 1.0,src_fm,dst_fm);
    else if(i_logo_info_ptr->pos == RBS_WM_POS_RB)//右下
        sprintf(m_filter_decs,"movie=%s,scale=iw*%f:-1:src_format=%d:dst_format=%d[over],[in][over]overlay=(main_w \
								-overlay_w-4):(main_h-overlay_h-4)[out]",i_logo_info_ptr->logo_path, 1.0,src_fm,dst_fm);
    else
        return EM_RC_UNKNOW;

    if (avfilter_graph_parse_ptr(m_filter_graph_ptr, m_filter_decs,&m_input_ptr, &m_output_ptr, NULL) < 0)
    {
        UnInit();
        LOG_ERROR("Parse[%s] graph failed!", i_logo_info_ptr->logo_path);
        return EM_RC_UNKNOW;
    }
    if (avfilter_graph_config(m_filter_graph_ptr, NULL) < 0)
    {
        UnInit();
        LOG_ERROR("Config graph[%s] failed!", i_logo_info_ptr->logo_path);
        return EM_RC_UNKNOW;
    }

    m_is_init_filter_flag = 1;
    LOG_INFO("Add Logo WaterMark!");

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: LogoFilter
* 描  述: 
* 参  数: io_pData
* 参  数: i_iDataLen
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月18日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFLogo::LogoFilter(uint8_t* io_data_ptr)
{   
    if( !m_is_init_filter_flag )
        return EM_RC_UNINIT_ERR;

    m_frame_ptr->data[0] = io_data_ptr;
    m_frame_ptr->data[1] = io_data_ptr + m_src_height*m_src_width;
    m_frame_ptr->data[2] = m_frame_ptr->data[1] + m_src_height*m_src_width / 4;
    m_frame_ptr->pts = m_frame_index * av_q2d(m_rational);
    m_frame_index++;
    if (av_buffersrc_add_frame(m_buffer_src_ctx_ptr, m_frame_ptr) < 0) 
    {
        LOG_ERROR("Error while feeding the filtergraph!");
        return EM_RC_DATA_ERR;
    }

    /* pull filtered pictures from the filtergraph */
    int iRet = av_buffersink_get_frame(m_buffer_sink_ctx_ptr,m_frame_ptr);
    if (iRet < 0)
        return EM_RC_UNKNOW;

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UnInit
* 描  述: 
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2016年02月19日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode FFLogo::UnInit()
{
    if(m_filter_graph_ptr)
    {
        avfilter_graph_free(&m_filter_graph_ptr);
        m_filter_graph_ptr = NULL;
    }
    if(m_output_ptr)
    {
        avfilter_inout_free(&m_output_ptr);
        m_output_ptr = NULL;
    }
    if(m_input_ptr)
    {
        avfilter_inout_free(&m_input_ptr);
        m_input_ptr = NULL;
    }
    if(m_frame_buffer_ptr)
    {
        av_free(m_frame_buffer_ptr);
        m_frame_buffer_ptr = NULL;
    }
    if(m_frame_ptr)
    {
        av_frame_free(&m_frame_ptr);
        m_frame_ptr = NULL;
    }
    m_frame_index = 0;
    m_is_init_filter_flag = 0;
    LOG_INFO("UnInit Logo Filter!");
    return EM_RC_OK;
}
