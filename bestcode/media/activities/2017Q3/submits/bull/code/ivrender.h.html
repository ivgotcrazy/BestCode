/***************************** 条件编译选项和头文件 *****************************/

#include "media_def.h"
#include "isource.h"

/*********************************** 常量和宏 **********************************/

/*********************************** 数据类型 **********************************/

/*********************************** 变量声明 **********************************/

/*********************************** 函数声明 **********************************/

/***********************************  类声明  **********************************/

class IVRender
{
public:

	virtual ~IVRender(){};

	/*------------------------------------------------------------------------------
	* 描  述： 创建渲染器
	* 参  数：[in] i_render_props_ptr 渲染器属性
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode CreateRender(TMediaRenderProps* i_render_props_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 添加数据源
	* 参  数：[in] i_src_id   数据源源ID
	* 参  数：[in] i_item_ptr 数据源布局信息
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode AddSource(int i_src_id, TProcItem* i_item_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 移除输入源
	* 参  数：[in] i_src_id   数据源源ID
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode DelSource(int i_src_id) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 渲染媒体数据（NV12）
	* 参  数：[in] i_data_ptr  媒体数据
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode RenderData(uint8_t* i_data_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 修改源布局
	* 参  数：[in] i_src_id       数据源源ID
	* 参  数：[in] i_out_rect_ptr 数据源位置属性
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcOutRect(int i_src_id, TRect* i_out_rect_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 修改源层级
	* 参  数：[in] i_src_id       数据源源ID
	* 参  数：[in] i_level        数据源层级
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcLevel(int i_src_id, EVMixLevel i_level) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 修改输入源
	* 参  数：[in] i_src_id      数据源源ID
	* 参  数：[in] i_source_ptr  新的数据源
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifySrcSource(int i_src_id, ISource* i_source_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述：  修改渲染器背景尺寸
	* 参  数：[in] i_size_ptr       合成源大小
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode ModifyRenderSize(TSize* i_size_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 设置分辨率
	* 参  数：[in] i_output_name_ptr  名字
	* 参  数：[in] i_width            宽
	* 参  数：[in] i_height           高
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode SetCurResolution(char *i_output_name_ptr, int i_width, int i_height) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 	获取显示器支持的分辨率列表
	* 参  数：[in] o_resolution_ptr  获取的分辨率列表
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode GetSurportResolutionList(TOutputResolutions **o_resolution_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 释放分辨率列表
	* 参  数：[in] i_resolution_ptr  分辨率列表
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual void FreeSurportResolutionList(TOutputResolutions *i_resolution_ptr) = 0;

	/*------------------------------------------------------------------------------
	* 描  述： 销毁渲染器
	* 参  数：[in] 无
	* 返回值：0 成功/非0 失败
	-----------------------------------------------------------------------------*/
	virtual ERetCode DestroyRender() = 0;
};
