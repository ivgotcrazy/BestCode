#include <stdint.h>

#ifndef _H264_SEI_H
#define _H264_SEI_H        1

#include <stdint.h>

#include "vamedia_bitstream.h"

#ifdef __cplusplus
extern "C" {
#endif

struct SEI
{
    int payload_type;    //负载类型
    int payload_size;    //负载大小
    uint8_t* payload_ptr;//负载数据
};

SEI* NewSEI();
void FreeSEI(SEI* s_ptr);

//D.1 SEI payload syntax
#define SEIYPE_BUFFERING_PERIOD                           0
#define SEIYPE_PIC_TIMING                                 1
#define SEIYPE_PAN_SCAN_RECT                              2
#define SEIYPE_FILLER_PAYLOAD                             3
#define SEIYPE_USER_DATA_REGISTERED_ITU_T_T35             4
#define SEIYPE_USER_DATA_UNREGISTERED                     5
#define SEIYPE_RECOVERY_POINT                             6
#define SEIYPE_DEC_REF_PIC_MARKING_REPETITION             7
#define SEIYPE_SPARE_PIC                                  8
#define SEIYPE_SCENE_INFO                                 9
#define SEIYPE_SUB_SEQ_INFO                               10
#define SEIYPE_SUB_SEQ_LAYER_CHARACTERISTICS              11
#define SEIYPE_SUB_SEQ_CHARACTERISTICS                    12
#define SEIYPE_FULL_FRAME_FREEZE                          13
#define SEIYPE_FULL_FRAME_FREEZE_RELEASE                  14
#define SEIYPE_FULL_FRAME_SNAPSHOT                        15
#define SEIYPE_PROGRESSIVE_REFINEMENT_SEGMENT_VAART       16
#define SEIYPE_PROGRESSIVE_REFINEMENT_SEGMENT_END         17
#define SEIYPE_MOTION_CONSTRAINED_SLICE_GROUP_SET         18
#define SEIYPE_FILM_GRAIN_CHARACTERISTICS                 19
#define SEIYPE_DEBLOCKING_FILTER_DISPLAY_PREFERENCE       20
#define SEIYPE_STEREO_VIDEO_INFO                          21

#ifdef __cplusplus
}
#endif

#endif
