/*********************************** 常量和宏 **********************************/

#define    EFFECTEVEN( x )          (((x)&0x01)==1)? (x) + 1 : (x)  

#define    EFFECT_SHUTTER_COUNT     12      // 百叶窗分割数 
#define    EFFECT_RUN_COUNT         24      // 特效处理帧数
#define    EFFECT_ALPHA_COUNT       100

/*********************************** 数据类型 **********************************/

/*********************************** 全局变量 **********************************/

/*********************************** 函数定义 **********************************/

#define     BEGIN_EFFECT_MAP()          TEffectTable DirectEffect::m_effect_table[] = { 
#define     ADD_EFFECT(type, func)      {&DirectEffect::func, type},
#define     END_EFFECT_MAP              {nullptr,(EEffectType)EM_EFFECT_NONE}};
#define     MAP_SIZE(x)                 (int)(sizeof(x) / sizeof(x[0]))

BEGIN_EFFECT_MAP()
    ADD_EFFECT(EM_EFFECT_FADE,     EffectFade)
    ADD_EFFECT(EM_EFFECT_ERASEH,   EffectEraseH)
    ADD_EFFECT(EM_EFFECT_ERASEH,   EffectEraseV)
    ADD_EFFECT(EM_EFFECT_SHUTTERH, EffectShutterH)
    ADD_EFFECT(EM_EFFECT_SHUTTERV, EffectShutterV)
    ADD_EFFECT(EM_EFFECT_BOXIN,    EffectBoxIn)
    ADD_EFFECT(EM_EFFECT_BOXOUT,   EffectBoxOut)
END_EFFECT_MAP

/***********************************  类定义  **********************************/

DirectEffect::DirectEffect()
{

}

DirectEffect::~DirectEffect()
{    
    SAFE_DELETE_ARRAY(m_last_dict_image_ptr);
}

/*******************************************************************************
* 函  数: SetDictSize
* 描  述: 初始化帧尺寸
* 参  数: i_iWidth
* 参  数: i_iHeight
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::SetDictSize(int i_width, int i_height)
{
	if ( m_width != i_width || m_height != i_height )
	{
		m_width = i_width;
		m_height = i_height;
		m_frame_size = m_width*m_height * 3 / 2;

		SAFE_DELETE_ARRAY(m_last_dict_image_ptr);
	}
    if( m_last_dict_image_ptr == nullptr )
        m_last_dict_image_ptr = new uint8_t[m_frame_size];

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: SetEffect
* 描  述: 设置特效类型
* 参  数: i_eEffect
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::SetEffect(EEffectType i_effect)
{
    if( m_effect_state != EM_ES_END )
        m_tmp_effect = i_effect;
    else
        m_dict_effect = i_effect;

    LOG_INFO("Set Effect Type[%d]", i_effect);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UpdateLastDictImage
* 描  述: 保存特效底图
* 参  数: i_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::UpdateLastDictImage(uint8_t* i_dict_data_ptr)
{  
    CHECK_POINTER(i_dict_data_ptr, EM_RC_PARAM_ERR);
    CHECK_POINTER(m_last_dict_image_ptr, EM_RC_UNINIT_ERR);

	m_frame_count = 1;
	m_effect_state = EM_ES_RUN;
	memcpy(m_last_dict_image_ptr, i_dict_data_ptr, m_frame_size);

	if (m_dict_effect == EM_EFFECT_RANDOM)
	{
		m_tmp_effect = EM_EFFECT_RANDOM;
		m_dict_effect = (EEffectType)(EM_EFFECT_FADE + stGetCurTick() % EM_EFFECT_RANDOM);
	}
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: AddEffect
* 描  述: 特效处理
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::AddEffect(uint8_t* io_dict_data_ptr)
{
    if (m_effect_state != EM_ES_RUN)
        return EM_RC_OK;

    int index = m_dict_effect - EM_EFFECT_FADE;     
    if (index < 0 || index > MAP_SIZE(m_effect_table))
    {        
        LOG_ERROR("Effect Type[%d] Out Range", m_dict_effect);                        
        m_dict_effect = EM_EFFECT_NONE;
        return EM_RC_PARAM_ERR;
    }

    //查询处理表，调用函数
    if(m_effect_table[index].effect_func != nullptr)
    {   
        (this->*(m_effect_table[index].effect_func))(io_dict_data_ptr);     

        m_frame_count++;
        if( m_frame_count > EFFECT_RUN_COUNT )
        {
            m_frame_count = 0;
            m_effect_state = EM_ES_END;
            if( m_tmp_effect != EM_EFFECT_NONE )
            {
                m_dict_effect = m_tmp_effect;
                m_tmp_effect = EM_EFFECT_NONE;
            }
        }
    }   

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectFade
* 描  述: 淡入淡出
* 参  数: o_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月20日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectFade(uint8_t* io_dict_data_ptr)
{
    if( m_frame_count < EFFECT_RUN_COUNT )
    {
        uint8_t* dict_ptr = io_dict_data_ptr;
        uint8_t* back_ptr = m_last_dict_image_ptr;

        int alpha_offset = m_frame_count*EFFECT_ALPHA_COUNT/EFFECT_RUN_COUNT;
        for( int j = 0; j < m_frame_size; j++)
        {
            *dict_ptr = *dict_ptr * alpha_offset / EFFECT_ALPHA_COUNT + 
				*back_ptr *(EFFECT_ALPHA_COUNT-alpha_offset)/EFFECT_ALPHA_COUNT;
            dict_ptr++; back_ptr++;
        }
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectEraseH
* 描  述: 从左向右擦除
* 参  数: i_pStrmData
* 参  数: o_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectEraseH(uint8_t* io_dict_data_ptr)
{
    int offset = EFFECTEVEN(m_width*m_frame_count/EFFECT_RUN_COUNT);
    int uv_offset = offset/2;
    int uv_height = m_height/2;
    int uv_width = m_width/2;

    for(int i = 0; i < m_height; i++)
        memcpy(io_dict_data_ptr+i*m_width+offset, m_last_dict_image_ptr+i*m_width+offset,m_width-offset);

    uint8_t* u_dict_ptr = io_dict_data_ptr + m_width*m_height;
    uint8_t* u_image_ptr = m_last_dict_image_ptr+m_width*m_height;
    for(int i = 0; i < uv_height; i++)
        memcpy(u_dict_ptr+i*uv_width+uv_offset, u_image_ptr+i*uv_width+uv_offset,uv_width-uv_offset);

    u_dict_ptr = io_dict_data_ptr + m_width*m_height*5/4;
    u_image_ptr = m_last_dict_image_ptr+m_width*m_height*5/4;
    for(int i = 0; i < uv_height; i++)
        memcpy(u_dict_ptr+i*uv_width+uv_offset, u_image_ptr+i*uv_width+uv_offset,uv_width-uv_offset);

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectEraseV
* 描  述: 从上向下擦除
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectEraseV(uint8_t* io_dict_data_ptr)
{
    int offset = EFFECTEVEN(m_frame_count*m_height/EFFECT_RUN_COUNT);
    int uv_offset = offset/2;
    int uv_height = m_height/2;
    int uv_width = m_width/2;

    memcpy(io_dict_data_ptr+offset*m_width,m_last_dict_image_ptr+offset*m_width,(m_height - offset)*m_width);

    uint8_t* u_dict_ptr = io_dict_data_ptr+m_width*m_height;
    uint8_t* u_image_ptr = m_last_dict_image_ptr+m_width*m_height;
    memcpy(u_dict_ptr+uv_offset*uv_width,u_image_ptr+uv_offset*uv_width,(uv_height - uv_offset)*uv_width);

    u_dict_ptr = io_dict_data_ptr+m_width*m_height*5/4;
    u_image_ptr = m_last_dict_image_ptr+m_width*m_height*5/4;
    memcpy(u_dict_ptr+uv_offset*uv_width,u_image_ptr+uv_offset*uv_width,(uv_height - uv_offset)*uv_width);

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectShutterH
* 描  述: 水平百叶窗
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectShutterH(uint8_t* io_dict_data_ptr)
{
    int shutter = m_height/EFFECT_SHUTTER_COUNT;
    int offset = EFFECTEVEN(m_frame_count*shutter/EFFECT_RUN_COUNT);
    int cpy_v_pos = offset*m_width;
    int cpy_data_offset = shutter*m_width;
    int per_cpy_data_size = cpy_data_offset - cpy_v_pos;    
    int uv_cpy_v_pos = cpy_v_pos/4;
    int uv_per_cpy_data_size = per_cpy_data_size/4;
    int uv_cpy_data_offset = cpy_data_offset/4;

	if (per_cpy_data_size < 0)
		return EM_RC_OK;

    uint8_t* u_dict_ptr  = io_dict_data_ptr + m_width*m_height;
    uint8_t* u_image_ptr = m_last_dict_image_ptr+m_width*m_height;
    uint8_t* v_dict_ptr  = u_dict_ptr + m_width*m_height/4;
    uint8_t* v_image_ptr = u_image_ptr+m_width*m_height/4;

    for( int i = 0; i < EFFECT_SHUTTER_COUNT; i++ )
    {
        memcpy(io_dict_data_ptr+cpy_v_pos+i*cpy_data_offset, m_last_dict_image_ptr+cpy_v_pos+i*cpy_data_offset, per_cpy_data_size);
        memcpy(u_dict_ptr+uv_cpy_v_pos+i*uv_cpy_data_offset,u_image_ptr+uv_cpy_v_pos+i*uv_cpy_data_offset, uv_per_cpy_data_size);
        memcpy(v_dict_ptr+uv_cpy_v_pos+i*uv_cpy_data_offset,v_image_ptr+uv_cpy_v_pos+i*uv_cpy_data_offset,uv_per_cpy_data_size);
    }

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectShutterV
* 描  述: 垂直百叶窗
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectShutterV(uint8_t* io_dict_data_ptr)
{
    int shutter = m_width / EFFECT_SHUTTER_COUNT;
    int offset = EFFECTEVEN(m_frame_count*shutter/EFFECT_RUN_COUNT);
    int uv_height = m_height/2;
    int uv_width = m_width/2;
    int uv_offset = offset/2;
    int uv_shutter = shutter/2;
    int cpy_data_size = shutter - offset;
    int uv_cpy_data_size = uv_shutter - uv_offset;

	if (cpy_data_size < 0 || uv_cpy_data_size < 0)
		return EM_RC_OK;

    uint8_t* u_dict_ptr  = io_dict_data_ptr + m_width*m_height;
    uint8_t* u_image_ptr = m_last_dict_image_ptr+m_width*m_height;
    uint8_t* v_dict_ptr  = u_dict_ptr + m_width*m_height/4;
    uint8_t* v_image_ptr = u_image_ptr+m_width*m_height/4;

    for( int i = 0; i < EFFECT_SHUTTER_COUNT; i++ )
    {
        int j = 0;
        for( j = 0; j < m_height; j++ )            
            memcpy(io_dict_data_ptr+offset+j*m_width+i*shutter, 
						m_last_dict_image_ptr+offset+j*m_width+i*shutter,cpy_data_size);

        for( j = 0; j < uv_height; j++ )            
            memcpy(u_dict_ptr+uv_offset+j*uv_width+i*uv_shutter, u_image_ptr+uv_offset+j*uv_width+i*uv_shutter,uv_cpy_data_size);

        for( j = 0; j < uv_height; j++ )            
            memcpy(v_dict_ptr+uv_offset+j*uv_width+i*uv_shutter, v_image_ptr+uv_offset+j*uv_width+i*uv_shutter,uv_cpy_data_size);
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EffectBoxIn
* 描  述: 向内收缩
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectBoxIn(uint8_t* io_pDictData)
{
    int box_width = 2 * (EFFECTEVEN((m_width - m_frame_count*m_width / EFFECT_RUN_COUNT) / 2));
    int box_height = 2 * (EFFECTEVEN((m_height - m_frame_count*m_height / EFFECT_RUN_COUNT) / 2));
    int x_start = (m_width - box_width) / 2;
    int y_start = (m_height - box_height) / 2;
    int uv_width = m_width / 2;
    int uv_box_height = box_height / 2;
    int uv_box_width = box_width / 2;
    int y_offset = m_width*y_start + x_start;
    int uv_offset = uv_width*y_start / 2 + x_start / 2;

    int i = 0;
    for (i = 0; i < box_height; i++)
        memcpy(io_pDictData + y_offset + i*m_width, m_last_dict_image_ptr + y_offset + i*m_width, box_width);

    uint8_t* uv_dict_ptr = io_pDictData + m_width*m_height + uv_offset;
    uint8_t* uv_image_ptr = m_last_dict_image_ptr + m_width*m_height + uv_offset;
    for (i = 0; i < uv_box_height; i++)
        memcpy(uv_dict_ptr + i*uv_width, uv_image_ptr + i*uv_width, uv_box_width);

    uv_dict_ptr = io_pDictData + m_width*m_height * 5 / 4 + uv_offset;
    uv_image_ptr = m_last_dict_image_ptr + m_width*m_height * 5 / 4 + uv_offset;
    for (i = 0; i < uv_box_height; i++)
        memcpy(uv_dict_ptr + i*uv_width, uv_image_ptr + i*uv_width, uv_box_width);

    return EM_RC_OK;
}


/*******************************************************************************
* 函  数: EffectBoxOut
* 描  述: 向外
* 参  数: io_pDictData
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 16年10月27日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode DirectEffect::EffectBoxOut(uint8_t* io_pDictData)
{
    if (m_frame_count < EFFECT_RUN_COUNT)
    {
        uint8_t* dict_ptr = io_pDictData;
        uint8_t* back_ptr = m_last_dict_image_ptr;

        int alpha_offset = m_frame_count*EFFECT_ALPHA_COUNT / EFFECT_RUN_COUNT;
        for (int j = 0; j < m_frame_size; j++)
        {
            *dict_ptr = *dict_ptr * alpha_offset / EFFECT_ALPHA_COUNT +
						*back_ptr *(EFFECT_ALPHA_COUNT - alpha_offset) / EFFECT_ALPHA_COUNT;
            dict_ptr++; back_ptr++;
        }
    }
    return EM_RC_OK;
}
