/*------------------------------------------------------------------------------
* 描  述：创建源对象
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode RBSMediaSource::CreateSource(TSourceProps* i_src_props_ptr)
{
	CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

	if (i_src_props_ptr->media_mode == EM_MEDIA_MODE_SOFT)
		m_source_ptr = new FFMediaSource;
	else
		m_source_ptr = new VAMediaSource();

	CHECK_POINTER(m_source_ptr, EM_RC_MEMROY_ALLOC_ERR);
	if (m_source_ptr->CreateSource(i_src_props_ptr) != EM_RC_OK)
	{
		LOG_ERROR("Stream [%d]  Init Source Failed", i_src_props_ptr->user_data);
		SAFE_DELETE(m_source_ptr);
		return EM_RC_INIT_ERR;
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：输入数据

ERetCode RBSMediaSource::InputSrcData(uint8_t* i_src_data_ptr, int i_data_len)

	CHECK_POINTER(m_source_ptr, EM_RC_INIT_ERR);

}

/*------------------------------------------------------------------------------
* 描  述：创建合成源对象
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaMixerSource::CreateSource(TSourceProps* i_src_props_ptr)
{
	CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);
	if (i_src_props_ptr->media_mode == EM_MEDIA_MODE_SOFT)
		m_mixer_source_ptr = new FFMixerSource;
	else
		m_mixer_source_ptr = new VAMixerSource();

	CHECK_POINTER(m_mixer_source_ptr, EM_RC_MEMROY_ALLOC_ERR);
	if (m_mixer_source_ptr->CreateSource(i_src_props_ptr) != EM_RC_OK)
	{
		LOG_ERROR("Stream [%d]  Init Source Failed", i_src_props_ptr->user_data);
		SAFE_DELETE(m_mixer_source_ptr);
		return EM_RC_INIT_ERR;
	}

	return EM_RC_OK;
}
