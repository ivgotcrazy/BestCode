#include "media_proc_route.h"
#include "BaseLog.h"
#include "media_decoder.h"
#include "media_source.h"

/********************************** 类实现 ***********************************/

/*------------------------------------------------------------------------------
* 描  述：创建媒体模式输出
* 参  数：[in] i_render_props_ptr 渲染器的属性
* 参  数：[out] o_decoder_ptr  创建的渲染器
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaProcRoute::CreateMediaProc(EMediaMode i_dst_media_mode, TSourceProps* i_src_props_ptr)
{
	CHECK_POINTER(i_src_props_ptr, EM_RC_PARAM_ERR);

	std::unique_lock<std::mutex> lock(m_media_proc_mtx);

	if (i_dst_media_mode == i_src_props_ptr->media_mode)
		return EM_RC_STATE_ERR;

	m_dst_media_mode = i_dst_media_mode;
	memcpy(&m_src_props, i_src_props_ptr, sizeof(TSourceProps));

	//对外提供VAImage的数据接口
	m_src_props.media_mode = m_dst_media_mode;

	m_src_ptr = new RBSMediaSource;
	ERetCode emRC = m_src_ptr->CreateSource(&m_src_props);
	if ( emRC != EM_RC_OK )
	{
		LOG_ERROR("Create The Media Route Failed [%d]", emRC);
		return emRC;
	}

	return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：创建媒体模式输出
* 参  数：[in] i_pSrcData 媒体数据
* 参  数：[in] i_iDataLen 媒体数据大小
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaProcRoute::InputSrcData(uint8_t* i_src_data_ptr, int i_data_len)
{
	CHECK_POINTER(i_src_data_ptr, EM_RC_PARAM_ERR);
	CHECK_POINTER(m_src_ptr, EM_RC_INIT_ERR);

	std::unique_lock<std::mutex> lock(m_media_proc_mtx);

	return m_src_ptr->InputSrcData(i_src_data_ptr, i_data_len);
}

/*------------------------------------------------------------------------------
* 描  述：获取媒体数据
* 参  数：[in] o_data_ptr 媒体数据输出
* 返回值：VAImage或者数据指针 成功/nullptr 失败
-----------------------------------------------------------------------------*/
VideoSrcImage MediaProcRoute::GetReadyImage(TOutFrame* o_data_ptr)
{
	if (m_src_ptr == nullptr)
		return nullptr;

	std::unique_lock<std::mutex> lock(m_media_proc_mtx);

	return m_src_ptr->GetReadyImage(o_data_ptr);
}

/*------------------------------------------------------------------------------
* 描  述：获取源属性
* 参  数：[in] 无
* 返回值：源属性指针
-----------------------------------------------------------------------------*/
TSourceProps* MediaProcRoute::GetSrcProps()
{
	return &m_src_props;
}

/*------------------------------------------------------------------------------
* 描  述：销毁媒体输出模式
* 参  数：[in] 无
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode MediaProcRoute::DestroyMediaProc()
{
	std::unique_lock<std::mutex> lock(m_media_proc_mtx);
	if ( m_src_ptr )
	{
		m_src_ptr->DestroySource();
		SAFE_DELETE(m_src_ptr);
	}

	LOG_INFO("Source [%d] Destroy The Media Proc Route", m_src_props.user_data);

	return EM_RC_OK;
}
