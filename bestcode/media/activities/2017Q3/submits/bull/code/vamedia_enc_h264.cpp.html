static int BuildPackedSeiBufferTiming(unsigned int init_cpb_removal_length,
                               unsigned int init_cpb_removal_delay,
                               unsigned int init_cpb_removal_delay_offset,
                               unsigned int cpb_removal_length,
                               unsigned int cpb_removal_delay,
                               unsigned int dpb_output_length,
                               unsigned int dpb_output_delay,
                               uint8_t **sei_buffer)
{
    uint8_t *byte_buf;
    int bp_byte_size, i, pic_byte_size;

    BitStream nal_bs;
    BitStream sei_bp_bs, sei_pic_bs;

    BitStreamStart(&sei_bp_bs);
    BitStreamPutUE(&sei_bp_bs, 0);       /*seq_parameter_set_id*/
    BitStreamPutUI(&sei_bp_bs, init_cpb_removal_delay, cpb_removal_length); 
    BitStreamPutUI(&sei_bp_bs, init_cpb_removal_delay_offset, cpb_removal_length); 
    if ( sei_bp_bs.bit_offset & 0x7) {
        BitStreamPutUI(&sei_bp_bs, 1, 1);
    }
    BitStreamEnd(&sei_bp_bs);
    bp_byte_size = (sei_bp_bs.bit_offset + 7) / 8;

    BitStreamStart(&sei_pic_bs);
    BitStreamPutUI(&sei_pic_bs, cpb_removal_delay, cpb_removal_length); 
    BitStreamPutUI(&sei_pic_bs, dpb_output_delay, dpb_output_length); 
    if ( sei_pic_bs.bit_offset & 0x7) {
        BitStreamPutUI(&sei_pic_bs, 1, 1);
    }
    BitStreamEnd(&sei_pic_bs);
    pic_byte_size = (sei_pic_bs.bit_offset + 7) / 8;

    BitStreamStart(&nal_bs);
    NalStartCodePrefix(&nal_bs);
    NalHeader(&nal_bs, NAL_REF_IDC_NONE, NAL_SEI);

    /* Write the SEI buffer period data */    
    BitStreamPutUI(&nal_bs, 0, 8);
    BitStreamPutUI(&nal_bs, bp_byte_size, 8);

    byte_buf = (uint8_t *)sei_bp_bs.buffer;
    for(i = 0; i < bp_byte_size; i++) 
    {
        BitStreamPutUI(&nal_bs, byte_buf[i], 8);
    }
    free(byte_buf);
    /* write the SEI timing data */
    BitStreamPutUI(&nal_bs, 0x01, 8);
    BitStreamPutUI(&nal_bs, pic_byte_size, 8);

    byte_buf = (uint8_t *)sei_pic_bs.buffer;
    for(i = 0; i < pic_byte_size; i++) {
        BitStreamPutUI(&nal_bs, byte_buf[i], 8);
    }
    free(byte_buf);

    RBSPTrailingBits(&nal_bs);
    BitStreamEnd(&nal_bs);

    *sei_buffer = (uint8_t *)nal_bs.buffer; 

    return nal_bs.bit_offset;
}

* 函  数: Init
* 描  述: 初始化编码器
* 参  数: i_pEncInfo
* 参  数: i_pCallBack
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码

ERetCode VAEncH264::CreateEncoder(TEncInfo* i_enc_info_ptr)
{   
    CHECK_POINTER(i_enc_info_ptr, EM_RC_PARAM_ERR);
    if (i_enc_info_ptr->encode_size.width == 0 || i_enc_info_ptr->encode_size.height == 0)
        return EM_RC_PARAM_ERR;

    if( m_is_va_inited )
        return EM_RC_INITED_ERR;

    memcpy(m_enc_info_ptr, i_enc_info_ptr, sizeof(*m_enc_info_ptr));
    ERetCode er = InitVAAPI();
    if(EM_RC_OK != er)
    {
        LOG_ERROR("Init Vaapi Fail[%d]",er);
        memset(m_enc_info_ptr,0,sizeof(TEncInfo));
        return er;
    }
        
    m_out_width = m_enc_info_ptr->encode_size.width;
    m_out_height = m_enc_info_ptr->encode_size.height;
    m_align_width = AlignSize(m_out_width);
    m_align_height = AlignSize(m_out_height);   

    VAStatus va_status = vaCreateConfig(g_va_display, m_va_profile, VAEntrypointEncSlice,
                                       &m_conf_attrib[0], m_conf_attrib_num, &m_va_config_id);
    CHECK_VASTATUS(va_status, "vaCreateConfig");
    /* create source surfaces */

    // va1.8.0必须设置为NV12surface，否则如果put的是yv12数据，surface会变成yv12，assert退出
    VASurfaceAttrib va_surf_attr;
    va_surf_attr.type = VASurfaceAttribPixelFormat;
    va_surf_attr.flags = VA_SURFACE_ATTRIB_SETTABLE;
    va_surf_attr.value.type = VAGenericValueTypeInteger;
    va_surf_attr.value.value.i = VA_FOURCC_NV12;
    va_status = vaCreateSurfaces(g_va_display,VA_RT_FORMAT_YUV420,m_align_width,m_align_height,
        &m_va_surface_id[0], VA_SURFACE_SIZE, &va_surf_attr, 1);
    CHECK_VASTATUS(va_status, "vaCreateSurfaces");

    VASurfaceID* va_Sur_id = (VASurfaceID*)calloc(VA_SURFACE_SIZE, sizeof(VASurfaceID));
    memcpy(va_Sur_id,m_va_surface_id, VA_SURFACE_SIZE * sizeof(VASurfaceID));
    
    /* Create a context for this encode pipe */
    va_status = vaCreateContext(g_va_display, m_va_config_id,m_align_width,m_align_height,
                                VA_PROGRESSIVE,va_Sur_id, VA_SURFACE_SIZE,&m_va_context_id);
    CHECK_VASTATUS(va_status, "vaCreateContext");
    free(va_Sur_id);

    int buf_size = (m_align_width * m_align_height * 400) / (16*16);
    for(int i = 0; i < VA_SURFACE_SIZE; i++)
    {   
        /* create coded buffer once for all
         * other VA buffers which won't be used again after vaRenderPicture.
         * so APP can always vaCreateBuffer for every frame
         * but coded buffer need to be mapped and accessed after vaRenderPicture/vaEndPicture
         * so VA won't maintain the coded buffer */
        va_status = vaCreateBuffer(g_va_display,m_va_context_id,VAEncCodedBufferType,buf_size,1,NULL,&m_va_buffer_id[i]);
        CHECK_VASTATUS(va_status,"vaCreateBuffer");
    }
    
    m_idr_period = m_enc_info_ptr->fps*2;        
    int max_rate = m_out_width * m_out_height * 12 * m_enc_info_ptr->fps / 50;
    if(m_enc_info_ptr->bitrate == 0)
        m_enc_info_ptr->bitrate = max_rate;
    if(m_enc_info_ptr->bitrate > max_rate && max_rate )
        m_enc_info_ptr->bitrate = max_rate;
    
    LOG_INFO("Create Encoder Success, Size[%dx%d], BitRate[%d], Quality[%d], UseData[%ld]",
        m_align_width,m_align_height,m_enc_info_ptr->bitrate,m_initial_qp, m_enc_info_ptr->userdata);
    
    m_is_va_inited = 1;
    return CreateThreads();
}// NOLINT(readability/fn_size)

/*******************************************************************************
* 函  数: InitVAAPI
* 描  述: 初始化VAAPI参数
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月09日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAEncH264::InitVAAPI()
{    
    int ep_num = 0, slice_ep = 0, support_enc = 0;
    ep_num = vaMaxNumEntrypoints(g_va_display);
    VAEntrypoint* entry_point_ptr = (VAEntrypoint*)malloc(ep_num * sizeof(*entry_point_ptr));
    if (!entry_point_ptr)
    {
        LOG_ERROR("Init EntryPoints Array Fail");
        return EM_RC_MEMROY_ALLOC_ERR;
    }

    m_va_profile = VAGetProfile(m_enc_info_ptr->profile);
    if (VAProfileNone == m_va_profile)
    {
        LOG_ERROR("Get VAProfile Failed!");
        return EM_RC_VA_PROFILE_ERR;
    }

    vaQueryConfigEntrypoints(g_va_display, m_va_profile, entry_point_ptr, &ep_num);
    for (slice_ep = 0; slice_ep < ep_num; slice_ep++)
    {
        if (entry_point_ptr[slice_ep] == VAEntrypointEncSlice)
        {
            support_enc = 1;
            break;
        }
    }

    if (support_enc == 0)
    {
        LOG_ERROR("Can't Support Encode H264!");
        return EM_RC_ENC_SUPPORT_ERR;
    }

    switch (m_va_profile)
    {
    case VAProfileH264Baseline:
        m_ip_period = 1;
        m_constraint_flag |= (1 << 0); /* Annex A.2.1 */
        m_h264_entropy_mode = 0;
        break;
    case VAProfileH264ConstrainedBaseline:
        m_constraint_flag |= (1 << 0 | 1 << 1); /* Annex A.2.2 */
        m_ip_period = 1;
        break;
    case VAProfileH264Main:
        m_constraint_flag |= (1 << 1); /* Annex A.2.2 */
        break;
    case VAProfileH264High:
        m_constraint_flag |= (1 << 3); /* Annex A.2.4 */
        break;
    default:
        LOG_WARN("Unknow Profile. Set To High!");
        m_constraint_flag |= (1 << 3); /* Annex A.2.4 */
        break;
    }

    /* find out the format for the render target, and rate control mode */
    for (int i = 0; i < VAConfigAttribTypeMax; i++)
        m_attrib[i].type = (VAConfigAttribType)i;

    VAStatus va_status = vaGetConfigAttributes(g_va_display, m_va_profile,
        VAEntrypointEncSlice, &m_attrib[0], VAConfigAttribTypeMax);
    CHECK_VASTATUS(va_status, "vaGetConfigAttributes");
    free(entry_point_ptr);

    return CheckVAAttri();    
}

/*******************************************************************************
* 函  数: CheckVAAttri
* 描  述: 
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月13日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAEncH264::CheckVAAttri()
{
    /* check the interested configattrib */
    if ((m_attrib[VAConfigAttribRTFormat].value & VA_RT_FORMAT_YUV420) == 0)
    {
        LOG_ERROR("Not Find Desired YUV420 RTFormat");
        return EM_RC_YUV_FORMAT_ERR;
    }
    m_conf_attrib[m_conf_attrib_num].type = VAConfigAttribRTFormat;
    m_conf_attrib[m_conf_attrib_num].value = VA_RT_FORMAT_YUV420;
    m_conf_attrib_num++;
    
    if (m_attrib[VAConfigAttribRateControl].value != VA_ATTRIB_NOT_SUPPORTED) 
    {       
        m_rate_control = VAGetRateControl(m_enc_info_ptr->encode_mode);
        if( m_rate_control == VA_RC_NONE )
        {
            LOG_ERROR("Has't Support RateControl Mode[%d]", m_enc_info_ptr->encode_mode);
            return EM_RC_ENC_SUPPORT_ERR;
        }

        m_conf_attrib[m_conf_attrib_num].type = VAConfigAttribRateControl;
        m_conf_attrib[m_conf_attrib_num].value = m_rate_control;//m_aAttrib[VAConfigAttribRateControl].value;
        m_conf_attrib_num++;
    }

    if (m_attrib[VAConfigAttribEncPackedHeaders].value != VA_ATTRIB_NOT_SUPPORTED) 
    {//not use
        int iValue = m_attrib[VAConfigAttribEncPackedHeaders].value;

        m_packed_header = 1;
        m_conf_attrib[m_conf_attrib_num].type = VAConfigAttribEncPackedHeaders;
        m_conf_attrib[m_conf_attrib_num].value = VA_ENC_PACKED_HEADER_NONE;

        if (iValue & VA_ENC_PACKED_HEADER_SEQUENCE) 
            m_conf_attrib[m_conf_attrib_num].value |= VA_ENC_PACKED_HEADER_SEQUENCE;
        if (iValue & VA_ENC_PACKED_HEADER_PICTURE)         
            m_conf_attrib[m_conf_attrib_num].value |= VA_ENC_PACKED_HEADER_PICTURE;
        if (iValue & VA_ENC_PACKED_HEADER_SLICE) 
            m_conf_attrib[m_conf_attrib_num].value |= VA_ENC_PACKED_HEADER_SLICE;
        if (iValue & VA_ENC_PACKED_HEADER_MISC) 
            m_conf_attrib[m_conf_attrib_num].value |= VA_ENC_PACKED_HEADER_MISC;
        
        m_enc_packed_header_idx = m_conf_attrib_num;
        m_conf_attrib_num++;
    }   

    if (m_attrib[VAConfigAttribEncMaxRefFrames].value != VA_ATTRIB_NOT_SUPPORTED) 
        m_h264_max_ref = m_attrib[VAConfigAttribEncMaxRefFrames].value;
        
    return EM_RC_OK;
}

ERetCode VAEncH264::CreateThreads()
{
    m_enc_thread_state = EM_TS_RUN;    
    ERetCode er = StartThreadEx(EncodeThread, this);
    if (er != EM_RC_OK)
    {
        LOG_ERROR("Create EncodeThread Fail[%d]", er);
        return er;
    }

    m_task_thread_state = EM_TS_RUN;
    er = StartThreadEx(StorageTaskThread, this);
    if (er != EM_RC_OK)
    {
        LOG_ERROR("Create StorageTaskFunc Fail[%d]", er);
        return er;
    }    
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: DestroyThreads
* 描  述: 
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月13日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAEncH264::DestroyThreads()
{
    if (m_enc_thread_state != EM_TS_STOP)
    {
        m_enc_thread_state = EM_TS_STOPING;
        while (m_enc_thread_state != EM_TS_STOP)
            usleep(0);
    }
    if (m_task_thread_state != EM_TS_STOP)
    {
        pthread_mutex_lock(&m_encode_mtx);
        m_task_thread_state = EM_TS_STOPING; 
        pthread_cond_signal(&m_encode_cond);
        pthread_mutex_unlock(&m_encode_mtx);        
        while (m_task_thread_state != EM_TS_STOP)
            usleep(0);
    }
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: UpdateDisplayOrder
* 描  述: 更新帧序号和帧类型
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月10日
* 作  者: 刘峰(Arthur)
********************************************************************************/
void VAEncH264::UpdateDisplayOrder()
{
    int enc_gop = 0;

    if (m_idr_period == 1) 
    { /* all are I/IDR frames */
        m_curr_frame_display = m_enc_frame;
        if (m_idr_period == 0)
            m_curr_frame_type = (m_enc_frame == 0)?FRAME_IDR:FRAME_I;
        else
            m_curr_frame_type = (m_enc_frame % m_idr_period == 0)?FRAME_IDR:FRAME_I;       
        return;
    }
    /* new sequence like
     * IDR PPPPP IPPPPP
     * IDR (PBB)(PBB)(IBB)(PBB)
     */
    enc_gop = (m_idr_period == 0)? m_enc_frame:
        (m_enc_frame % (m_idr_period + ((m_ip_period == 1)?0:1)));
         
    if (enc_gop == 0) 
    { /* the first frame */
        m_curr_frame_type = FRAME_IDR;
        m_curr_frame_display = m_enc_frame;
    }
    else if (((enc_gop - 1) % m_ip_period) != 0) 
    { /* B frames */
        m_curr_frame_type = FRAME_B;
        m_curr_frame_display = m_enc_frame - 1;
    }
    else if ((m_idr_period != 0) && /* have I frames */
        (enc_gop >= 2) &&
        ((m_ip_period == 1 && enc_gop % m_idr_period == 0) || /* for IDR PPPPP IPPPP */
        /* for IDR (PBB)(PBB)(IBB) */
        (m_ip_period >= 2 && ((enc_gop - 1) / m_ip_period % (m_idr_period / m_ip_period)) == 0))) 
    {
        m_curr_frame_type = FRAME_I;
        m_curr_frame_display = m_enc_frame + m_ip_period - 1;
    } 
    else 
    {
        m_curr_frame_type = FRAME_P;
        m_curr_frame_display = m_enc_frame + m_ip_period - 1;
    }
}

/*******************************************************************************
* 函  数: EncodeThreadFunc
* 描  述: 编码线程函数
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年06月13日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAEncH264::EncodeThreadFunc()
{
    int surf_idx = 0, loop_num = 0;
    uint64_t tick = stGetCurrentMicroSecond();
    uint64_t frame_timestamp = COMMON_TICK_BASE/COMMON_DEFAULT_FPS;
    TSourceProps* src_props_ptr = nullptr;
    VAImage* image_ptr = nullptr;

    if (m_enc_info_ptr->fps)
        frame_timestamp = COMMON_TICK_BASE/m_enc_info_ptr->fps;
    
    LOG_INFO("Start EncodeThread[%ld]", m_enc_info_ptr->userdata);
    while (m_enc_thread_state == EM_TS_RUN)
    {
        if (m_source_ptr == nullptr)
        {
            usleep(COMMON_TICK_SECOND);
            continue;
        }

        pthread_mutex_lock(&m_source_mtx);
        
        m_source_ptr->LockReadyImage(); 
        src_props_ptr = m_source_ptr->GetSrcProps();       
        image_ptr = static_cast<VAImage*>(m_source_ptr->GetReadyImage());
        if (image_ptr == nullptr || src_props_ptr == nullptr)
        {
            m_source_ptr->UnLockReadyImage();
            pthread_mutex_unlock(&m_source_mtx);
            usleep(10000);
            continue;
        }
        
        vaPutImage(g_va_display, m_va_surface_id[surf_idx], image_ptr->image_id,
                0, 0, src_props_ptr->src_size.width, src_props_ptr->src_size.height,
                0, 0, m_enc_info_ptr->encode_size.width, m_enc_info_ptr->encode_size.height);        
        m_source_ptr->UnLockReadyImage();
        pthread_mutex_unlock(&m_source_mtx);
        
        pthread_mutex_lock(&m_encode_mtx);
        m_lst_enc_surf_idx.push_back(surf_idx);
        pthread_cond_signal(&m_encode_cond);
        pthread_mutex_unlock(&m_encode_mtx);

        surf_idx++;
        if (surf_idx == VA_SURFACE_SIZE)
            surf_idx = 0;
        
        loop_num++;
        while (stGetCurrentMicroSecond() - tick < loop_num * frame_timestamp)
            usleep(COMMON_TICK_SECOND);
    }

    m_enc_thread_state = EM_TS_STOP;
    LOG_INFO("Stop EncodeThread[%ld]", m_enc_info_ptr->userdata);
    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: EncodeFrame
* 描  述: 编码
* 参  数: i_vaSurfID: 编码surface的id
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月10日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAEncH264::EncodeFrame(VASurfaceID i_va_surf_id)
{
    VAStatus va_status;
    va_status = vaBeginPicture(g_va_display, m_va_context_id, i_va_surf_id);
    CHECK_VASTATUS(va_status,"vaBeginPicture");

    RenderSequence();       
    RenderPicture(); 
    if (m_curr_frame_type == FRAME_IDR && m_packed_header)
    {
        RenderPackedSequence();
        RenderPackedPicture();
    }
    if(m_rate_control == VA_RC_CBR)
    {
        RenderPackedSei();
        RenderHrd();
    }   
    RenderSlice();

    va_status = vaEndPicture(g_va_display,m_va_context_id);
    CHECK_VASTATUS(va_status,"vaEndPicture");

    return EM_RC_OK;
}

/*******************************************************************************
* 函  数: StorageTask
* 描  述: 执行编码任务
* 参  数: i_iSurfIdx
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 17年08月01日
* 作  者: 刘峰(Arthur)
*******************************************************************************/
ERetCode VAEncH264::RunEncodeTask(int i_surf_idx)
{   
    //更新帧序号和帧类型    
    if(!m_enc_start_tick)
        m_enc_start_tick = stGetCurTick();

    UpdateDisplayOrder();
    if (m_curr_frame_type == FRAME_IDR)
    {//idr帧，清空引用帧计数 
        m_short_term_num = 0;
        m_enc_curr_frame = 0;
        m_curr_idr_display = m_curr_frame_display;
    }

    ERetCode er = EncodeFrame(m_va_surface_id[i_surf_idx]);
    if (er != EM_RC_OK)
        return er;

    VAStatus va_status = vaSyncSurface(g_va_display, m_va_surface_id[i_surf_idx]);
    CHECK_VASTATUS(va_status, "vaSyncSurface");

    VACodedBufferSegment* segment_ptr = NULL;
    va_status = vaMapBuffer(g_va_display, m_va_buffer_id[i_surf_idx], (void **)(&segment_ptr));
    CHECK_VASTATUS(va_status, "vaMapBuffer");
    if (segment_ptr != NULL)
    {
        TOutFrame enc_frame;    
        enc_frame.frame_data_ptr = (uint8_t*)segment_ptr->buf;
        enc_frame.data_len = segment_ptr->size;
        enc_frame.width = m_enc_info_ptr->encode_size.width;
        enc_frame.height = m_enc_info_ptr->encode_size.height;
        enc_frame.key_frame_flag = (m_curr_frame_type == FRAME_IDR);
        if (m_callback_ptr != nullptr)
            m_callback_ptr->OnEncodeCallBack(&enc_frame, m_enc_info_ptr->userdata);
    }
    vaUnmapBuffer(g_va_display, m_va_buffer_id[i_surf_idx]);
    
    m_enc_frame++;
    if (m_enc_frame % COMMON_SIZE_128K == 0)
    {// 每10多分钟打印一次帧率
        LOG_INFO("Encode[%ld] Fps[%.3f], Frames[%d]", m_enc_info_ptr->userdata,
            (float)m_enc_frame * COMMON_TICK_SECOND / (stGetCurTick() - m_enc_start_tick), m_enc_frame);
    }
    return UpdateRefFrames();
}

* 描  述: 编码线程

*******************************************************************************/
void* VAEncH264::StorageTaskThread(void* i_args_ptr)
{
    VAEncH264* encode_ptr = static_cast<VAEncH264 *>(i_args_ptr);
    if (encode_ptr != nullptr)
        encode_ptr->StorageTaskThreadFunc();
    return nullptr;
}

/*******************************************************************************
* 函  数: StorageTaskThreadFunc
* 描  述: 编码数据回调线程
* 返回值: enumRC_OK: 成功 enumRC_XX: 失败错误码
* 时  间: 2015年12月09日
* 作  者: 刘峰(Arthur)
********************************************************************************/
ERetCode VAEncH264::StorageTaskThreadFunc()
{       
    timespec tm_cond;
    int surf_idx = 0;

    LOG_INFO("Start StorageTaskThreadFunc[%ld]",m_enc_info_ptr->userdata);
    while (m_task_thread_state == EM_TS_RUN)
    {
        pthread_mutex_lock(&m_encode_mtx);
        if (m_lst_enc_surf_idx.size() <= 0)
        {// 无数据时等待
            tm_cond.tv_nsec = 0;
            tm_cond.tv_sec = (long)time(nullptr) + COMMON_TICK_THREADTMO;

            pthread_cond_timedwait(&m_encode_cond, &m_encode_mtx, &tm_cond);
            pthread_mutex_unlock(&m_encode_mtx);
            continue;
        }

        surf_idx = m_lst_enc_surf_idx.front();
        m_lst_enc_surf_idx.pop_front();
        pthread_mutex_unlock(&m_encode_mtx);

        RunEncodeTask(surf_idx);
    }

    m_task_thread_state = EM_TS_STOP;
    LOG_INFO("Stop StorageTaskThreadFunc[%d]", m_enc_info_ptr->userdata);
    return EM_RC_OK;
}

/*------------------------------------------------------------------------------
* 描  述：va处理sei信息
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
ERetCode VAEncH264::RenderPackedSei()
{
    VAStatus va_status;
    VABufferID va_render_x11_id[2];
    VABufferID va_packed_sei_buffer_id;
    VABufferID va_packed_sei_header_param_buffer_id;
    VAEncPackedHeaderParameterBuffer* pack_header_param_buffer;
    
    int cpb_removal_delay, cpb_removal_delay_length,dpb_output_delay_length;
    int init_cpb_size, target_bit_rate, init_cpb_removal_delay_length, init_cpb_removal_delay;
    uint8_t *build_packed_sei_buffer = NULL,*packed_sei_buffer = NULL;
    uint32_t len_In_bits /*offset_in_bytes*/;

    target_bit_rate = m_enc_info_ptr->bitrate;
    init_cpb_size = (target_bit_rate * 8) >> 10;
    init_cpb_removal_delay = (init_cpb_size<<1)*1024/target_bit_rate * 90000;

    cpb_removal_delay = 2;
    init_cpb_removal_delay_length = 24;
    cpb_removal_delay_length = 24;
    dpb_output_delay_length = 24;

    len_In_bits = BuildPackedSeiBufferTiming(
        init_cpb_removal_delay_length,
        init_cpb_removal_delay,
        0,
        cpb_removal_delay_length,
        cpb_removal_delay * m_enc_curr_frame,
        dpb_output_delay_length,
        0,
        &build_packed_sei_buffer);

    va_status = vaCreateBuffer(g_va_display,
        m_va_context_id,
        VAEncPackedHeaderParameterBufferType,
        sizeof(VAEncPackedHeaderParameterBuffer), 1, NULL,
        &va_packed_sei_header_param_buffer_id);
    CHECK_VASTATUS(va_status,"vaCreateBuffer");
        
    vaMapBuffer(g_va_display,va_packed_sei_header_param_buffer_id,(void**)&pack_header_param_buffer);        
    pack_header_param_buffer->type = VAEncPackedHeaderH264_SEI;
    pack_header_param_buffer->bit_length = len_In_bits;
    pack_header_param_buffer->has_emulation_bytes = 0;
    vaUnmapBuffer(g_va_display,va_packed_sei_header_param_buffer_id);
    
    va_status = vaCreateBuffer(g_va_display,
        m_va_context_id,
        VAEncPackedHeaderDataBufferType,
        (len_In_bits + 7) / 8, 1, NULL,
        &va_packed_sei_buffer_id);
    CHECK_VASTATUS(va_status,"vaCreateBuffer");

    /* it comes for the bps defined in SPS */
    vaMapBuffer(g_va_display,va_packed_sei_buffer_id,(void**)&packed_sei_buffer); 
    
    memcpy(packed_sei_buffer,build_packed_sei_buffer,(len_In_bits + 7) / 8);
    free(build_packed_sei_buffer);
    vaUnmapBuffer(g_va_display,va_packed_sei_buffer_id);

    va_render_x11_id[0] = va_packed_sei_header_param_buffer_id;
    va_render_x11_id[1] = va_packed_sei_buffer_id;
    va_status = vaRenderPicture(g_va_display,m_va_context_id, va_render_x11_id, 2);
    CHECK_VASTATUS(va_status,"vaRenderPicture");
    
    vaDestroyBuffer(g_va_display,va_packed_sei_header_param_buffer_id);
    vaDestroyBuffer(g_va_display,va_packed_sei_buffer_id);

    return EM_RC_OK;
}

/*------------------------------------------------------------------------------

ERetCode VAEncH264::RenderSlice()
{
    int i = 0;
    VAStatus va_status;
    VABufferID va_enc_slice_param_buffer_id;
    VAEncSliceParameterBufferH264* slice_param_ptr = NULL;
    
    UpdateRefPicList();
        
    /* one frame, one slice */
    m_va_enc_slice_param.macroblock_address = 0;
    m_va_enc_slice_param.num_macroblocks = m_align_width * m_align_height/(16*16); /* Measured by MB */
    m_va_enc_slice_param.slice_type = (m_curr_frame_type == FRAME_IDR)?2:m_curr_frame_type;
    if (m_curr_frame_type == FRAME_IDR) 
    {
        if (m_enc_curr_frame != 0)
            ++m_va_enc_slice_param.idr_pic_id;
    }
    else if (m_curr_frame_type == FRAME_P)
    {
        int refpiclist0_max = m_h264_max_ref & 0xffff;
        memcpy(m_va_enc_slice_param.RefPicList0, m_va_ref_pic_list0_P, refpiclist0_max*sizeof(VAPictureH264));

        for (i = refpiclist0_max; i < 32; i++) {

            m_va_enc_slice_param.RefPicList0[i].flags = VA_PICTURE_H264_INVALID;
        }
    } 
    else if (m_curr_frame_type == FRAME_B) 
    {
        int refpiclist0_max = m_h264_max_ref & 0xffff;
        int refpiclist1_max = (m_h264_max_ref >> 16) & 0xffff;

        memcpy(m_va_enc_slice_param.RefPicList0, m_va_ref_pic_list0_B, refpiclist0_max*sizeof(VAPictureH264));
        for (i = refpiclist0_max; i < 32; i++)

            m_va_enc_slice_param.RefPicList0[i].flags = VA_PICTURE_H264_INVALID;

        }
    }

    m_va_enc_slice_param.slice_alpha_c0_offset_div2 = 0;
    m_va_enc_slice_param.slice_beta_offset_div2 = 0;
    m_va_enc_slice_param.direct_spatial_mv_pred_flag = 1;
    m_va_enc_slice_param.pic_order_cnt_lsb = (m_curr_frame_display - m_curr_idr_display) % m_max_pic_order_cnt_lsb;

    if (m_packed_header &&  
		m_conf_attrib[m_enc_packed_header_idx].value & VA_ENC_PACKED_HEADER_SLICE)
        RenderPackedSlice();
       
	
    va_status = vaCreateBuffer(g_va_display,m_va_context_id,VAEncSliceParameterBufferType,
        sizeof(VAEncSliceParameterBufferH264),1,NULL,&va_enc_slice_param_buffer_id);
    CHECK_VASTATUS(va_status,"vaCreateBuffer-EncSliceParam");

    memcpy(slice_param_ptr,&m_va_enc_slice_param,sizeof(VAEncSliceParameterBufferH264));
    vaUnmapBuffer(g_va_display,va_enc_slice_param_buffer_id);

    va_status = vaRenderPicture(g_va_display,m_va_context_id, &va_enc_slice_param_buffer_id,1);
    CHECK_VASTATUS(va_status,"vaRenderPicture");
    
    vaDestroyBuffer(g_va_display,va_enc_slice_param_buffer_id);   
    return EM_RC_OK;

{
    if (m_curr_frame_type == FRAME_B)
        return EM_RC_OK;
    
    m_va_curr_pic.flags = VA_PICTURE_H264_SHORT_TERM_REFERENCE;
    m_short_term_num++;
    if (m_short_term_num > m_ref_frame)

        m_enc_curr_frame++;

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
int VAEncH264::CalcPoc(int i_iPicOrderCntLsb)
{
    static int pic_order_cnt_msb_ref = 0, pic_order_cnt_lsb_ref = 0;
    int prev_pic_order_cnt_msb, prev_pic_order_cnt_lsb;
    int pic_order_cnt_msb, top_field_order_cnt;

    if (m_curr_frame_type == FRAME_IDR)
        prev_pic_order_cnt_msb = prev_pic_order_cnt_lsb = 0;
    else 
    {
        prev_pic_order_cnt_msb = pic_order_cnt_msb_ref;
        prev_pic_order_cnt_lsb = pic_order_cnt_lsb_ref;
    }

    if ((i_iPicOrderCntLsb < prev_pic_order_cnt_lsb) && 
        ((prev_pic_order_cnt_lsb - i_iPicOrderCntLsb) >= (int)(m_max_pic_order_cnt_lsb / 2)))
        pic_order_cnt_msb = prev_pic_order_cnt_msb + m_max_pic_order_cnt_lsb;
    else if ((i_iPicOrderCntLsb > prev_pic_order_cnt_lsb) &&
        ((i_iPicOrderCntLsb - prev_pic_order_cnt_lsb) > (int)(m_max_pic_order_cnt_lsb / 2)))
        pic_order_cnt_msb = prev_pic_order_cnt_msb - m_max_pic_order_cnt_lsb;
    else
        pic_order_cnt_msb = prev_pic_order_cnt_msb;
    
    top_field_order_cnt = pic_order_cnt_msb + i_iPicOrderCntLsb;

    if (m_curr_frame_type != FRAME_B)
    {
        pic_order_cnt_msb_ref = pic_order_cnt_msb;
        pic_order_cnt_lsb_ref = i_iPicOrderCntLsb;
    }

    return top_field_order_cnt;
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
int VAEncH264::BuildPackedSliceBuffer(uint8_t** i_header_buffer_ptr)
{
    BitStream bs;
    int is_idr = !!m_va_enc_pic_param.pic_fields.bits.idr_pic_flag;
    int is_ref = !!m_va_enc_pic_param.pic_fields.bits.reference_pic_flag;

    BitStreamStart(&bs);
    NalStartCodePrefix(&bs);

    if (IS_I_SLICE(m_va_enc_slice_param.slice_type)) 
        NalHeader(&bs, NAL_REF_IDC_HIGH, is_idr ? NAL_IDR : NAL_NON_IDR);
    else if (IS_P_SLICE(m_va_enc_slice_param.slice_type)) 
        NalHeader(&bs, NAL_REF_IDC_MEDIUM, NAL_NON_IDR);
    else
        NalHeader(&bs, is_ref ? NAL_REF_IDC_LOW : NAL_REF_IDC_NONE, NAL_NON_IDR);

    SliceHeader(&bs);
    BitStreamEnd(&bs);

    *i_header_buffer_ptr = (uint8_t *)bs.buffer;
    return bs.bit_offset;
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：

int VAEncH264::BuildPackedSeqBuffer(uint8_t** i_seq_buffer_ptr)
{

    BitStreamStart(&bs);
    NalStartCodePrefix(&bs);
    NalHeader(&bs, NAL_REF_IDC_HIGH, NAL_SPS);
    SPS_Rbsp(&bs);
    BitStreamEnd(&bs);

    *i_seq_buffer_ptr = (uint8_t *)bs.buffer;
    return bs.bit_offset;
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void VAEncH264::SPS_Rbsp(BitStream *i_bs_ptr)
{
    int profile_idc = PROFILE_IDC_BASELINE;

    if (m_va_profile  == VAProfileH264High)
        profile_idc = PROFILE_IDC_HIGH;
    else if (m_va_profile  == VAProfileH264Main)
        profile_idc = PROFILE_IDC_MAIN;

    BitStreamPutUI(i_bs_ptr, profile_idc, 8);               /* profile_idc */
    BitStreamPutUI(i_bs_ptr, !!(m_constraint_flag & 1), 1);                         /* constraint_set0_flag */
    BitStreamPutUI(i_bs_ptr, !!(m_constraint_flag & 2), 1);                         /* constraint_set1_flag */
    BitStreamPutUI(i_bs_ptr, !!(m_constraint_flag & 4), 1);                         /* constraint_set2_flag */
    BitStreamPutUI(i_bs_ptr, !!(m_constraint_flag & 8), 1);                         /* constraint_set3_flag */
    BitStreamPutUI(i_bs_ptr, 0, 4);                         /* reserved_zero_4bits */
    BitStreamPutUI(i_bs_ptr, m_va_enc_seq_param.level_idc, 8);      /* level_idc */
    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.seq_parameter_set_id);      /* m_vaEncSeqParameter_set_id */

    if ( profile_idc == PROFILE_IDC_HIGH) 
    {
        BitStreamPutUE(i_bs_ptr, 1);        /* chroma_format_idc = 1, 4:2:0 */ 
        BitStreamPutUE(i_bs_ptr, 0);        /* bit_depth_luma_minus8 */
        BitStreamPutUE(i_bs_ptr, 0);        /* bit_depth_chroma_minus8 */
        BitStreamPutUI(i_bs_ptr, 0, 1);     /* qpprime_y_zero_transform_bypass_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1);     /* seq_scaling_matrix_present_flag */
    }

    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.seq_fields.bits.log2_max_frame_num_minus4); /* log2_max_frame_num_minus4 */
    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.seq_fields.bits.pic_order_cnt_type);        /* pic_order_cnt_type */

    if (m_va_enc_seq_param.seq_fields.bits.pic_order_cnt_type == 0)
        BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.seq_fields.bits.log2_max_pic_order_cnt_lsb_minus4);  
    else
        assert(0);
    
    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.max_num_ref_frames);        /* num_ref_frames */
    BitStreamPutUI(i_bs_ptr, 0, 1);                                 /* gaps_in_frame_num_value_allowed_flag */

    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.picture_width_in_mbs - 1);  /* pic_width_in_mbs_minus1 */
    BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.picture_height_in_mbs - 1); /* pic_height_in_map_units_minus1 */
    BitStreamPutUI(i_bs_ptr, m_va_enc_seq_param.seq_fields.bits.frame_mbs_only_flag, 1);    /* frame_mbs_only_flag */

    if (!m_va_enc_seq_param.seq_fields.bits.frame_mbs_only_flag) 
        assert(0);

    BitStreamPutUI(i_bs_ptr, m_va_enc_seq_param.seq_fields.bits.direct_8x8_inference_flag, 1);     
    BitStreamPutUI(i_bs_ptr, m_va_enc_seq_param.frame_cropping_flag, 1);            /* frame_cropping_flag */

    if (m_va_enc_seq_param.frame_cropping_flag) 
    {
        BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.frame_crop_left_offset);        /* frame_crop_left_offset */
        BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.frame_crop_right_offset);       /* frame_crop_right_offset */
        BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.frame_crop_top_offset);         /* frame_crop_top_offset */
        BitStreamPutUE(i_bs_ptr, m_va_enc_seq_param.frame_crop_bottom_offset);      /* frame_crop_bottom_offset */
    }

    //if ( frame_bit_rate < 0 ) { //TODO EW: the vui header isn't correct
    if ( 1 ) 
        BitStreamPutUI(i_bs_ptr, 0, 1); /* vui_parameters_present_flag */    
    else 
    {
        BitStreamPutUI(i_bs_ptr, 1, 1); /* vui_parameters_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1); /* aspect_ratio_info_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1); /* overscan_info_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1); /* video_sigNALype_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1); /* chroma_loc_info_present_flag */
        BitStreamPutUI(i_bs_ptr, 1, 1); /* timing_info_present_flag */
        {
            BitStreamPutUI(i_bs_ptr, 15, 32);
            BitStreamPutUI(i_bs_ptr, 900, 32);
            BitStreamPutUI(i_bs_ptr, 1, 1);
        }
        BitStreamPutUI(i_bs_ptr, 1, 1); /* nal_hrd_parameters_present_flag */
        {
            // hrd_parameters 
            BitStreamPutUE(i_bs_ptr, 0);    /* cpb_cnt_minus1 */
            BitStreamPutUI(i_bs_ptr, 4, 4); /* bit_rate_scale */
            BitStreamPutUI(i_bs_ptr, 6, 4); /* cpb_size_scale */

            BitStreamPutUE(i_bs_ptr, m_enc_info_ptr->bitrate - 1); /* bit_rate_value_minus1[0] */
            BitStreamPutUE(i_bs_ptr, m_enc_info_ptr->bitrate*8 - 1); /* cpb_size_value_minus1[0] */
            BitStreamPutUI(i_bs_ptr, 1, 1);  /* cbr_flag[0] */

            BitStreamPutUI(i_bs_ptr, 23, 5);   /* initial_cpb_removal_delay_length_minus1 */
            BitStreamPutUI(i_bs_ptr, 23, 5);   /* cpb_removal_delay_length_minus1 */
            BitStreamPutUI(i_bs_ptr, 23, 5);   /* dpb_output_delay_length_minus1 */
            BitStreamPutUI(i_bs_ptr, 23, 5);   /* time_offset_length  */
        }
        BitStreamPutUI(i_bs_ptr, 0, 1);   /* vcl_hrd_parameters_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1);   /* low_delay_hrd_flag */ 

        BitStreamPutUI(i_bs_ptr, 0, 1); /* pic_struct_present_flag */
        BitStreamPutUI(i_bs_ptr, 0, 1); /* bitstream_restriction_flag */
    }

    RBSPTrailingBits(i_bs_ptr);     /* RBSPTrailingBits */
}// NOLINT(readability/fn_size)

* 参  数：
* 返回值：0 成功/非0 失败

void  VAEncH264::PPS_Rbsp(BitStream *i_bs_ptr)
{
    BitStreamPutUE(i_bs_ptr, m_va_enc_pic_param.pic_parameter_set_id);      /* m_vaEncPicParameter_set_id */
    BitStreamPutUE(i_bs_ptr, m_va_enc_pic_param.seq_parameter_set_id);      /* m_vaEncSeqParameter_set_id */

    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.pic_fields.bits.entropy_coding_mode_flag, 1);  /* entropy_coding_mode_flag */

    BitStreamPutUI(i_bs_ptr, 0, 1);                         /* pic_order_present_flag: 0 */
    BitStreamPutUE(i_bs_ptr, 0);                            /* num_slice_groups_minus1 */

    BitStreamPutUE(i_bs_ptr, m_va_enc_pic_param.num_ref_idx_l0_active_minus1);      /* num_ref_idx_l0_active_minus1 */
    BitStreamPutUE(i_bs_ptr, m_va_enc_pic_param.num_ref_idx_l1_active_minus1);      /* num_ref_idx_l1_active_minus1 1 */

    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.pic_fields.bits.weighted_pred_flag, 1);     /* weighted_pred_flag: 0 */
    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.pic_fields.bits.weighted_bipred_idc, 2);   /* weighted_bipred_idc: 0 */

    BitStreamPutSE(i_bs_ptr, m_va_enc_pic_param.pic_init_qp - 26);  /* pic_init_qp_minus26 */
    BitStreamPutSE(i_bs_ptr, 0);                            /* pic_init_qs_minus26 */
    BitStreamPutSE(i_bs_ptr, 0);                            /* chroma_qp_index_offset */

    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.pic_fields.bits.deblocking_filter_control_present_flag, 1); 
    BitStreamPutUI(i_bs_ptr, 0, 1);                         /* constrained_intra_pred_flag */
    BitStreamPutUI(i_bs_ptr, 0, 1);                         /* redundant_pic_cnt_present_flag */

    /* MoreRBSPData */
    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.pic_fields.bits.transform_8x8_mode_flag, 1); 
    BitStreamPutUI(i_bs_ptr, 0, 1);                         /* pic_scaling_matrix_present_flag */
    BitStreamPutSE(i_bs_ptr, m_va_enc_pic_param.second_chroma_qp_index_offset );    /*second_chroma_qp_index_offset */

    RBSPTrailingBits(i_bs_ptr);
}

/*------------------------------------------------------------------------------
* 描  述：
* 参  数：
* 返回值：0 成功/非0 失败
-----------------------------------------------------------------------------*/
void VAEncH264::SliceHeader(BitStream *i_bs_ptr)
{
    int first_mb_in_slice = m_va_enc_slice_param.macroblock_address;

    BitStreamPutUE(i_bs_ptr, first_mb_in_slice);        /* first_mb_in_slice: 0 */
    BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.slice_type);   /* slice_type */
    BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.pic_parameter_set_id);        /* pic_parameter_set_id: 0 */
    BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.frame_num, m_va_enc_seq_param.seq_fields.bits.log2_max_frame_num_minus4 + 4);

    /* frame_mbs_only_flag == 1 */
    if (!m_va_enc_seq_param.seq_fields.bits.frame_mbs_only_flag)
        assert(0);
    
    if (m_va_enc_pic_param.pic_fields.bits.idr_pic_flag)
        BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.idr_pic_id);     /* idr_pic_id: 0 */

    if (m_va_enc_seq_param.seq_fields.bits.pic_order_cnt_type == 0) 
        BitStreamPutUI(i_bs_ptr, m_va_enc_pic_param.CurrPic.TopFieldOrderCnt, 
            m_va_enc_seq_param.seq_fields.bits.log2_max_pic_order_cnt_lsb_minus4 + 4);
   else 
       assert(0);

    /* redundant_pic_cnt_present_flag == 0 */
    /* slice type */
    if (IS_P_SLICE(m_va_enc_slice_param.slice_type)) 
    {
        BitStreamPutUI(i_bs_ptr, m_va_enc_slice_param.num_ref_idx_active_override_flag, 1);           

        if (m_va_enc_slice_param.num_ref_idx_active_override_flag)
            BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.num_ref_idx_l0_active_minus1);

        /* ref_pic_list_reordering */
        BitStreamPutUI(i_bs_ptr, 0, 1);            /* ref_pic_list_reordering_flag_l0: 0 */
    } else if (IS_B_SLICE(m_va_enc_slice_param.slice_type)) 
    {
        BitStreamPutUI(i_bs_ptr, m_va_enc_slice_param.direct_spatial_mv_pred_flag, 1);   
        BitStreamPutUI(i_bs_ptr, m_va_enc_slice_param.num_ref_idx_active_override_flag, 1);    

        if (m_va_enc_slice_param.num_ref_idx_active_override_flag) 
        {
            BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.num_ref_idx_l0_active_minus1);
            BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.num_ref_idx_l1_active_minus1);
        }

        /* ref_pic_list_reordering */
        BitStreamPutUI(i_bs_ptr, 0, 1);            /* ref_pic_list_reordering_flag_l0: 0 */
        BitStreamPutUI(i_bs_ptr, 0, 1);            /* ref_pic_list_reordering_flag_l1: 0 */
    }

    if ((m_va_enc_pic_param.pic_fields.bits.weighted_pred_flag && IS_P_SLICE(m_va_enc_slice_param.slice_type)) ||
        ((m_va_enc_pic_param.pic_fields.bits.weighted_bipred_idc == 1) && IS_B_SLICE(m_va_enc_slice_param.slice_type))) 
            assert(0); /* FIXME: fill weight/offset table */

    /* dec_ref_pic_marking */
    if (m_va_enc_pic_param.pic_fields.bits.reference_pic_flag) {     /* nal_ref_idc != 0 */
        uint8_t no_output_of_prior_pics_flag = 0;
        uint8_t long_term_reference_flag = 0;
        uint8_t adaptive_ref_pic_marking_mode_flag = 0;

        if (m_va_enc_pic_param.pic_fields.bits.idr_pic_flag) {
            BitStreamPutUI(i_bs_ptr, no_output_of_prior_pics_flag, 1);           
            BitStreamPutUI(i_bs_ptr, long_term_reference_flag, 1);       
        } else 
            BitStreamPutUI(i_bs_ptr, adaptive_ref_pic_marking_mode_flag, 1);        
    }

    if (m_va_enc_pic_param.pic_fields.bits.entropy_coding_mode_flag &&
        !IS_I_SLICE(m_va_enc_slice_param.slice_type))
        BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.cabac_init_idc);               /* cabac_init_idc: 0 */

    BitStreamPutSE(i_bs_ptr, m_va_enc_slice_param.slice_qp_delta);                   /* slice_qp_delta: 0 */

    if (m_va_enc_pic_param.pic_fields.bits.deblocking_filter_control_present_flag) 
    {
        BitStreamPutUE(i_bs_ptr, m_va_enc_slice_param.disable_deblocking_filter_idc);     

        if (m_va_enc_slice_param.disable_deblocking_filter_idc != 1) 
        {
            BitStreamPutSE(i_bs_ptr, m_va_enc_slice_param.slice_alpha_c0_offset_div2);     
            BitStreamPutSE(i_bs_ptr, m_va_enc_slice_param.slice_beta_offset_div2);              
        }
    }

    if (m_va_enc_pic_param.pic_fields.bits.entropy_coding_mode_flag) 
        BitStreamByteAligning(i_bs_ptr, 1);    
}
