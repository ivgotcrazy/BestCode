typedef struct { //NOLINT
	int		nCodecID;
	WCHAR   wszCodecName[64];
}VideoCodecInfo;

const VideoCodecInfo g_VideoCodeArray[] =
{
	{ VIDEO_CODEC_DIVX, L"Mpeg4 Codec 1" },
	{ VIDEO_CODEC_WMV9,L"Mpeg4 Codec 2"},
	{ VIDEO_CODEC_H264,L"H264 Codec"},
//	{ VIDEO_CODEC_H264PLUS,L"H264 Codec Plus"},
	{ VIDEO_CODEC_H264PLUS,L"TIVC Codec"},
	{ VIDEO_CODEC_XVID,L"XVID Codec"},
	{ VIDEO_CODEC_VP8,L"VP8 Codec"},
	{ VIDEO_CODEC_VP9,L"VP9 Codec"}
};

//---------------------
//获取编码器个数
//---------------------
int 	VIDEO_Codec_GetCount()
{
	return sizeof(g_VideoCodeArray)/sizeof(VideoCodecInfo);
}

//---------------------
//获取编码器名称
//---------------------
FS_UINT    VIDEO_Codec_GetInfo(int nIndex, WCHAR *wszName, FS_UINT nSize, int &nCodecID) //NOLINT
{
	if( nIndex >= VIDEO_Codec_GetCount() )
		return 0;
	
	if( nSize <= wcslen(g_VideoCodeArray[nIndex].wszCodecName))
		return 0;
	
	wcscpy( wszName,g_VideoCodeArray[nIndex].wszCodecName );
	nCodecID = g_VideoCodeArray[nIndex].nCodecID;
	
	return wcslen( wszName );	
}

//---------------------
//创建压缩器
//EncodeType: 压缩器类型
//lpbiIn:压缩前位图格式
//Return:压缩器句柄
//-------------------
HANDLE	VIDEO_Encode_StartCompress( const Video_Encoder_Param &param,const BITMAPINFOHEADER &biIn )
{
	CVideoEncoder *pEncoder = NULL;
	switch( param.nEncoderID ) 
	{
#if USE_DIVXCODEC
	case VIDEO_CODEC_DIVX:
		pEncoder = new CVideoEncoderDivX;
		break;
#endif
#if USE_MSWMV9ENC
	case VIDEO_CODEC_WMV9:
		pEncoder = new CVideoEncoderWMV9;
		break;
#endif
#if USE_XVIDCODEC
	case VIDEO_CODEC_XVID:
		pEncoder = new CVideoEncoderXvid;
		break;
#endif		
#if USE_VPXVP8ENC
	case VIDEO_CODEC_VP8:
		pEncoder = new CVideoEncoderVP8;
		break;
#endif
#if USE_VPXVP9CODEC
	case VIDEO_CODEC_VP9:
		pEncoder = new CVideoEncoderVP9;
		break;
#endif
    case VIDEO_CODEC_H264:
        {
#ifdef _FS_OS_WIN
			if(g_IntelCodecConfig.m_bUsable && g_hwaccEncodeControl.IncreaseCurHWAccEncoder())
			{
				LOG("INF:VIDEO_Encode_StartCompress intel hardware encoder create!\n");
				pEncoder = new CVideoEncoderIntelH264;
			}
#endif
#if USE_H264IOSHDENC
            extern double GetIOSSystemVersion();
            extern bool IsSupportH264HDEnc( int nWidth,int nHeight );
            if( IsSupportH264HDEnc( biIn.biWidth,biIn.biHeight ) ){
                pEncoder = new CVideoEncoderH264IOSHD;
            }
#endif
#ifdef ANDROID
			//pEncoder = new CVideoEncoderH264AndroidHD();
#endif
#if USE_X264ENC
            if( NULL == pEncoder ){
                pEncoder = new CVideoEncoderX264;
            }
#endif
        }
        break;
	}
	if( !pEncoder ){
		return NULL;
	}

	if( !pEncoder->StartCompress( biIn,param )){
	
		delete pEncoder;
		pEncoder = NULL;
#if defined USE_X264ENC && ( USE_H264IOSHDENC || defined ANDROID )
		if( param.nEncoderID == VIDEO_CODEC_H264 ){
            pEncoder = new CVideoEncoderX264;
			if( pEncoder && !pEncoder->StartCompress( biIn,param ) ){
				delete pEncoder;
				pEncoder = NULL;
			}
		}
#endif
	}

	return (HANDLE)pEncoder;
} //NOLINT


//---------------------
//压缩数据
//frame：压缩数据包
//Return:返回是否成功
//----------------------
BOOL	VIDEO_Encode_Compress(HANDLE hEncoder, Video_Code_Frame &frame) //NOLINT
{
	if( !hEncoder )
		return FALSE;

	CVideoEncoder *pEncoder = (CVideoEncoder*)hEncoder;
	return pEncoder->Compress( frame );
}

//--------------------
//停止压缩器
//hEncoder: 压缩器句柄
//--------------------
VOID	VIDEO_Encode_StopCompress(HANDLE &hEncoder) //NOLINT
{
	if( hEncoder ){
		CVideoEncoder *pEncoder = (CVideoEncoder*)hEncoder;
		pEncoder->StopCompress();
		if(pEncoder->IsHWAccUsed())
		{
			g_hwaccEncodeControl.DecreaseCurHWAccEncoder();
		}
		delete pEncoder;
		hEncoder = NULL;
	}
}

VOID	VIDEO_Encode_SetMaxHWAccEncoder(DWORD nMax)
{
	g_hwaccEncodeControl.SetMaxHWAccEncoder(nMax);
}

//---------------------
//创建解压器
//CodeType:解压器句柄
//biOut:解压之后格式
//Return: 解压器句柄
//---------------------
HANDLE	VIDEO_Decode_StartDecompress( int nCodeID,const BITMAPINFOHEADER &biOut )
{
	CVideoDecoder *pDecoder = NULL;
	LOG("INF:VIDEO_Decode_StartDecompress CodeID[%d].\n", nCodeID);

	switch( nCodeID ) 
	{
#if USE_FFWMV9DEC
	case VIDEO_CODEC_WMV9:
	    LOG(_T("INF:VIDEO_Decode_StartDecompress create CVideoDecoderFFWMV9!\n"));
		pDecoder = new CVideoDecoderFFWMV9;
		break;
#endif
#if USE_DIVXCODEC
	case VIDEO_CODEC_DIVX:
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderDivX!\n");
		pDecoder = new CVideoDecoderDivX;
		break;
#endif
#if USE_MSWMV9DEC
	case VIDEO_CODEC_WMV9:
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderWMV9!\n");
		pDecoder = new CVideoDecoderWMV9;
		break;
#endif
#if USE_XVIDCODEC
	case VIDEO_CODEC_XVID:
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderXvid!\n");
		pDecoder = new CVideoDecoderXvid;
		break;
#endif
#if USE_VPXVP8DEC//@@@fixme 这里怎么没打开，导致看android黑屏
	case VIDEO_CODEC_VP8:
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderVP8!\n");
		pDecoder = new CVideoDecoderVP8;
		break;
#endif
#if USE_VPXVP9CODEC
	case VIDEO_CODEC_VP9:
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderVP9!\n");
		pDecoder = new CVideoDecoderVP9;
		break;
#endif
#if USE_FFVP8DEC
    case VIDEO_CODEC_VP8:
        LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderFFVP8!\n");
        pDecoder = new CVideoDecoderFFVP8;
        break;
#endif
#if USE_FFH264DEC
	case VIDEO_CODEC_H264:
	case VIDEO_CODEC_H264PLUS:	    
#if USE_H264ANDROIDHDDEC
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderH264Android!\n");
	    pDecoder = new CVideoDecoderH264Android;
#else
	    LOG("INF:VIDEO_Decode_StartDecompress create CVideoDecoderFFH264!\n");
		pDecoder = new CVideoDecoderFFH264;
#endif
		break;
#endif
	}

	if( !pDecoder )
	{
    	LOG("ERR:VIDEO_Decode_StartDecompress CodeID[%d] is not support!\n", nCodeID);
		return NULL;
	}

	if( !pDecoder->StartDecompress( biOut ))
	{
	    LOG("ERR:VIDEO_Decode_StartDecompress StartDecompress[%d] Fail!\n", nCodeID);
		delete pDecoder;
		return NULL;
	}
	return (HANDLE)pDecoder;
} //NOLINT


//-------------------
//解压图象
//hDecoder:解压器句柄
//frame:解压包
//Return: 是否解压成功
//--------------------
BOOL	VIDEO_Decode_Decompress(HANDLE hDecoder, Video_Code_Frame &frame) //NOLINT
{
	if( !hDecoder )
		return FALSE;

	CVideoDecoder *pDecoder = (CVideoDecoder*)hDecoder;
	return pDecoder->Decompress( frame );
}

//---------------------
//停止压缩器
//hDecoder:解压器句柄
//--------------------
VOID	VIDEO_Decode_StopDecompress(HANDLE &hDecoder) //NOLINT
{
	if( hDecoder ){
		CVideoDecoder *pDecoder = (CVideoDecoder *)hDecoder;
		pDecoder->StopDecompress();
		delete pDecoder;
		hDecoder = NULL;
	}
