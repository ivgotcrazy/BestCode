BOOL VideoCodecGetCurrentDllPath(HMODULE hInstance, TCHAR* path)
{
#ifdef _FS_OS_WIN
	if(!GetModuleFileName(hInstance, path, MAX_PATH))
	{
		LOG("GetModuleFileName failed, error[%d]", GetLastError());
		return FALSE;
	}
	TCHAR *szFind = _tcsrchr(path, '\\');
	if (!szFind)
		return FALSE;
	szFind[1] = 0;
#elif defined _FS_OS_ANDROID
	Dl_info dlinfo;
	int rc = dladdr((void*)VideoCodecGetCurrentDllPath, &dlinfo);
	if (!rc) 
	{
		LOG("Problem retrieving address information for %x:  %s\n", VideoCodecGetCurrentDllPath, dlerror());
		return FALSE;
	}
	_tcscpy(path, dlinfo.dli_fname);
	TCHAR *szFind = _tcsrchr(path, '/');
	if (!szFind)
	{
	    path[0] = '\0';
	    return TRUE;
	}
	szFind[1] = 0;
#endif
	return TRUE;
}



CVideoCodecLoader::CVideoCodecLoader()
{	
	ZeroMemory(m_dllPath, sizeof(m_dllPath));

	//x265 encoder
	ZeroMemory(&m_x265EncDll, sizeof(m_x265EncDll));

	//x264 encoder
	ZeroMemory(&m_x264EncDll, sizeof(m_x264EncDll));

	//intel media sdk encoder
	ZeroMemory(&m_intelH264EncDll, sizeof(m_intelH264EncDll));
	ZeroMemory(&m_intelH265EncDll, sizeof(m_intelH265EncDll));

	//intel media sdk decoder
	ZeroMemory(&m_intelH264DecDll, sizeof(m_intelH264DecDll));
	ZeroMemory(&m_intelH265DecDll, sizeof(m_intelH265DecDll));

	//ffmpeg software decoder(h264,h265,wmv9,vp8/9)
	//HW mediacodec encoder for android platform
	ZeroMemory(&m_ffmpegH264DecDll, sizeof(m_ffmpegH264DecDll));
	ZeroMemory(&m_ffmpegH265DecDll, sizeof(m_ffmpegH265DecDll));
	ZeroMemory(&m_ffmpegWMV9DecDll, sizeof(m_ffmpegWMV9DecDll));
	ZeroMemory(&m_ffmpegVP8DecDll, sizeof(m_ffmpegVP8DecDll));
	ZeroMemory(&m_ffmpegVP9DecDll, sizeof(m_ffmpegVP9DecDll));
	ZeroMemory(&m_ffmpegMCH264EncDll, sizeof(m_ffmpegMCH264EncDll));
	ZeroMemory(&m_ffmpegMCH265EncDll, sizeof(m_ffmpegMCH265EncDll));

	//external video codecs which are all encapsulated in VCodeWrapper.dll
	//Codecs including: wmv9dmo(wmv9 dec/enc), mpeg4plus(xvid dec/encoder)
	//Also includes native media codec decoder wrapper for Android platform
	ZeroMemory(&m_vWrapperWMV9DecDll, sizeof(m_vWrapperWMV9DecDll));
	ZeroMemory(&m_vWrapperWMV9EncDll, sizeof(m_vWrapperWMV9EncDll));
	ZeroMemory(&m_vWrapperXvidDecDll, sizeof(m_vWrapperXvidDecDll));
	ZeroMemory(&m_vWrapperXvidEncDll, sizeof(m_vWrapperXvidEncDll));
	ZeroMemory(&m_vWrapperNMCH264DecDll, sizeof(m_vWrapperNMCH264DecDll));
	ZeroMemory(&m_vWrapperNMCH265DecDll, sizeof(m_vWrapperNMCH265DecDll));

	//external vp8/9 decoder/encoder
	ZeroMemory(&m_vpCodecVP8DecDll, sizeof(m_vpCodecVP8DecDll));
	ZeroMemory(&m_vpCodecVP9DecDll, sizeof(m_vpCodecVP9DecDll));
	ZeroMemory(&m_vpCodecVP8EncDll, sizeof(m_vpCodecVP8EncDll));
	ZeroMemory(&m_vpCodecVP9EncDll, sizeof(m_vpCodecVP9EncDll));


	m_bInited = 0;
}

CVideoCodecLoader::~CVideoCodecLoader()
{
	Release();
}

VOID CVideoCodecLoader::Release()
{
	//X265 encoder
	m_x265Enc.FreeLibrary();

	//X264 encoder
	m_x264EncLib.FreeLibrary();

	//Intel HW decoder & encoder
	m_intelEncLib.FreeLibrary();
	m_intelDecLib.FreeLibrary();

	//FFmpeg decoder & encoder
	m_ffmpegDecLib.FreeLibrary();

	m_vWrapperLib.FreeLibrary();
	m_vpCodecLib.FreeLibrary();



}


BOOL CVideoCodecLoader::LoadX265Encoder()
{
#ifdef _WIN32
	//check OS: X265 encoder doesn't support winxp. 
	//Performance will be slightly decreased when enabled with winxp. See x265 document.
	if (!CheckOSCompatibility())
		return FALSE;

#endif
	if (m_x265EncDll.m_bInited)
		return TRUE;

#if USE_STATICCOMPONENT
	//used by IOS and MAC
	LoadVideoCodecEx funLoad = FMVCodecEx_H265ENC;
#else
	TCHAR szDllPath[MAX_PATH];

#ifdef _WIN32
	_stprintf(szDllPath, _T("%s%s"), m_dllPath, _T("H265EncPlus.dll"));
#elif defined _FS_OS_ANDROID
	_stprintf(szDllPath, _T("%s%s"), m_dllPath, _T("libh265enc.so"));
#else
	_tcscpy(szDllPath, _T("libh265enc.so"));
#endif	
	if (!m_x265Enc.LoadLibrary(szDllPath)){
		LOG("Load X265 encoder library %s failed.\n", szDllPath);
		return FALSE;
	}

	if (!m_x265EncDll.m_bInited){

BOOL CVideoCodecLoader::LoadFFmpegCodec()
{
	if (m_ffmpegH264DecDll.m_bInited)
		return TRUE;

	TCHAR szDllPath[MAX_PATH];
#if !USE_STATICCOMPONENT
#ifdef _WIN32
	_stprintf(szDllPath, _T("%s%s"), m_dllPath, _T("H264DecPlus.dll"));
#elif defined _FS_OS_ANDROID
	_stprintf(szDllPath, _T("%s%s"), m_dllPath, _T("libh264dec.so"));
#else
	_tcscpy(szDllPath, _T("libh264dec.so"));
#endif	
	if (!m_ffmpegDecLib.LoadLibrary(szDllPath)){
		LOG("CVideoCodecLoader: Load ffmpeg codec library %s failed.\n", szDllPath);
		return FALSE;
	}
#endif
    
	if (!m_ffmpegH264DecDll.m_bInited){
#if USE_STATICCOMPONENT
        LoadVideoCodecEx funLoad = FMVCodecEx_FFMPEGDEC;
#else
		LoadVideoCodecEx funLoad = (LoadVideoCodecEx)m_ffmpegDecLib.GetFunction(FMVCODEC_FUNNAMEEX);;
		if (NULL == funLoad)
			return FALSE;
#endif
		funLoad(&m_ffmpegH264DecDll);
	}

	if (NULL == m_ffmpegH264DecDll.DllDecOpen)
		return FALSE;
#if USE_STATICCOMPONENT
    GetVideoCodecCountEx funGetCount = FMVCodecGetCountEx_FFMPEGDEC;
    GetVideoCodecInfoEx funGetInfo = FMVCodecGetInfoEx_FFMPEGDEC;
#else
    GetVideoCodecCountEx funGetCount = (GetVideoCodecCountEx)m_ffmpegDecLib.GetFunction(FMVCODECGETCOUNT_FUNNAMEEX);
	GetVideoCodecInfoEx funGetInfo = (GetVideoCodecInfoEx)m_ffmpegDecLib.GetFunction(FMVCODECGETINFO_FUNNAMEEX);
#endif

		int nCount = 0;
		funGetCount(&nCount);
		for (int i = 0; i<nCount; i++){

			VideoCodecDllEx dll;
			if (funGetInfo(i, &dll) == S_OK){

				VideoCodecPluginInfoEx info;
				dll.DllGetInfo(&info);
				if (info.nCodecID == VIDEO_CODEC_H264 && info.bSupportDec){

					CopyMemory(&m_ffmpegH265DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg H265 decoder Success.");

				if (info.nCodecID == VIDEO_CODEC_WMV9) {
					CopyMemory(&m_ffmpegWMV9DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg wmv9 decoder Success.");
				}

				if (info.nCodecID == VIDEO_CODEC_VP8) {
					CopyMemory(&m_ffmpegVP8DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg VP8 decoder success.");
				}

				if (info.nCodecID == VIDEO_CODEC_VP9) {
					CopyMemory(&m_ffmpegVP9DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg VP9 decoder success.");
				}
#ifdef _FS_OS_ANDROID
				if (info.nCodecID == VIDEO_CODEC_H264 && info.bSupportEnc){
					CopyMemory(&m_ffmpegMCH264EncDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg H264 MC Encoder Success.");
				}
				if (info.nCodecID == VIDEO_CODEC_H265 && info.bSupportEnc){
					CopyMemory(&m_ffmpegMCH265EncDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load ffmpeg H265 MC Encoder Success.");
				}

#endif


			}
		}
	}

	if (m_ffmpegWMV9DecDll.DllDecOpen) {
		m_ffmpegWMV9DecDll.m_bInited = TRUE;
	}
	if (m_ffmpegVP8DecDll.DllDecOpen) {
		m_ffmpegVP8DecDll.m_bInited = TRUE;
	}
	if (m_ffmpegVP9DecDll.DllDecOpen) {

	if (m_ffmpegH264DecDll.DllDecOpen) {

					CopyMemory(&m_vWrapperNMCH264DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load vcodec wrapper Native MeidaCodec H.264 Decoder Success.");
#ifdef _FS_OS_ANDROID
					ALOG("CVideoCodecLoader: Load vcodec wrapper Native MeidaCodec H.264 Decoder Success.");
#endif

				if(info.nCodecID == VIDEO_CODEC_H265) {
					CopyMemory(&m_vWrapperNMCH265DecDll, &dll, sizeof(VideoCodecDllEx));
					LOG("CVideoCodecLoader: Load vcodec wrapper Native MeidaCodec H.265 Decoder Success.");
#ifdef _FS_OS_ANDROID
					ALOG("CVideoCodecLoader: Load vcodec wrapper Native MeidaCodec H.265 Decoder Success.");
#endif
				}
			}
		}
	}
	if (m_vWrapperWMV9EncDll.DllEncOpen)

	if (m_vWrapperWMV9DecDll.DllDecOpen)

//check whether specified codec has HW accelerated ability or not
BOOL CVideoCodecLoader::HasHWAcc(int nCodecId, BOOL isEncoder)
{
	int i;
	if (!m_bInited)
		return FALSE;

	for (i = 0; i < m_nCodecCount; i++) {
		if (m_pVCodecHub[i].isEncoder == isEncoder && m_pVCodecHub[i].nCodecId == nCodecId &&
			m_pVCodecHub[i].isHWAccel)
		{
			return TRUE;
		}
	}

	return FALSE;

}
/*
* Find codec according to codecid + decoder/encoder indicator + hw enable indicator
*
*/
VideoCodecDllEx* CVideoCodecLoader::FindCodecById(int nCodecId, BOOL isEncoder, BOOL isHWAcc)
{
	int i;
	int score = 0;
	int index = -1;


	for (i = 0; i < m_nCodecCount; i++) {
		if (m_pVCodecHub[i].isEncoder == isEncoder && m_pVCodecHub[i].nCodecId == nCodecId &&
			m_pVCodecHub[i].isHWAccel == isHWAcc)
		{
			if (m_pVCodecHub[i].nCodecScore > score) {
				score = m_pVCodecHub[i].nCodecScore;
				index = i;
			}		

		}			
	}
	if (index >= 0)
	{
		LOG("cszShortName:%s, isEncoder:%d, isHWAccel:%d\n", m_pVCodecHub[index].cszShortName, m_pVCodecHub[index].isEncoder,
			m_pVCodecHub[index].isHWAccel);
		return m_pVCodecHub[index].pCodecDll;
	}

	return NULL;
}

BOOL CVideoCodecLoader::CheckOSCompatibility()
{
#ifdef _WIN32

	OSVERSIONINFOEX osvi;

	ZeroMemory(&osvi, sizeof(OSVERSIONINFOEX));
	osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFOEX);
	if (!GetVersionEx((OSVERSIONINFO *)&osvi))
	{
		osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
		if (!GetVersionEx((OSVERSIONINFO *)&osvi))
			return FALSE;
	}
	if (osvi.dwPlatformId == VER_PLATFORM_WIN32_NT && osvi.dwMajorVersion >= 6)
		return TRUE;
	return FALSE;
#else
	return TRUE;
#endif

}
