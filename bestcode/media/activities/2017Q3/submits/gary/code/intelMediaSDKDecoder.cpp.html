void   STDAPICALLTYPE	_DllGetInfoH264(VideoCodecPluginInfoEx *pInfo)
{
	if (pInfo){
		ZeroMemory(pInfo, sizeof(VideoCodecPluginInfoEx));
		pInfo->dwVersion = 0x01; //unused
		pInfo->bSupportDec = TRUE;
		pInfo->bSupportEnc = FALSE;
		pInfo->bIsHWAccel = TRUE;
		pInfo->nCodecID = VIDEO_CODEC_H264;
		pInfo->bIsRenerBuiltIn = TRUE;
		pInfo->nModuleID = MODULE_ID_INTELHW;
		wcscpy(pInfo->wszName, L"Intel H264 Decoder");
	}
}

HANDLE  STDAPICALLTYPE _DllDecOpenH264(HWND hWnd, PBYTE data, int size)
{
	if (!data)
		return NULL;
	CDecodingPipeline* pDec = new CDecodingPipeline;
	if (!pDec)
		return NULL;
	if (MFX_ERR_NONE != pDec->Init(hWnd, data, size, MFX_CODEC_AVC))
	{
		pDec->Close();
		return NULL;
	}
	return static_cast<void*>(pDec);
}

VOID	STDAPICALLTYPE _DllDecClose(HANDLE hDecoder)
{
	if (!hDecoder)
		return;
	CDecodingPipeline* pDec = static_cast<CDecodingPipeline*>(hDecoder);
	pDec->Close();
	delete pDec;
}


VOID STDAPICALLTYPE _DllEnableRender(HANDLE hDecoder, BOOL bRender)
{
	if (!hDecoder)
		return;
	CDecodingPipeline* pDec = static_cast<CDecodingPipeline*>(hDecoder);
	pDec->SetIsRender(bRender);
}

int  STDAPICALLTYPE _DllDecodeAndRender(HANDLE hDecoder, PBYTE data, int size, PBYTE pDecData, unsigned int* pDecSize)
{
	if (!hDecoder || !data)
		return -1;
	CDecodingPipeline* pDec = static_cast<CDecodingPipeline*>(hDecoder);
	unsigned decsize;
	if (MFX_ERR_NONE != pDec->DecodeAndRender(data, size, pDecData, decsize))
		return -1;
	if (pDecSize)
		*pDecSize = decsize;
	return 0;
}

int		g_nCodecCount = sizeof(g_VideoCodec) / sizeof(VideoCodecDllEx);

#define LOG		wprintf

STDAPI	FMVCodecEx(VideoCodecDllEx *pDll)
{
	if (!pDll)
		return E_INVALIDARG;
	ZeroMemory(pDll, sizeof(*pDll));

	mfxStatus sts;
	MFXVideoSession mfxSession;
	mfxInitParam initPar;
	mfxVersion version;     // real API version with which library is initialized
	MSDK_ZERO_MEMORY(initPar);

	// we set version to 1.0 and later we will query actual version of the library which will got leaded
	initPar.Version.Major = 1;
	initPar.Version.Minor = 0;

	initPar.GPUCopy = MFX_GPUCOPY_ON;
	// try searching on all display adapters
	initPar.Implementation = MFX_IMPL_HARDWARE_ANY;

	sts = mfxSession.InitEx(initPar);

	// MSDK API version may not support multiple adapters - then try initialize on the default
	if (MFX_ERR_NONE != sts) {
		initPar.Implementation = (initPar.Implementation & !MFX_IMPL_HARDWARE_ANY) | MFX_IMPL_HARDWARE;
		sts = mfxSession.InitEx(initPar);
	}

	if (sts < MFX_ERR_NONE)
	{
		LOG(_T("Hardware accelerator is not usable.\n"));
		return E_FAIL;
	}

	LOG(_T("We'll be able to Intel Hardware Accelerator.\n"));
	CopyMemory(pDll, &g_VideoCodec[0], sizeof(VideoCodecDllEx));
	mfxSession.Close();
	return S_OK;
}

STDAPI	FMVCodecGetCountEx(int* pCount)
{
	if (pCount)
		*pCount = g_nCodecCount;
	return S_OK;
}

STDAPI	FMVCodecGetInfoEx(int nIndex, VideoCodecDllEx* pDll)
{
	if (nIndex < 0 || nIndex >= g_nCodecCount || NULL == pDll)
		return E_INVALIDARG;
	CopyMemory(pDll, &g_VideoCodec[nIndex], sizeof(VideoCodecDllEx));
	return S_OK;
}
