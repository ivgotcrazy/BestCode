#ifndef __VIDEO_CODEC_LOADER_H
#define __VIDEO_CODEC_LOADER_H

#if _MSC_VER > 1000
#pragma once
#endif // _MSC_VER > 1000

#include "VideoCodecPlugin.h"
#include "wdynamiclibrary.h"

#define MAX_SUPPORTED_CODEC_NUM 32

namespace WVideo{ //NOLINT

	class CVideoCodecLoader
	{
	public:
		CVideoCodecLoader();
		virtual ~CVideoCodecLoader();
	public:
		BOOL Init(HMODULE hInstance);
		VOID Release();
		BOOL IsInited();
		BOOL HasHWAcc(int nCodecId, BOOL isEncoder); //whether supported codecs have HW acceleration capablity
		VideoCodecDllEx * FindCodecById(int nCodecId, BOOL isEncoder, BOOL isHWAcc);
	private:
		//check OS version
		BOOL CheckOSCompatibility();

		BOOL LoadX265Encoder();
		WDynamicLibrary	m_x265Enc;
		VideoCodecDllEx	m_x265EncDll;

		BOOL LoadX264Encoder();
		WDynamicLibrary m_x264EncLib;
		VideoCodecDllEx m_x264EncDll;

		BOOL LoadIntelEncoder();
		WDynamicLibrary m_intelEncLib;
		VideoCodecDllEx m_intelH264EncDll;
		VideoCodecDllEx m_intelH265EncDll;

		BOOL LoadIntelDecoder();
		WDynamicLibrary m_intelDecLib;
		VideoCodecDllEx m_intelH264DecDll;
		VideoCodecDllEx m_intelH265DecDll;

		BOOL LoadFFmpegCodec();
		WDynamicLibrary m_ffmpegDecLib;
		VideoCodecDllEx m_ffmpegH264DecDll;
		VideoCodecDllEx m_ffmpegH265DecDll;
		VideoCodecDllEx m_ffmpegWMV9DecDll;
		VideoCodecDllEx m_ffmpegVP8DecDll;
		VideoCodecDllEx m_ffmpegVP9DecDll;
		VideoCodecDllEx m_ffmpegMCH264EncDll; //only supported under android
		VideoCodecDllEx m_ffmpegMCH265EncDll;

		BOOL LoadVCodecWrapper();
		WDynamicLibrary m_vWrapperLib;
		VideoCodecDllEx m_vWrapperWMV9EncDll;
		VideoCodecDllEx m_vWrapperWMV9DecDll;
		VideoCodecDllEx m_vWrapperXvidEncDll;
		VideoCodecDllEx m_vWrapperXvidDecDll;
		VideoCodecDllEx m_vWrapperNMCH264DecDll;
		VideoCodecDllEx m_vWrapperNMCH265DecDll;

		BOOL LoadVPCodec();
		WDynamicLibrary m_vpCodecLib;
		VideoCodecDllEx m_vpCodecVP8DecDll;
		VideoCodecDllEx m_vpCodecVP9DecDll;
		VideoCodecDllEx m_vpCodecVP8EncDll;
		VideoCodecDllEx m_vpCodecVP9EncDll;

		// newly add video codec dll here
	private:
		HMODULE m_hModule;
		TCHAR m_dllPath[MAX_PATH];
		//video codec center
		VideoCodecHub m_pVCodecHub[MAX_SUPPORTED_CODEC_NUM];
		FS_UINT32 m_nCodecCount;

		BOOL m_bInited;
	};

}
