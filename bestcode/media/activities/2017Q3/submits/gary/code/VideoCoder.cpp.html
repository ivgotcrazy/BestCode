CVideoCodecLoader g_videoCodecLoader;

typedef struct { //NOLINT
	int		nCodecID;
	WCHAR   wszCodecName[64];
}VideoCodecInfo;

const VideoCodecInfo g_VideoCodeArray[] =
{
	{ VIDEO_CODEC_DIVX, L"Mpeg4 Codec 1" },
	{ VIDEO_CODEC_WMV9,L"Mpeg4 Codec 2"},
	{ VIDEO_CODEC_H264,L"H264 Codec"},
//	{ VIDEO_CODEC_H264PLUS,L"H264 Codec Plus"},
	{ VIDEO_CODEC_H264PLUS,L"TIVC Codec"},
	{ VIDEO_CODEC_XVID,L"XVID Codec"},
	{ VIDEO_CODEC_VP8,L"VP8 Codec"},
	{ VIDEO_CODEC_VP9,L"VP9 Codec"},
	{VIDEO_CODEC_H265, L"H265 Codec"},
};

//---------------------
//获取编码器个数
//---------------------
int 	VIDEO_Codec_GetCount()
{
	return sizeof(g_VideoCodeArray)/sizeof(VideoCodecInfo);
}

//---------------------
//获取编码器名称
//---------------------
FS_UINT    VIDEO_Codec_GetInfo(int nIndex, WCHAR *wszName, FS_UINT nSize, int &nCodecID) //NOLINT
{
	if( nIndex >= VIDEO_Codec_GetCount() )
		return 0;
	
	if( nSize <= wcslen(g_VideoCodeArray[nIndex].wszCodecName))
		return 0;
	
	wcscpy( wszName,g_VideoCodeArray[nIndex].wszCodecName );
	nCodecID = g_VideoCodeArray[nIndex].nCodecID;
	
	return wcslen( wszName );	
}

//---------------------
//创建压缩器
//EncodeType: 压缩器类型
//lpbiIn:压缩前位图格式
//Return:压缩器句柄
//-------------------
VCodecHandle	VIDEO_Encode_StartCompress( const Video_Encoder_Param &param,const BITMAPINFOHEADER &biIn )
{
	VideoCodecDllEx *pCodecDll;
	HANDLE hCodec;
	VCodecHandle vHandle;
	BOOL bHasHW = FALSE;
	BOOL isHW = FALSE;
	BOOL bEnableH265 = FALSE;
	vHandle.pCodec = NULL;
	vHandle.pCodecDll = NULL;

	int nCodecId =  param.nEncoderID;
    //if (param.nEncoderID == VIDEO_CODEC_H264 || param.nEncoderID == VIDEO_CODEC_H264PLUS)
    //    nCodecId = VIDEO_CODEC_H265;
	if (!g_videoCodecLoader.IsInited())
		return vHandle;

	if (g_pVideoConfigCenter)
	{
		if (g_pVideoConfigCenter->GetBool("avcore.codec.video.h265", &bEnableH265)){
			LOG("INF:VIDEO_Encode_StartCompress: enable H.265 feature status [%d]!\n", bEnableH265);
		}
		else {
			bEnableH265 = FALSE;
			LOG("INF:VIDEO_Encode_StartCompress: disable H.265 feature!\n");
		}
	}

	//if H.265 is disabled in config center, fall back to H.264
	if (!bEnableH265 && nCodecId == VIDEO_CODEC_H265) {
		nCodecId = VIDEO_CODEC_H264;
	}

	//check HW acc control
	bHasHW = g_videoCodecLoader.HasHWAcc(nCodecId, TRUE);
	if (bHasHW && g_hwAccController.IncreaseCurHWAccEncoder()) {
		isHW = TRUE;
	}

	pCodecDll = g_videoCodecLoader.FindCodecById(nCodecId, 1, isHW);
	if (!pCodecDll)
		return vHandle;

	VideoCodecPluginInfoEx codecInfo;
	pCodecDll->DllGetInfo(&codecInfo);

#ifdef _FS_OS_ANDROID
	//Some HW accelerated codecs need specific parameter be configured
	// before opening it.
	if (codecInfo.nModuleID == MODULE_ID_FFMPEG &&
			codecInfo.nCodecID == VIDEO_CODEC_H264){
		//codec is MediaCodec HWAcc encoder
		pCodecDll->DllEncConfig(NULL, ENC_CONFIG_TYPE_JVM, (void*)g_hVideoModule);
	}
#endif

	hCodec = pCodecDll->DllEncOpen(&param, &biIn);
	if (!hCodec)
		return vHandle;

	vHandle.pCodec = hCodec;
	vHandle.pCodecDll = (HANDLE)pCodecDll;

	return vHandle;
	
}

//---------------------
//压缩数据
//frame：压缩数据包
//Return:返回是否成功
//----------------------
BOOL	VIDEO_Encode_Compress(VCodecHandle hEncoder,Video_Code_Frame &frame ) //NOLINT
{
	if( !hEncoder.pCodec || !hEncoder.pCodecDll )
		return FALSE;

	VideoCodecDllEx *pCodecDll = (VideoCodecDllEx *)hEncoder.pCodecDll;

	return pCodecDll->DllEncode(hEncoder.pCodec, &frame);
}

//--------------------
//停止压缩器
//hEncoder: 压缩器句柄
//--------------------
VOID	VIDEO_Encode_StopCompress(VCodecHandle &hEncoder)//NOLINT
{
	if (!hEncoder.pCodec || !hEncoder.pCodecDll)
		return;

	VideoCodecDllEx *pCodecDll = (VideoCodecDllEx *)hEncoder.pCodecDll;
	pCodecDll->DllEncClose(hEncoder.pCodec);
	hEncoder.pCodec = NULL;
	hEncoder.pCodecDll = NULL;

	VideoCodecPluginInfoEx codecInfo;
	pCodecDll->DllGetInfo(&codecInfo);
	if (codecInfo.bIsHWAccel) {
		g_hwAccController.DecreaseCurHWAccEncoder();
	}
	
}

VOID	VIDEO_Encode_SetMaxHWAccEncoder(DWORD nMax)
{
	g_hwAccController.SetMaxHWAccEncoder(nMax);
}

//---------------------
//创建解压器
//CodeType:解压器句柄
//biOut:解压之后格式
//Return: 解压器句柄
//---------------------
VCodecHandle VIDEO_Decode_StartDecompress(int nCodecId, const BITMAPINFOHEADER &biOut)
{
	VideoCodecDllEx *pCodecDll;
	HANDLE hCodec;
	VCodecHandle vHandle;
	VideoCodecPluginInfoEx vInfo;
	BOOL bHasHW = FALSE;
	BOOL isHW = FALSE;
	vHandle.pCodec = NULL;
	vHandle.pCodecDll = NULL;

	//if (nCodecId == VIDEO_CODEC_H264 || nCodecId == VIDEO_CODEC_H264PLUS)
	   // nCodecId = VIDEO_CODEC_H265;

	if (!g_videoCodecLoader.IsInited())
		return vHandle;

	//check HW acc control
	bHasHW = g_videoCodecLoader.HasHWAcc(nCodecId, FALSE);
	if (bHasHW && g_hwAccController.IncreaseCurHWAccDecoder()) {
		isHW = TRUE;
	}

	pCodecDll = g_videoCodecLoader.FindCodecById(nCodecId, FALSE, isHW);
	if (!pCodecDll)
		return vHandle;

	pCodecDll->DllGetInfo(&vInfo);
	if (vInfo.bIsHWAccel) {
		vHandle.pCodecDll = pCodecDll;
#ifdef _FS_OS_ANDROID
		pCodecDll->DllDecConfig(NULL, DEC_CONFIG_TYPE_JVM, g_hVideoModule);
#endif
		return vHandle;
	}

	hCodec = pCodecDll->DllDecOpen(&biOut);
	if (!hCodec)
		return vHandle;

	vHandle.pCodec = hCodec;
	vHandle.pCodecDll = (HANDLE)pCodecDll;

	return vHandle;

}


//-------------------
//解压图象
//hDecoder:解压器句柄
//frame:解压包
//Return: 是否解压成功
//--------------------
BOOL	VIDEO_Decode_Decompress(VCodecHandle hDecoder,Video_Code_Frame &frame )//NOLINT
{
	if (!hDecoder.pCodec || !hDecoder.pCodecDll)
		return FALSE;

	VideoCodecDllEx *pCodecDll = (VideoCodecDllEx *)hDecoder.pCodecDll;
	return pCodecDll->DllDecode(hDecoder.pCodec, &frame);

}

//---------------------
//停止压缩器
//hDecoder:解压器句柄
//--------------------
VOID	VIDEO_Decode_StopDecompress(VCodecHandle &hDecoder)//NOLINT
{
	ALOG("VIDEO_Decode_StopDecompress:: stop decoder.");
	if (!hDecoder.pCodecDll)
		return;
	VideoCodecDllEx *pCodecDll = (VideoCodecDllEx *)hDecoder.pCodecDll;
	pCodecDll->DllDecClose(hDecoder.pCodec);
	hDecoder.pCodec = NULL;
	hDecoder.pCodecDll = NULL;

	VideoCodecPluginInfoEx codecInfo;
	pCodecDll->DllGetInfo(&codecInfo);
	ALOG("VIDEO_Decode_StopDecompress:: get decoder info.");
	if (codecInfo.bIsHWAccel) {
		ALOG("VIDEO_Decode_StopDecompress:: decrease HW decoder count.");
		g_hwAccController.DecreaseCurHWAccDecoder();
	}

}
