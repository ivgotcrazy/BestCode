#ifndef __VIDEO_CODEC_PLUGIN_H
#define __VIDEO_CODEC_PLUGIN_H

#include "wbasetype.h"

typedef struct{

	DWORD	dwVersion;
	int		nCodecID;
	BOOL	bSupportEnc;
	BOOL	bSupportDec;
	WCHAR	wszName[64];

}VideoCodecPluginInfo;

//视频编码器动态库接口定义
typedef struct{
		
	VOID    ( STDAPICALLTYPE* DllGetInfo)		( VideoCodecPluginInfo *pInfo );
	
	HANDLE	( STDAPICALLTYPE* DllEncOpen )		( const Video_Encoder_Param* pParam,const BITMAPINFOHEADER *pbiIn );
		
	BOOL	( STDAPICALLTYPE* DllEncode )		( HANDLE hEncoder,Video_Code_Frame *pFrame );
		
	BOOL	( STDAPICALLTYPE* DllEncConfig )	( HANDLE hEncoder,const Video_Encoder_Param* pParam );

	VOID	( STDAPICALLTYPE* DllEncClose )		(HANDLE hEncoder);	
	
	HANDLE	( STDAPICALLTYPE* DllDecOpen )		( const BITMAPINFOHEADER *pbiOut );
	
	BOOL	( STDAPICALLTYPE* DllDecode )		(HANDLE hDecoder,Video_Code_Frame *pFrame );
	
	VOID	( STDAPICALLTYPE* DllDecClose )		(HANDLE hDecoder);	

}VideoCodecDll;

typedef HRESULT	(STDAPICALLTYPE*	LoadVideoCodec )		( VideoCodecDll *pDll );
typedef HRESULT	(STDAPICALLTYPE*	GetVideoCodecCount )	( int* pCount );
typedef HRESULT	(STDAPICALLTYPE*	GetVideoCodecInfo )		( int nIndex,VideoCodecDll* pDll );

#define FMVCODEC_FUNNAME			"FMVCodec"
#define FMVCODECGETCOUNT_FUNNAME	"FMVCodecGetCount"
#define FMVCODECGETINFO_FUNNAME		"FMVCodecGetInfo"

STDAPI	FMVCodec( VideoCodecDll *pDll );
STDAPI	FMVCodecGetCount( int* pCount );
STDAPI	FMVCodecGetInfo( int nIndex,VideoCodecDll* pDll );

//intel的解码和渲染库封装如下
//视频编码器动态库接口定义
typedef struct{

	HANDLE(STDAPICALLTYPE* DecodeOpen)		(HWND hWnd, PBYTE data, int size);

	VOID(STDAPICALLTYPE* DecodeClose)		(HANDLE hDecoder);
	
	VOID(STDAPICALLTYPE* EnableRender)		(HANDLE hDecoder, BOOL bRender);
	
	int(STDAPICALLTYPE* DecodeAndRender)		(HANDLE hDecoder, PBYTE data, int size, PBYTE pDecData, unsigned int* pDecSize);

	VOID(STDAPICALLTYPE* SetDisplayMode)		(HANDLE hDecoder, int mode);
	VOID(STDAPICALLTYPE* SetRenderWnd)			(HANDLE hDecoder, HWND hWnd);
	VOID(STDAPICALLTYPE* SetRenderPos)			(HANDLE hDecoder, const RECT& rect);

}IntelVideoDecDll;

//Extended Dll definition
//视频编码器动态库接口定义

typedef struct{

	DWORD	dwVersion;
	int		nCodecID;
	int     nModuleID;
	BOOL	bIsHWAccel;
	BOOL    bIsRenerBuiltIn;
	BOOL	bSupportEnc;
	BOOL	bSupportDec;
	WCHAR	wszName[64];

}VideoCodecPluginInfoEx;

typedef enum EncConfigType {
	ENC_CONFIG_TYPE_PARAM = 0,
	ENC_CONFIG_TYPE_JVM = 1,
}EncConfigType;


typedef enum DecConfigType {
	DEC_CONFIG_TYPE_BITMAPHEADER = 0,
	DEC_CONFIG_TYPE_JVM = 1,
}DecConfigType;

typedef struct{

	VOID(STDAPICALLTYPE* DllGetInfo)		(VideoCodecPluginInfoEx *pInfo);

	HANDLE(STDAPICALLTYPE* DllEncOpen)		(const Video_Encoder_Param* pParam, const BITMAPINFOHEADER *pbiIn);
	BOOL(STDAPICALLTYPE* DllEncode)		(HANDLE hEncoder, Video_Code_Frame *pFrame);
	BOOL(STDAPICALLTYPE* DllEncConfig)	(HANDLE hEncoder, EncConfigType ecType, const void* pParam);
	VOID(STDAPICALLTYPE* DllEncClose)		(HANDLE hEncoder);

	HANDLE(STDAPICALLTYPE* DllDecOpen)		(const BITMAPINFOHEADER *pbiOut);
	BOOL(STDAPICALLTYPE* DllDecode)		(HANDLE hDecoder, Video_Code_Frame *pFrame);
	VOID(STDAPICALLTYPE* DllDecClose)		(HANDLE hDecoder);

	HANDLE(STDAPICALLTYPE* DllDecOpen2)		(HWND hWnd, PBYTE data, int size);
	VOID(STDAPICALLTYPE* DllEnableRender)		(HANDLE hDecoder, BOOL bRender);
	int(STDAPICALLTYPE* DllDecodeAndRender)		(HANDLE hDecoder, PBYTE data, int size, 
												PBYTE pDecData, unsigned int* pDecSize);
	VOID(STDAPICALLTYPE* DllSetDisplayMode)		(HANDLE hDecoder, int mode);
	VOID(STDAPICALLTYPE* DllSetRenderWnd)			(HANDLE hDecoder, HWND hWnd);
	VOID(STDAPICALLTYPE* DllSetRenderPos)			(HANDLE hDecoder, const RECT& rect);
	VOID(STDAPICALLTYPE* DllDecConfig) (HANDLE hDecoder, DecConfigType dcType, const void *pParam);

	BOOL m_bInited;

}VideoCodecDllEx;

typedef HRESULT(STDAPICALLTYPE*	LoadVideoCodecEx)		(VideoCodecDllEx *pDll);
typedef HRESULT(STDAPICALLTYPE*	GetVideoCodecCountEx)	(int* pCount);
typedef HRESULT(STDAPICALLTYPE*	GetVideoCodecInfoEx)		(int nIndex, VideoCodecDllEx* pDll);

#define FMVCODEC_FUNNAMEEX			"FMVCodecEx"
#define FMVCODECGETCOUNT_FUNNAMEEX	"FMVCodecGetCountEx"
#define FMVCODECGETINFO_FUNNAMEEX		"FMVCodecGetInfoEx"

STDAPI	FMVCodecEx(VideoCodecDllEx *pDll);
STDAPI	FMVCodecGetCountEx(int* pCount);
STDAPI	FMVCodecGetInfoEx(int nIndex, VideoCodecDllEx* pDll);
